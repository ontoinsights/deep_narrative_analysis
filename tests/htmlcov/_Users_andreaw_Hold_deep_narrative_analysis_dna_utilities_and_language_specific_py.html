144def add_to_dictionary_values(dictionary: dict, key: str, value, value_type): 

145 """ 

146 Add the value (cast using the value_type parameter) for the specified key in the dictionary, 

147 if that key exists, or create a new dictionary entry for the key (where the value is a list). 

148 

149 :param dictionary: The dictionary to be updated 

150 :param key: String holding the dictionary key 

151 :param value: The value to be added to the current list of values of the dictionary entry, or 

152 a new list is created with the value 

153 :param value_type: The type of the value 

154 :return: None (Dictionary is updated) 

155 """ 

156 values = [] 

157 if key in dictionary: 

158 values = dictionary[key] 

159 values.append(value_type(value)) 

160 dictionary[key] = values

176def check_name_gender(name_str: str) -> str: 

177 """ 

178 Determines if a reference is to male/female name. Returns a type of either <FEMALE/MALE>SINGPERSON 

179 (if the gender can be determined) or 'SINGPERSON'. 

180 

181 :param name_str: The string representing the noun or proper name 

182 :return: A tuple of 2 strings representing the entity's text and type (with gender if known) 

183 """ 

184 gender = empty_string 

185 if space in name_str: 

186 names = name_str.split() 

187 else: 

188 names = [name_str] 

189 for name in names: 

190 gender = 'FEMALE' if name in female_names else ('MALE' if name in male_names else empty_string) 

191 if gender: 

192 break 

193 return f'{gender}SINGPERSON' if gender else 'SINGPERSON'
