66def get_birth_family_details(narrative: str) -> (list, list, dict): 

67 """ 

68 Use spaCy Dependency Parsing and rules-based matching to get the birth date and place details, 

69 and family member names from a narrative. 

70 

71 :param narrative: String holding the narrative 

72 :return: Two lists where the first contains the tokens related to date and the second 

73 contains the tokens related to location, and a dictionary containing the names 

74 of family members and their relationship to the narrator/subject 

75 """ 

76 logging.info('Getting birth and family data from narrative') 

77 # Analyze the text using spaCy and get matches to the 'born' and 'member' rules 

78 doc = nlp(narrative.replace('\n', '')) 

79 matches = matcher(doc) 

80 # Get born on and born in details from the matches 

81 born_on_date = set() 

82 born_in_place = set() 

83 family_dict = dict() 

84 # Each token_id corresponds to one pattern dict 

85 for match in matches: 

86 match_id, token_ids = match # Indicates which pattern is matched and the specific tokens 

87 # Get the string representation 

88 string_id = nlp.vocab.strings[match_id] 

89 if string_id == 'born_date': 

90 born_on_date.add(doc[token_ids[2]].text) 

91 elif string_id == 'born_place': 

92 born_in_place.add(doc[token_ids[2]].text) 

93 elif string_id == 'family_member_name': 

94 family_dict[doc[token_ids[1]].text] = doc[token_ids[0]].text # Key = proper name, Value = relationship 

95 # Verify that dates actually have a year 

96 found_year = any([value for value in born_on_date if (value.isnumeric() and int(value) > 1000)]) 

97 if not found_year: 

98 born_on_date = set() 

99 return list(born_on_date), list(born_in_place), family_dict

102def get_synonym(text: str) -> list: 

103 """ 

104 Get the synonyms/lemma of the input text. 

105 

106 :param text: String to be lemmatized 

107 :return An array holding the synonyms/lemmas of the input text 

108 """ 

109 words = [] 

110 syns = wn.synsets(text) 

111 for syn in syns: 

112 text = str(syn) 

113 words.append(text[text.index("'") + 1:text.index('.')]) 

114 return words

117def get_named_entity_in_string(text: str) -> (str, str): 

118 """ 

119 Returns information on any Named Entity that are Agents (people, organizations, geopolitical 

120 entities, ...) or Events in the input text. 

121 

122 :param text: The text to be parsed 

123 :return Two strings - the first is the named entity's text and the second string is its 

124 type mapped to the DNA Agent sub-classing hierarchy or to :EventAndState; If the input text 

125 does not contain any Named Entities, then two empty strings are returned 

126 """ 

127 doc = nlp(text) 

128 for ent in doc.ents: 

129 if ent.label_ in ner_types: 

130 return ent.text, ner_dict[ent.label_] 

131 return empty_string, empty_string

134def get_noun(text: str, get_first_occurrence: bool) -> str: 

135 """ 

136 Creates a spacy Doc from the input text and returns the first (or last) noun found. 

137 

138 :param text: The text to parse 

139 :param get_first_occurrence: Boolean indicating that the first noun is returned (if True); 

140 otherwise, the last noun is returned 

141 :return: The first or last (depending on whether the first variable is True or False) noun 

142 in the input text, or an empty string if no noun is found 

143 """ 

144 doc = nlp(text) 

145 word = empty_string 

146 for token in doc: 

147 if token.pos_ in ('NOUN', 'PROPN'): 

148 if token.dep_ == 'compound': 

149 continue 

150 word = token.text 

151 if get_first_occurrence: 

152 break 

153 return word

156def get_nouns_verbs(sentences: str) -> (dict, dict): 

157 """ 

158 Parses all the sentences in the input parameter to return counts of each distinct 

159 noun and verb. 

160 

161 :param sentences: String with the text of one or more sentences 

162 :return: Two dictionaries of the counts of the noun/verb lemmas where the noun 

163 dictionary is returned first 

164 """ 

165 noun_dict = dict() 

166 verb_dict = dict() 

167 doc = nlp(sentences) 

168 for sentence in doc.sents: 

169 for token in sentence: 

170 if token.pos_ in ('NOUN', 'PROPN'): 

171 update_dictionary_count(noun_dict, token.lemma_.lower()) 

172 if token.pos_ == 'VERB': 

173 update_dictionary_count(verb_dict, token.lemma_.lower()) 

174 sorted_nouns = dict(sorted(noun_dict.items(), key=lambda item: item[1], reverse=True)) 

175 sorted_verbs = dict(sorted(verb_dict.items(), key=lambda item: item[1], reverse=True)) 

176 return sorted_nouns, sorted_verbs

179def get_proper_nouns(text: str) -> str: 

180 """ 

181 Extract a string consisting of the proper nouns in a text string. For example, "New York City ghetto" 

182 would return "New York City". 

183 

184 :param text: The text string to be parsed 

185 :return A string of the proper nouns 

186 """ 

187 phrase = nlp(text) 

188 proper_nouns = [] 

189 for token in phrase: 

190 noun_type = token.morph.get('NounType') 

191 if noun_type and noun_type[0] == 'Prop': # Have a proper noun 

192 proper_nouns.append(token.text) 

193 if proper_nouns: 

194 return ' '.join(proper_nouns) 

195 else: 

196 return empty_string

199def get_sentence_sentiment(sentence: str) -> float: 

200 """ 

201 Use TextBlob to get sentence polarity/sentiment. 

202 

203 :param sentence: The sentence to be analyzed. 

204 :return The polarity (-1 for negative, 1 for positive) of the sentence 

205 """ 

206 blob = TextBlob(sentence) 

207 # TextBlob's sentiment property returns a namedtuple of the form, (polarity, subjectivity). 

208 # The polarity score is a float within the range [-1.0, 1.0] - negative to positive. 

209 # The subjectivity is a float within the range [0.0, 1.0] - 0.0 = very objective and 1.0 = very subjective. 

210 return blob.sentiment.polarity

213def get_time_details(phrase: str) -> (int, str): 

214 """ 

215 For a phrase (such as 'a year later'), get the time increment and number of increments. 

216 

217 :param phrase: String to be processed 

218 :return A tuple of the number of increments and the increment length (for example, 'year') 

219 """ 

220 number = 0 

221 increment = '' 

222 nlp_date = nlp(phrase) 

223 for token in nlp_date: 

224 if token.dep_ == 'det': # Assumes 'a' or 'the' means 1 

225 number = 1 

226 elif token.dep_ == 'nummod': 

227 if token.text.isnumeric(): 

228 number = int(token.text) 

229 else: 

230 number = w2n.word_to_num(token.lemma_) 

231 elif token.dep_ == 'npadvmod': 

232 increment = token.text 

233 return number, increment

236def parse_narrative(narr_text: str, gender: str, family_dict: dict) -> list: 

237 """ 

238 Creates a spacy Doc from the input text, splits sentences by conjunctions into clauses with 

239 their own subjects/verbs, and then parses each of the resulting sentences to create a 

240 dictionary holding the subject/verb/object/preposition/... details. Each of the sentence 

241 dictionaries are added to an array, which is returned. 

242 

243 :param narr_text: The narrative text 

244 :param gender: Either an empty string or one of the values, AGENDER, BIGENDER, FEMALE or 

245 MALE - indicating the gender of the narrator 

246 :param family_dict: A dictionary containing the names of family members and their 

247 relationship to the narrator/subject 

248 :return: An array of dictionaries holding the details of each sentence (after splitting) 

249 """ 

250 doc = nlp(narr_text) 

251 split_sentences = [] 

252 sentence_dicts = [] 

253 # Split sentences by conjunctions 

254 for sentence in doc.sents: 

255 if sentence.text.startswith(new_line): 

256 # When using a text editor on the narrative, the new line may be part of the 'next' line of text 

257 split_sentences.append(nlp("New line")) 

258 sent_text = sentence.text.replace(new_line, '') 

259 if not sent_text: 

260 continue 

261 else: 

262 sent_text = sentence.text 

263 # Change conjunctions into new sentences 

264 for sent in split_clauses(sent_text, nlp): 

265 sent_nlp = nlp(sent) 

266 # Determine the spans of individual nouns and noun chunks 

267 spans = list(sent_nlp.ents) + list(sent_nlp.noun_chunks) 

268 spans = spacy.util.filter_spans(spans) 

269 # Reset the sentence parse to maintain chunks 

270 with sent_nlp.retokenize() as retokenizer: 

271 [retokenizer.merge(span, attrs={'tag': span.root.tag, 

272 'dep': span.root.dep}) for span in spans] 

273 # Store new sentence details 

274 split_sentences.append(sent_nlp) 

275 # Get the details of each sentence 

276 sentence_offset = 1 

277 for sentence in split_sentences: 

278 revised = _replace_words(sentence.text) 

279 extract_dictionary_details(revised, sentence_dicts, nlp, gender, family_dict, sentence_offset) 

280 sentence_offset += 1 

281 return sentence_dicts

285def _replace_words(sentence: str) -> str: 

286 """ 

287 Replace specific phrases/terms with simpler ones - such as replacing 'as well as' with the word, 'and'. 

288 

289 :param sentence: The sentence whose text should be updated 

290 :return The updated sentence 

291 """ 

292 for key, value in replacement_words.items(): 

293 if key in sentence: 

294 sentence = sentence.replace(key, value) 

295 if key.title() in sentence: 

296 sentence = sentence.replace(key.title(), value.title()) 

297 return sentence
