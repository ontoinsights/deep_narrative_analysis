26def _get_geonames_alt_names(root: etree.Element) -> (list, str): 

27 """ 

28 Retrieve all alternativeName elements (there is more than 1) having the specified language(s) 

29 in the XML tree. And, return the Wikipedia link for the entity, if available. 

30 

31 :param root: The root element of the XML tree 

32 :return: A tuple consisting of an array holding the string values of the alternative names 

33 or an empty array (if not defined), and a string holding the Wikipedia link for 

34 additional details 

35 """ 

36 # TODO: Language specific results 

37 names = set() 

38 link = empty_string 

39 elems = root.findall('./geoname/alternateName[@lang]') 

40 for elem in elems: 

41 lang = elem.get('lang') 

42 access = True if ('isPreferredName' in elem.attrib or 'isShortName' in elem.attrib) else False 

43 if lang == 'en' or lang == 'abbr' or access: 

44 names.add(elem.text) 

45 if lang == 'link' and 'en.wikipedia' in elem.text: 

46 link = elem.text 

47 return list(names), link

50def _get_geonames_response(request: str, loc_str: str) -> Union[etree.Element, None]: 

51 """ 

52 Send and process a query to the GeoNames API. 

53 

54 :param request: A string holding the query request. 

55 :param loc_str: A string holding the text identifying the location. 

56 :return: The GeoNames response 

57 """ 

58 try: 

59 response = requests.get(request) 

60 except requests.exceptions.RequestException as e: 

61 logging.error(f'External source query error: Query={request} and Exception={str(e)}') 

62 return None 

63 orig_root = etree.fromstring(response.content) 

64 toponym_name = _get_xml_value('./geoname/toponymName', orig_root) 

65 ascii_name = _get_xml_value('./geoname/asciiName', orig_root) 

66 if toponym_name.lower() == loc_str.lower(): 

67 return orig_root 

68 # Try again, forcing a match of fcode = ADM1 - for ex, New York returns New York City, not the state 

69 if 'fcode=' not in request: 

70 request += '&fcode=ADM1' 

71 root = _get_geonames_response(request, loc_str) 

72 if root: 

73 return root 

74 # Relax the match to the toponym or ascii name containing the location string 

75 if loc_str.lower() in toponym_name.lower() or loc_str.lower() in ascii_name.lower(): 

76 return orig_root 

77 return None

80def _get_wikidata_time(query: str, is_start: bool) -> str: 

81 """ 

82 Send the query, query_wikidata_time, to WDQS asking for the start and end times of an 

83 event identified by text (in the mwapi search parameters). 

84 

85 :param query: The query to be sent with the wikidata ID already in the string 

86 :param is_start: Boolean indicating that the start time is requested (if true); Otherwise, 

87 the end time is requested 

88 :return: A string holding the start or end time if defined; Otherwise, an empty string 

89 """ 

90 if is_start: 

91 query = query.replace('propTime', 'P580') 

92 else: 

93 query = query.replace('propTime', 'P582') 

94 try: 

95 response = requests.get(f'https://query.wikidata.org/sparql?format=json&query={query}').json() 

96 except requests.exceptions.RequestException as e: 

97 request = f'https://query.wikidata.org/sparql?format=json&query={query}' 

98 logging.error(f'External source query error: Query={request} and Exception={str(e)}') 

99 return empty_string 

100 results = [] 

101 if 'results' in response and 'bindings' in response['results']: 

102 results = response['results']['bindings'] 

103 if len(results) > 0: 

104 return results[0]['time']['value'] 

105 else: 

106 return empty_string

109def _get_xml_value(xpath: str, root: etree.Element) -> str: 

110 """ 

111 Use the input xpath string to access specific elements in an XML tree. 

112 

113 :param xpath: String identifying the path to the element to be retrieved 

114 :param root: The root element of the XML tree 

115 :return: String representing the value of the specified element or an empty string 

116 (if not defined) 

117 """ 

118 elems = root.findall(xpath) 

119 if elems: 

120 return elems[0].text 

121 else: 

122 return empty_string

125def check_wordnet(word: str) -> str: 

126 """ 

127 Check the most common wordnet definition (the first one) for an unmapped term. 

128 

129 :param word: The term to be mapped 

130 :return: String holding the head word from the definition of the first synset for the term, or 

131 an empty string 

132 """ 

133 if space in word: 

134 single_word = get_head_word(word)[0] 

135 else: 

136 single_word = word 

137 synsets = wn.synsets(single_word, pos=wn.NOUN) 

138 if synsets: 

139 return get_head_word(synsets[0].definition())[0] 

140 else: 

141 return empty_string

144def get_event_details_from_wikidata(event_text: str) -> (str, str, str, list): 

145 """ 

146 Get the start and end times and alternate names of an event, if it is known to Wikidata. 

147 

148 :param event_text: Text defining the event 

149 :return: A tuple consisting of strings holding the Wikipedia description of the event, 

150 strings identifying the start and end times and an array of labels/alternate names 

151 """ 

152 logging.info(f'Getting Wikidata event details for {event_text}') 

153 start_time = empty_string 

154 end_time = empty_string 

155 labels = [] 

156 wiki_details = get_wikipedia_description(event_text.replace(space, '_')) 

157 if wiki_details and 'See the web site' not in wiki_details: 

158 wikidata_id = wiki_details.split('wikibase_item: ')[1].split(')')[0] 

159 labels = get_wikidata_labels(wikidata_id) 

160 time_query = query_wikidata_time.replace('?item', f'wd:{wikidata_id}') 

161 start_time = _get_wikidata_time(time_query, True) 

162 end_time = _get_wikidata_time(time_query, False) 

163 return wiki_details, start_time, end_time, labels

166def get_geonames_location(loc_text: str) -> (str, str, int, list, str): 

167 """ 

168 Get the type of location from its text as well as its country and administrative level (if relevant). 

169 

170 :param loc_text: Location text 

171 :return: A tuple holding the location's class mapping (for geonames, it is always a single string), 

172 country name (or an empty string or None), an administrative level (if 0, then admin level 

173 is not applicable) or GeoNames ID (for a Country), a list of alternate names and a link 

174 to a Wikipedia page for the location 

175 """ 

176 logging.info(f'Getting geonames details for {loc_text}') 

177 # Future: May need to add sleep to meet geonames timing requirements 

178 name_startswith = False 

179 if ',' in loc_text: 

180 request = f'http://api.geonames.org/search?q={loc_text.lower().replace(space, "+").replace(",", "+")}' \ 

181 f'&style=full&maxRows=1&username={geonamesUser}' 

182 elif space in loc_text: 

183 request = f'http://api.geonames.org/search?q={loc_text.lower().replace(space, "+")}' \ 

184 f'&name_equals={loc_text.lower().replace(space, "+")}&style=full&maxRows=1&username={geonamesUser}' 

185 else: 

186 name_startswith = True 

187 request = f'http://api.geonames.org/search?q={loc_text.lower()}&name_startsWith={loc_text.lower()}' \ 

188 f'&style=full&maxRows=1&username={geonamesUser}' 

189 root = _get_geonames_response(request, loc_text) 

190 if root is None: 

191 if name_startswith: 

192 # Try less restrictive search 

193 request = f'http://api.geonames.org/search?q={loc_text.lower()}' \ 

194 f'&style=full&maxRows=1&username={geonamesUser}' 

195 root = _get_geonames_response(request, loc_text) 

196 if root is None: 

197 return empty_string, empty_string, 0, [], empty_string 

198 country = _get_xml_value('./geoname/countryName', root) 

199 feature = _get_xml_value('./geoname/fcl', root) 

200 fcode = _get_xml_value('./geoname/fcode', root) 

201 ascii_name = _get_xml_value('./geoname/asciiName', root) 

202 alt_names, wiki_link = _get_geonames_alt_names(root) # TODO: Language specific results 

203 if not country and not feature: 

204 return empty_string, empty_string, 0, [], empty_string 

205 # Process the alternate names 

206 if loc_text not in alt_names: 

207 alt_names.append(loc_text) 

208 if ascii_name not in alt_names: 

209 alt_names.append(ascii_name) 

210 # Process the rest of the GeoNames info 

211 admin_level = 0 

212 class_type = ':PopulatedPlace' 

213 if feature == 'A': # Administrative area 

214 if fcode.startswith('ADM'): 

215 class_type += '+:AdministrativeDivision' 

216 # Get administrative level 

217 if any([i for i in fcode if i.isdigit()]): 

218 admin_level = [int(i) for i in fcode if i.isdigit()][0] 

219 elif fcode.startswith('PCL'): # Some kind of political entity, assuming for now that it is a country 

220 class_type = ":Country" 

221 country = empty_string 

222 elif fcode.startswith('L') or fcode.startswith('Z'): 

223 # Leased area usually for military installations or a zone, such as a demilitarized zone 

224 class_type = ':DesignatedArea' 

225 elif feature == 'P' and fcode.startswith('PPLA'): # PopulatedPlace that is an administrative division 

226 class_type += '+:AdministrativeDivision' 

227 # Get administrative level 

228 admin_levels = [int(i) for i in fcode if i.isdigit()] 

229 if admin_levels: 

230 admin_level = admin_levels[0] 

231 else: 

232 admin_level = 1 

233 elif feature in ('H', 'L', 'R', 'S', 'T', 'U', 'V'): 

234 class_type = geocodes_mapping[feature] 

235 return class_type, country, admin_level, alt_names, wiki_link

238def get_wikidata_labels(wikidata_id: str) -> list: 

239 """ 

240 Get the labels (labels and alt_names) for a Wikidata item. 

241 

242 :param wikidata_id: String holding the Q ID of the Wikidata item 

243 :return: An array of strings that are the rdfs:label and skos:alt_names of the item 

244 """ 

245 labels = [] 

246 query_labels = query_wikidata_labels.replace('?item', f'wd:{wikidata_id}') 

247 try: 

248 response = requests.get( 

249 f'https://query.wikidata.org/sparql?format=json&query={query_labels}').json() 

250 except requests.exceptions.RequestException as e: 

251 request = f'https://query.wikidata.org/sparql?format=json&query={query_labels}' 

252 logging.error(f'External source query error: Query={request} and Exception={str(e)}') 

253 return labels 

254 if 'results' in response and 'bindings' in response['results']: 

255 results = response['results']['bindings'] 

256 for result in results: 

257 labels.append(result['label']['value'].replace('"', "'")) 

258 return labels

261def get_wikipedia_description(noun: str, explicit_link: str = empty_string) -> str: 

262 """ 

263 Get the first paragraph of the Wikipedia web page for the specified organization, group, ... 

264 

265 :param noun: String holding the organization/group name 

266 :param explicit_link: A link retrieved from another source (such as GeoNames) to a Wikipedia 

267 entry for the noun 

268 :return: String that is the first paragraph of the Wikipedia page (if the org/group is found), 

269 information on a disambiguation page or an empty string 

270 """ 

271 logging.info(f'Getting wikipedia details for {noun}') 

272 noun_underscore = noun.replace(space, '_') 

273 try: 

274 resp = requests.get(f'https://en.wikipedia.org/api/rest_v1/page/summary/' 

275 f'{explicit_link.split("/")[-1] if explicit_link else noun_underscore}') 

276 except requests.exceptions.RequestException as e: 

277 request = f'https://en.wikipedia.org/api/rest_v1/page/summary/' \ 

278 f'{explicit_link.split("/")[-1] if explicit_link else noun_underscore}' 

279 logging.error(f'External source query error: Query={request} and Exception={str(e)}') 

280 return empty_string 

281 wikipedia = resp.json() 

282 if 'type' in wikipedia and wikipedia['type'] == 'disambiguation': 

283 return f'Needs disambiguation; See the web site, https://en.wikipedia.org/wiki/{noun_underscore}' 

284 if 'extract' in wikipedia: 

285 extract_text = wikipedia['extract'].replace('"', "'").replace('\xa0', space).replace('\n', space).\ 

286 encode('ASCII', errors='replace').decode('utf-8') 

287 wiki_text = f"'{extract_text}'" 

288 return f'From Wikipedia (wikibase_item: {wikipedia["wikibase_item"]}): {wiki_text}' 

289 return empty_string
