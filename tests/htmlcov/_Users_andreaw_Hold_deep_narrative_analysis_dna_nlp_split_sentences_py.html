16def _chunk_quotations(text_str: str) -> list: 

17 """ 

18 Given a sentence text that includes 'Quotation#', chunk that text into individual clauses for 

19 each Quotation reference and any remaining text. 

20 

21 :param text_str: The text that contains at least one 'Quotation#' reference 

22 :return: A list of the text chunks, splitting each 'Quotation#' reference into its own chunk 

23 and including each intervening text as a chunk 

24 """ 

25 quotation_refs = re.findall(r'Quotation[0-9]+', text_str) # Really only care about the first ref 

26 chunks = [] 

27 quote_index = text_str.index(quotation_refs[0]) 

28 if quote_index > 0: 

29 chunk_text = text_str[0:quote_index].strip() 

30 chunks.append(chunk_text[0:-1] if chunk_text.endswith('.') else chunk_text) 

31 text_str = text_str[quote_index:] 

32 chunks.append(quotation_refs[0]) 

33 text_str = text_str.replace(quotation_refs[0], empty_string) 

34 if not text_str or text_str == space: 

35 return chunks 

36 text_str = text_str.strip() 

37 if 'Quotation' not in text_str: 

38 chunks.append(text_str) 

39 return chunks 

40 chunks.extend(_chunk_quotations(text_str)) # Recursively deal with the rest of the sentence 

41 return chunks

44def _get_chunks(verb: Token, connector: Union[Token, None], chunk_sentence: Span, processing_type: str, 

45 subjs: Union[list, None]) -> list: 

46 """ 

47 Given (1) a conjunctive or clausal phrase for a verb, (2) an optional connector token relating the 

48 phrase to the verb (such as 'and', 'but', ...) and (3) the sentence that contains these tokens, determine 

49 if the sentence can be separated into clauses. The sentence is separated if the clause has its own 

50 subject and verb. 

51 

52 Note that this algorithm (seen/unseen_chunk) is taken from the StackOverflow response at 

53 https://stackoverflow.com/questions/65227103/clause-extraction-long-sentence-segmentation-in-python. 

54 

55 :param verb: The token of the verb with the conjunctive or clausal phrase 

56 :param connector: Token of the connector between the root and conjunctive or clausal verb, 

57 or None (which may occur if this is a clausal complement - e.g., "He said 

58 Joe is ill.") 

59 :param chunk_sentence: A Span representing the parsed sentence 

60 :param processing_type: A string indicating whether this is for a conjunctive verb ('conj'), 

61 adverbial clause verb ('advcl'), clausal complement ('comp') or 

62 relative clause ('relcl') 

63 :param subjs: An array holding the tokens of any subjects or expletives (it is only needed when 

64 processing conjunctions 

65 :return: A list of the text of the clauses of the sentence (split if indicated by the 

66 logic below) or just the original sentence returned 

67 """ 

68 if subjs is None: 

69 subjs = [] 

70 chunks = [] 

71 subj_texts = [] 

72 # Does the other clause's verb have a subject ('nsubj'/'nsubjpass'/'expl' (e.g., 'there')) in the dep parse? 

73 subj2 = [subj for subj in verb.children if ('subj' in subj.dep_ or subj.dep_ == 'expl') and 

74 ((subj.pos_ == 'PRON' and subj.morph.get('PronType') != ['Dem']) or subj.pos_ != 'PRON')] 

75 if not subj2 and (processing_type in ('conj', 'advcl')): 

76 # Attach the root verb's subject(s) to the other verb 

77 for subj in subjs: 

78 for subtree in subj.subtree: 

79 subj_texts.append(subtree.text) 

80 if len(subj2) or subj_texts: 

81 # Yes ... Separate clauses 

82 # Get the tokens in sentence related to the 'other' verb's subtree 

83 seen = [ww for ww in verb.subtree if ( 

84 (ww.pos_ == 'PUNCT' and ww.lemma_ == '?') or 

85 not ((ww.dep_ == 'dobj' and ww.pos_ == 'PRON') or ww.pos_ == 'PUNCT'))] 

86 seen_chunk = ' '.join([ww.text for ww in seen]).strip() 

87 seen_chunk = seen_chunk.replace(' .', empty_string) if seen_chunk.endswith(' .') else seen_chunk 

88 unseen = [ww for ww in chunk_sentence if ww not in seen and 

89 ((ww.pos_ == 'PUNCT' and ww.lemma_ == '?') or ww.pos_ != 'PUNCT')] 

90 unseen_chunk = ' '.join([ww.text for ww in unseen]).strip() 

91 unseen_chunk = unseen_chunk.replace(' .', empty_string) if unseen_chunk.endswith(' .') else unseen_chunk 

92 if connector: 

93 seen_chunk = _remove_connector_text(seen_chunk, connector.text) 

94 unseen_chunk = _remove_connector_text(unseen_chunk, connector.text) 

95 if processing_type == 'conj': 

96 # Store the unseen and seen clauses in that order 

97 chunks.extend( 

98 _store_chunks_in_order(unseen_chunk, 

99 f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk)) 

100 elif processing_type == 'advcl': 

101 if connector.text.lower() in later_connectors: # Store unseen_chunk first 

102 if 'eventually' in unseen_chunk.lower() or 'afterwards' in unseen_chunk.lower() or \ 

103 'finally' in unseen_chunk.lower(): # Keywords override ordering 

104 chunks.extend( 

105 _store_chunks_in_order( 

106 f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk, unseen_chunk)) 

107 else: 

108 chunks.extend( 

109 _store_chunks_in_order( 

110 unseen_chunk, f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk)) 

111 else: # Store seen_chunk first 

112 if connector.text.lower() in special_marks: 

113 seen_chunk += f'$&{connector.text.lower()}' 

114 chunks.extend( 

115 _store_chunks_in_order( 

116 f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk, unseen_chunk)) 

117 else: 

118 chunks.extend( 

119 _store_chunks_in_order( 

120 unseen_chunk, f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk)) 

121 if len(chunks) > 0: 

122 return chunks 

123 else: 

124 return [chunk_sentence.text[0:-1] if chunk_sentence.text.endswith('.') else chunk_sentence.text]

127def _remove_connector_text(chunk_text: str, connector_text: str) -> str: 

128 """ 

129 Returns first string with the second string removed from its start or end. 

130 

131 :param chunk_text: String to be updated 

132 :param connector_text: String to be removed from the beginning or end of the chunk 

133 :return: Updated 'chunk' string 

134 """ 

135 if chunk_text.startswith(f'{connector_text} ') or chunk_text.startswith(f'{connector_text.title()} '): 

136 chunk_text = chunk_text[len(connector_text) + 1:] 

137 elif chunk_text.startswith(f' {connector_text} '): 

138 chunk_text = chunk_text[len(connector_text) + 2:] 

139 elif chunk_text.endswith(f' {connector_text} '): 

140 chunk_text = chunk_text[0:(len(connector_text) + 2) * -1] 

141 elif chunk_text.endswith(f' {connector_text}'): 

142 chunk_text = chunk_text[0:(len(connector_text) + 1) * -1] 

143 return chunk_text

146def _split_advcl_clauses(sentence: str, nlp: Language) -> list: 

147 """ 

148 Split the adverbial clauses of a sentence into separate sentences, if the clauses have their 

149 own subject and verb. 

150 

151 :param sentence: The sentence which is analyzed 

152 :param nlp: A spacy Language model 

153 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

154 returned, not the spacy tokens) or the original sentence if there is no advcl verb 

155 or if the connector is not one of the cause_ or effect_connectors 

156 """ 

157 sent_span = next(nlp(sentence).sents) # There should only be 1 sentence for each function call 

158 subjs = [child for child in sent_span.root.children if 'subj' in child.dep_] 

159 advcl_verbs = [child for child in sent_span.root.children if child.dep_ == 'advcl'] 

160 new_chunks = [] 

161 for advcl_verb in advcl_verbs: 

162 connectors = [conn for conn in advcl_verb.children if conn.dep_ in ('advmod', 'mark')] 

163 # Process the verb and the first connector (there should only be 1, but there may be 0) 

164 if connectors: 

165 chunks = _get_chunks(advcl_verb, connectors[0], sent_span, 'advcl', subjs) 

166 else: 

167 chunks = _get_chunks(advcl_verb, None, sent_span, 'advcl', subjs) 

168 revised_chunks = [] 

169 if connectors: 

170 # Remove the connector from the middle of the text of the sentence 

171 revised_chunks.extend([chunk.replace(f' {connectors[0]} ', space) for chunk in chunks]) 

172 else: 

173 revised_chunks = chunks 

174 new_chunks.extend(revised_chunks) 

175 return new_chunks if new_chunks else [sentence]

178def _split_by_conjunctions(sentence: str, nlp: Language) -> list: 

179 """ 

180 Split the clauses of a sentence into separate sentences, if the verbs are connected by a coordinating 

181 conjunction (for, and, but, or, nor, yet, so). For example,"Mary and John walked to the store." would 

182 NOT be split, but "Mary biked to the market and John walked to the store." should be split. 

183 

184 :param sentence: The sentence which is analyzed 

185 :param nlp: A spacy Language model 

186 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

187 returned, not the spacy tokens) or the original sentence if there are no 

188 conjunctive verbs 

189 """ 

190 sent_span = next(nlp(sentence).sents) # There should only be 1 sentence for each function call 

191 new_sents = [] 

192 conj_verbs = [child for child in sent_span.root.children if child.dep_ == 'conj'] 

193 connectors = [conn for conn in sent_span.root.children if conn.dep_ == 'cc'] 

194 if conj_verbs and len(conj_verbs) == len(connectors): 

195 subjects = [child for child in sent_span.root.children if 'subj' in child.dep_] 

196 expls = [child for child in sent_span.root.children if child.dep_ == 'expl'] 

197 if expls: 

198 subjects.extend(expls) 

199 # Process the first 'chunk' and then return - Subsequent iterations will process the complete text 

200 chunks = _get_chunks(conj_verbs[0], connectors[0], sent_span, 'conj', subjects) 

201 if len(chunks) > 1: 

202 for chunk in chunks: 

203 new_sents.extend(_split_by_conjunctions(chunk, nlp)) 

204 return new_sents 

205 else: 

206 return [chunks[0]] 

207 else: 

208 return [sentence[0:-1] if sentence.endswith('.') else sentence]

211def _split_complement_clauses(sentence: str, nlp: Language) -> list: 

212 """ 

213 Split the clausal complements of a sentence into separate sentences, if the clauses have their 

214 own subject and verb. 

215 

216 :param sentence: The sentence which is analyzed 

217 :param nlp: A spacy Language model 

218 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

219 returned, not the spacy tokens) or the original sentence if there is no ccomp verb 

220 """ 

221 sent_span = next(nlp(sentence).sents) # There should only be 1 sentence for each function call 

222 comp_verbs = [child for child in sent_span.root.children if child.dep_ == 'ccomp'] 

223 for comp_verb in comp_verbs: 

224 connectors = [conn for conn in comp_verb.children if conn.dep_ in ('advmod', 'mark')] 

225 # Process the verb and the first connector (there may not be any) 

226 if connectors: 

227 return _get_chunks(comp_verb, connectors[0], sent_span, 'comp', None) 

228 else: 

229 return _get_chunks(comp_verb, None, sent_span, 'comp', None) 

230 return [sentence]

233def _split_relcl_clauses(clause: str, nlp: Language) -> list: 

234 """ 

235 Split the relative clauses of a sentence into separate sentences, if the clauses have their 

236 own subject and verb. 

237 

238 :param clause: The sentence which is analyzed 

239 :param nlp: A spacy Language model 

240 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

241 returned, not the spacy tokens) or the original sentence if there is no relcl verb 

242 """ 

243 clause_span = next(nlp(clause).sents) # There should only be 1 sentence for each function call 

244 new_clauses = [] 

245 relcl_dict = dict() 

246 for token in clause_span: 

247 if 'subj' in token.dep_ or 'obj' in token.dep_: 

248 relcls = [child for child in token.children if child.dep_ == 'relcl' and child.pos_ in ('VERB', 'AUX')] 

249 if relcls: 

250 relcl_dict[token] = relcls 

251 for noun_token, relcls in relcl_dict.items(): 

252 for relcl in relcls: 

253 noun_details = [ww.text for ww in noun_token.subtree] 

254 noun_text = ' '.join(noun_details) 

255 seen = [ww for ww in relcl.subtree] 

256 seen_chunk = ' '.join([ww.text for ww in seen]) 

257 unseen = [ww for ww in clause_span if ww not in seen and 

258 ((ww.pos_ == 'PUNCT' and ww.lemma_ == '?') or ww.pos_ != 'PUNCT')] 

259 unseen_chunk = ' '.join([ww.text for ww in unseen]) 

260 # Need to remove the relcl subtree from the noun's subtree 

261 noun_text = noun_text.replace(seen_chunk, empty_string).strip() 

262 # Relcl may have a relative pronoun or adverb (e.g., 'which Mary bought', 'Arizona where Cheney lost' - 

263 # which and where are the relative pronoun and adverb, respectively) that should be removed 

264 # Also need to add the relcl noun text back into the seen chunk (the relcl subtree) 

265 relcl_objs = [ww for ww in relcl.subtree if ww.dep_ == 'dobj' and ww.text in relative_clause_words] 

266 relcl_subjs = [ww for ww in relcl.subtree if 'subj' in ww.dep_ and ww.text in relative_clause_words] 

267 if relcl_subjs: 

268 seen_chunk = noun_text + space + seen_chunk.replace(f'{relcl_subjs[0].text}', empty_string).strip() 

269 elif relcl_objs: 

270 seen_chunk = seen_chunk.replace(f'{relcl_objs[0].text}', empty_string).strip() + space + noun_text 

271 else: 

272 seen_chunk = seen_chunk.strip() + space + noun_text 

273 # Does the 'seen' chunk (the relcl) have an advcl? 

274 adv_clauses = _split_advcl_clauses(seen_chunk, nlp) 

275 split_adv_clauses = [] 

276 for adv_sent in adv_clauses: # Do the advcls have conjunctions? 

277 comp_clauses = _split_complement_clauses(adv_sent, nlp) 

278 for comp_clause in comp_clauses: 

279 split_adv_clauses.extend(_split_by_conjunctions(comp_clause, nlp)) 

280 # Reassemble all the sentences 

281 new_clauses.extend(split_adv_clauses) 

282 new_clauses.append(unseen_chunk) 

283 return new_clauses if new_clauses else [clause]

286def _store_chunks_in_order(chunk1: str, chunk2: str) -> list: 

287 """ 

288 Save two strings in an array where chunk1 is first. Also, this method removes any 

289 ending periods. 

290 

291 :param chunk1: The first string 

292 :param chunk2: The second string 

293 :return: An array consisting of the two strings 

294 """ 

295 return [chunk1[0:-1] if chunk1.endswith('.') else chunk1, chunk2[0:-1] if chunk2.endswith('.') else chunk2]

298def split_clauses(sent_text: str, nlp: Language) -> list: 

299 """ 

300 Perform splitting of sentences by (1) finding the root verb, (2) working from its conjunctions (conj), 

301 clausal complements (ccomp) and adverbial clauses (advcl), (3) iterating through/splitting these clauses 

302 if they include a subject/object and verb,and (4) checking the final clauses if their subject/object 

303 contains a relative clause with a subject and verb. 

304 

305 :param sent_text: The sentence to be split (as a string) 

306 :param nlp: A spaCy Language model 

307 :return: A list of new sentences (text only) created from the splits 

308 """ 

309 initial_sents = [] 

310 if 'Quotation' in sent_text: 

311 initial_sents.extend(_chunk_quotations(sent_text)) 

312 else: 

313 initial_sents.append(sent_text) 

314 final_with_conn_words = [] 

315 for initial_sent in initial_sents: 

316 if initial_sent.startswith('Quotation'): 

317 final_with_conn_words.append(initial_sent) 

318 continue 

319 new_sents = _split_by_conjunctions(initial_sent, nlp) 

320 split_sents = [] 

321 # Split by advcl IF these have their own subject/verb 

322 # Example: 'When I went to the store, I met George.' ('when ...' is an adverbial clause) 

323 for sent in new_sents: 

324 adv_sents = _split_advcl_clauses(sent, nlp) 

325 # Split by ccomp IF these have their own subject/verb 

326 # Example: 'He said Joe is ill.' ('Joe is ill' is a clausal complement) 

327 for adv_sent in adv_sents: 

328 comp_sents = _split_complement_clauses(adv_sent, nlp) 

329 for comp_sent in comp_sents: 

330 split_sents.extend(_split_by_conjunctions(comp_sent, nlp)) 

331 # Check relcl 

332 split_sents2 = [] 

333 for sent in split_sents: 

334 split_sents2.extend(_split_relcl_clauses(sent, nlp)) 

335 # Check for advcls that are not directly associated with the root verb but still have a subject/object and verb 

336 for sent in split_sents2: 

337 sent_span = next(nlp(sent).sents) 

338 advcl_verbs = [] 

339 for token in sent_span: 

340 advcl_verbs.extend([child for child in token.children if child.dep_ == 'advcl']) 

341 new_chunks = [] 

342 for advcl_verb in advcl_verbs: # There are some advcls remaining that are not associated w/ the root verb 

343 connectors = [conn for conn in advcl_verb.children if conn.dep_ in ('advmod', 'mark')] 

344 # Process the verb and the first connector (there should only be 1) 

345 if connectors: 

346 connector = connectors[0] 

347 chunks = _get_chunks(advcl_verb, connector, sent_span, 'advcl', None) 

348 revised_chunks = [] 

349 for chunk in chunks: 

350 # Remove the connector from the middle of the text of the sentence 

351 revised_chunks.append(chunk.replace(f' {connector} ', space)) 

352 new_chunks.extend(revised_chunks) 

353 final_with_conn_words.extend(new_chunks if new_chunks else [sent]) 

354 # Chunks may still have beginning or trailing 'mark' words (such as 'that' in 'she claimed that') 

355 final_chunks = [] 

356 for clause in final_with_conn_words: 

357 # Relative and connector words may be present at the beginning or end of the clauses, and should be removed 

358 # TODO: Is the ordering (relcl to conj) correct? Should it be recursive? 

359 for word in relative_clause_words: 

360 clause = _remove_connector_text(clause, word) 

361 for word in conjunction_words: 

362 clause = _remove_connector_text(clause, word) 

363 # May still have "special mark"s that need to be addressed in the semantics 

364 for word in special_marks: 

365 revised_clause = _remove_connector_text(clause, word) 

366 if clause != revised_clause: 

367 clause = f'{revised_clause}$&{word}' 

368 final_chunks.append(clause) 

369 return final_chunks
