16def _chunk_quotations(text_str: str) -> list: 

17 """ 

18 Given a sentence text that includes 'Quotation#', chunk that text into individual clauses for 

19 each Quotation reference and any remaining text. 

20 

21 :param text_str: The text that contains at least one 'Quotation#' reference 

22 :return: A list of the text chunks, splitting each 'Quotation#' reference into its own chunk 

23 and including each intervening text as a chunk 

24 """ 

25 quotation_refs = re.findall(r'Quotation[0-9]+', text_str) # Really only care about the first ref 

26 chunks = [] 

27 quote_index = text_str.index(quotation_refs[0]) 

28 if quote_index > 0: 

29 chunk_text = text_str[0:quote_index].strip() 

30 chunks.append(chunk_text[0:-1] if chunk_text.endswith('.') else chunk_text) 

31 text_str = text_str[quote_index:] 

32 chunks.append(quotation_refs[0]) 

33 text_str = text_str.replace(quotation_refs[0], empty_string) 

34 if not text_str or text_str == space: 

35 return chunks 

36 text_str = text_str.strip() 

37 if 'Quotation' not in text_str: 

38 chunks.append(text_str) 

39 return chunks 

40 chunks.extend(_chunk_quotations(text_str)) # Recursively deal with the rest of the sentence 

41 return chunks

44def _get_chunks(verb: Token, connector: Union[Token, None], chunk_sentence: Span, processing_type: str,

46 """ 

47 Given (1) a conjunctive or clausal phrase for a verb, (2) an optional connector token relating the 

48 phrase to the verb (such as 'and', 'but', ...) and (3) the sentence that contains these tokens, determine 

49 if the sentence can be separated into clauses. The sentence is separated if the clause has its own 

50 subject and verb. 

51 

52 Note that this algorithm (seen/unseen_chunk) is taken from the StackOverflow response at 

53 https://stackoverflow.com/questions/65227103/clause-extraction-long-sentence-segmentation-in-python. 

54 

55 :param verb: The token of the verb with the conjunctive or clausal phrase 

56 :param connector: Token of the connector between the root and conjunctive or clausal verb, 

57 or None (which may occur if this is a clausal complement - e.g., "He said 

58 Joe is ill.") 

59 :param chunk_sentence: A Span representing the parsed sentence 

60 :param processing_type: A string indicating whether this is for a conjunctive verb ('conj'), 

61 adverbial clause verb ('advcl'), clausal complement ('comp') or 

62 relative clause ('relcl') 

63 :param subjs: An array holding the tokens of any subjects or expletives (it is only needed when 

64 processing conjunctions 

65 :return: A list of the text of the clauses of the sentence (split if indicated by the 

66 logic below) or just the original sentence returned 

67 """ 

68 if subjs is None: 

69 subjs = [] 

70 chunks = [] 

71 subj_texts = [] 

72 # Does the other clause's verb have a subject ('nsubj'/'nsubjpass'/'expl' (e.g., 'there')) in the dep parse? 

73 subj2 = [subj for subj in verb.children if ('subj' in subj.dep_ or subj.dep_ == 'expl') and 

74 ((subj.pos_ == 'PRON' and subj.morph.get('PronType') != ['Dem']) or subj.pos_ != 'PRON')] 

75 if not subj2 and (processing_type in ('conj', 'advcl')): 

76 # Attach the root verb's subject(s) to the other verb 

77 for subj in subjs: 

78 for subtree in subj.subtree: 

79 subj_texts.append(subtree.text) 

80 if len(subj2) or subj_texts: 

81 # Yes ... Separate clauses 

82 # Get the tokens in sentence related to the 'other' verb's subtree 

83 seen = [ww for ww in verb.subtree if ( 

84 (ww.pos_ == 'PUNCT' and (ww.lemma_ == '?' or ww.lemma_ == ',')) or 

85 not ((ww.dep_ == 'dobj' and ww.pos_ == 'PRON') or ww.pos_ == 'PUNCT'))] 

86 seen_chunk = ' '.join([ww.text for ww in seen]).strip() 

87 seen_chunk = seen_chunk.replace(' .', empty_string) if seen_chunk.endswith(' .') else seen_chunk 

88 unseen = [ww for ww in chunk_sentence if ww not in seen and 

89 ((ww.pos_ == 'PUNCT' and (ww.lemma_ == '?' or ww.lemma_ == ',')) or ww.pos_ != 'PUNCT')] 

90 unseen_chunk = ' '.join([ww.text for ww in unseen]).strip() 

91 unseen_chunk = unseen_chunk.replace(' .', empty_string) if unseen_chunk.endswith(' .') else unseen_chunk 

92 if connector: 

93 seen_chunk = _remove_connector_text(seen_chunk, connector.text) 

94 unseen_chunk = _remove_connector_text(unseen_chunk, connector.text) 

95 if processing_type == 'conj': 

96 # Store the unseen and seen clauses in that order 

97 chunks.extend( 

98 _store_chunks_in_order(unseen_chunk, 

99 f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk)) 

100 elif processing_type == 'advcl': 

101 if connector.text.lower() in later_connectors: # Store unseen_chunk first 

102 if 'eventually' in unseen_chunk.lower() or 'afterwards' in unseen_chunk.lower() or \ 

103 'finally' in unseen_chunk.lower(): # Keywords override ordering 

104 chunks.extend( 

105 _store_chunks_in_order( 

106 f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk, unseen_chunk)) 

107 else: 

108 chunks.extend( 

109 _store_chunks_in_order( 

110 unseen_chunk, f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk)) 

111 else: # Store seen_chunk first 

112 if connector.text.lower() in special_marks: 

113 seen_chunk += f'$&{connector.text.lower()}' 

114 chunks.extend( 

115 _store_chunks_in_order( 

116 f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk, unseen_chunk)) 

117 else: 

118 chunks.extend( 

119 _store_chunks_in_order( 

120 unseen_chunk, f"{' '.join(subj_texts)} {seen_chunk}" if subj_texts else seen_chunk)) 

121 if len(chunks) > 0: 

122 return chunks 

123 else: 

124 return [chunk_sentence.text[0:-1] if chunk_sentence.text.endswith('.') else chunk_sentence.text]

160def _split_advcl_clauses(sentence: str, nlp: Language) -> list: 

161 """ 

162 Split the adverbial clauses of a sentence into separate sentences, if the clauses have their 

163 own subject and verb. 

164 

165 :param sentence: The sentence which is analyzed 

166 :param nlp: A spacy Language model 

167 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

168 returned, not the spacy tokens) or the original sentence if there is no advcl verb 

169 or if the connector is not one of the cause_ or effect_connectors 

170 """ 

171 sent_span = next(nlp(sentence).sents) # There should only be 1 sentence for each function call 

172 subjs = [child for child in sent_span.root.children if 'subj' in child.dep_] 

173 advcl_verbs = [child for child in sent_span.root.children if child.dep_ == 'advcl'] 

174 new_chunks = [] 

175 for advcl_verb in advcl_verbs: 175 ↛ 176line 175 didn't jump to line 176, because the loop on line 175 never started

176 connectors = [conn for conn in advcl_verb.children if conn.dep_ in ('advmod', 'mark')] 

177 # Process the verb and the first connector (there should only be 1, but there may be 0) 

178 if connectors: 

179 chunks = _get_chunks(advcl_verb, connectors[0], sent_span, 'advcl', subjs) 

180 else: 

181 chunks = _get_chunks(advcl_verb, None, sent_span, 'advcl', subjs) 

182 revised_chunks = [] 

183 if connectors: 

184 # Remove the connector from the middle of the text of the sentence 

185 revised_chunks.extend([chunk.replace(f' {connectors[0]} ', space) for chunk in chunks]) 

186 else: 

187 revised_chunks = chunks 

188 new_chunks.extend(revised_chunks) 

189 return new_chunks if new_chunks else [sentence]

192def _split_by_conjunctions(sentence: str, nlp: Language) -> list: 

193 """ 

194 Split the clauses of a sentence into separate sentences, if the verbs are connected by a coordinating 

195 conjunction (for, and, but, or, nor, yet, so). For example,"Mary and John walked to the store." would 

196 NOT be split, but "Mary biked to the market and John walked to the store." should be split. 

197 

198 :param sentence: The sentence which is analyzed 

199 :param nlp: A spacy Language model 

200 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

201 returned, not the spacy tokens) or the original sentence if there are no 

202 conjunctive verbs 

203 """ 

204 sent_span = next(nlp(sentence).sents) # There should only be 1 sentence for each function call 

205 new_sents = [] 

206 conj_verbs = [child for child in sent_span.root.children if child.dep_ == 'conj'] 

207 connectors = [conn for conn in sent_span.root.children if conn.dep_ == 'cc'] 

208 if conj_verbs and len(conj_verbs) == len(connectors): 208 ↛ 209line 208 didn't jump to line 209, because the condition on line 208 was never true

209 subjects = [child for child in sent_span.root.children if 'subj' in child.dep_] 

210 expls = [child for child in sent_span.root.children if child.dep_ == 'expl'] 

211 if expls: 

212 subjects.extend(expls) 

213 # Process the first 'chunk' and then return - Subsequent iterations will process the complete text 

214 chunks = _get_chunks(conj_verbs[0], connectors[0], sent_span, 'conj', subjects) 

215 if len(chunks) > 1: 

216 for chunk in chunks: 

217 new_sents.extend(_split_by_conjunctions(chunk, nlp)) 

218 return new_sents 

219 else: 

220 return [chunks[0]] 

221 else: 

222 return [sentence[0:-1] if sentence.endswith('.') else sentence]

247def _split_relcl_clauses(clause: str, nlp: Language) -> list: 

248 """ 

249 Split the relative clauses of a sentence into separate sentences, if the clauses have their 

250 own subject and verb. 

251 

252 :param clause: The sentence which is analyzed 

253 :param nlp: A spacy Language model 

254 :return: An array of simpler sentence(s) (note that only the text of the sentences is 

255 returned, not the spacy tokens) or the original sentence if there is no relcl verb 

256 """ 

257 clause_span = next(nlp(clause).sents) # There should only be 1 sentence for each function call 

258 new_clauses = [] 

259 relcl_dict = dict() 

260 for token in clause_span: 

261 if 'subj' in token.dep_ or 'obj' in token.dep_: 

262 relcls = [child for child in token.children if child.dep_ == 'relcl' and child.pos_ in ('VERB', 'AUX')] 

263 if relcls: 263 ↛ 264line 263 didn't jump to line 264, because the condition on line 263 was never true

264 relcl_dict[token] = relcls 

265 for noun_token, relcls in relcl_dict.items(): 265 ↛ 266line 265 didn't jump to line 266, because the loop on line 265 never started

266 for relcl in relcls: 

267 noun_details = [ww.text for ww in noun_token.subtree] 

268 noun_text = ' '.join(noun_details) 

269 seen = [ww for ww in relcl.subtree] 

270 seen_chunk = ' '.join([ww.text for ww in seen]) 

271 unseen = [ww for ww in clause_span if ww not in seen and 

272 ((ww.pos_ == 'PUNCT' and (ww.lemma_ == '?' or ww.lemma_ == ',')) or ww.pos_ != 'PUNCT')] 

273 unseen_chunk = ' '.join([ww.text for ww in unseen]) 

274 # Need to remove the relcl subtree from the noun's subtree 

275 noun_text = noun_text.replace(seen_chunk, empty_string).strip() 

276 # Relcl may have a relative pronoun or adverb (e.g., 'which Mary bought', 'Arizona where Cheney lost' - 

277 # which and where are the relative pronoun and adverb, respectively) that should be removed 

278 # Also need to add the relcl noun text back into the seen chunk (the relcl subtree) 

279 relcl_objs = [ww for ww in relcl.subtree if ww.dep_ == 'dobj' and ww.text in relative_clause_words] 

280 relcl_subjs = [ww for ww in relcl.subtree if 'subj' in ww.dep_ and ww.text in relative_clause_words]

282 seen_chunk = noun_text + space + seen_chunk.replace(f'{relcl_subjs[0].text}', empty_string).strip()

284 seen_chunk = seen_chunk.replace(f'{relcl_objs[0].text}', empty_string).strip() + space + noun_text 

285 else: 

286 seen_chunk = seen_chunk.strip() + space + noun_text 

287 # Does the 'seen' chunk (the relcl) have an advcl? 

288 adv_clauses = _split_advcl_clauses(seen_chunk, nlp) 

289 split_adv_clauses = [] 

290 for adv_sent in adv_clauses: # Do the advcls have conjunctions?

292 for comp_clause in comp_clauses: 

293 split_adv_clauses.extend(_split_by_conjunctions(comp_clause, nlp)) 

294 # Reassemble all the sentences 

295 new_clauses.extend(split_adv_clauses) 

296 new_clauses.append(unseen_chunk) 

297 return new_clauses if new_clauses else [clause]

312def split_clauses(sent_text: str, nlp: Language) -> list:

317 contains a relative clause with a subject and verb.

319 :param sent_text: The sentence to be split (as a string)

325 initial_sents.extend(_chunk_quotations(sent_text))

328 final_with_conn_words = [] 

329 for initial_sent in initial_sents: 

330 if initial_sent.startswith('Quotation'): 330 ↛ 331line 330 didn't jump to line 331, because the condition on line 330 was never true

331 final_with_conn_words.append(initial_sent) 

332 continue 

333 new_sents = _split_by_conjunctions(initial_sent, nlp) 

334 split_sents = [] 

335 # Split by advcl IF these have their own subject/verb 

336 # Example: 'When I went to the store, I met George.' ('when ...' is an adverbial clause) 

337 for sent in new_sents: 

338 adv_sents = _split_advcl_clauses(sent, nlp) 

339 # Split by ccomp IF these have their own subject/verb 

340 # Example: 'He said Joe is ill.' ('Joe is ill' is a clausal complement) 

341 for adv_sent in adv_sents: 

342 comp_sents = _split_complement_clauses(adv_sent, nlp) 

343 for comp_sent in comp_sents: 

344 split_sents.extend(_split_by_conjunctions(comp_sent, nlp)) 

345 # Check relcl

347 for sent in split_sents: 

348 split_sents2.extend(_split_relcl_clauses(sent, nlp)) 

349 # Check for advcls that are not directly associated with the root verb but still have a subject/object and verb 

350 for sent in split_sents2: 

351 sent_span = next(nlp(sent).sents) 

352 advcl_verbs = [] 

353 for token in sent_span: 

354 advcl_verbs.extend([child for child in token.children if child.dep_ == 'advcl'])

356 for advcl_verb in advcl_verbs: # There are some advcls remaining that are not associated w/ the root verb 356 ↛ 357line 356 didn't jump to line 357, because the loop on line 356 never started

365 revised_chunks.append(chunk.replace(f' {connector} ', space)) 

366 new_chunks.extend(revised_chunks) 

367 final_with_conn_words.extend(new_chunks if new_chunks else [sent]) 

368 # Chunks may still have beginning or trailing 'mark' words (such as 'that' in 'she claimed that') 

369 final_chunks = [] 

370 for clause in final_with_conn_words: 

371 # Relative and connector words may be present at the beginning or end of the clauses, and should be removed
