26def _adjust_xcomp_prt(processing: list) -> list: 

27 """ 

28 Clean up the verb_processing text if there are both prt and xcomp details. If there is only a 

29 single 'xcomp' or 'prt', then just remove the final '$'. If both, correct either the first or 

30 second verb of the 'xcomp' details for one of the verbs (the one matching the prt verb) to 

31 have it reference the full 'prt' text (verb + prt). 

32 

33 :param processing: Array holding the xcomp and prt details 

34 :return: List holding the final xcomp/prt details 

35 """ 

36 xcomp_tuples = [] 

37 prts = [] 

38 for detail in processing: 

39 if 'xcomp' in detail: 

40 xcomp_tuples.append((detail.split('> ')[1].split(', ')[0], detail.split(', ')[1])) 

41 elif 'prt' in detail: 

42 prts.append(detail.split('> ')[1]) 

43 if xcomp_tuples and prts: 

44 new_xcomps = [] 

45 for verb1, verb2 in xcomp_tuples: 

46 for prt in prts: 

47 if prt.startswith(verb1): 

48 verb1 = prt 

49 break 

50 elif prt.startswith(verb2): 

51 verb2 = prt 

52 break 

53 new_xcomps.append(f'xcomp > {verb1}, {verb2}') 

54 return new_xcomps 

55 elif xcomp_tuples: 

56 return [f'xcomp > {verb1}, {verb2}' for verb1, verb2 in xcomp_tuples] 

57 elif prts: 

58 return [f'prt > {prt}' for prt in prts]

61def _process_aux(aux_token: Token, verb_dictionary: dict): 

62 """ 

63 When processing a token that is an auxiliary, such as a modal or a primary auxiliary 

64 (be, do, have), get the details. 

65 

66 :param aux_token: Token holding the spaCy details about the auxiliary 

67 :param verb_dictionary: Dictionary holding all the details about the verb/sentence 

68 being processed 

69 :return: None (the verb_dictionary is updated) 

70 """ 

71 if aux_token.lemma_ == 'to': 

72 return 

73 elif aux_token.lemma_ in ('be', 'do', 'have'): 

74 child_tense = aux_token.morph.get('Tense') 

75 if verb_dictionary['tense'] == 'Pres' and child_tense and child_tense == 'Past': 

76 verb_dictionary['tense'] = 'Past' 

77 else: 

78 if aux_token.lemma_ == 'will': 

79 verb_dictionary['tense'] = 'Fut' 

80 # Aux is expressing mood (ex: enjoyed/hated being with), possibility (could, may, might, should), 

81 # intent (would, will), ability (can/cannot), suggestion (should) or necessity/obligation (must), 

82 # or is asking permission (can, may, could) or requesting (would, will, can, could) 

83 verb_dictionary['verb_aux'] = aux_token.text

86def extract_chunk_details(chunk: str, chunk_dict: dict, nlp: Language) -> dict: 

87 """ 

88 Extract the subject, verb, object details from a sentence and store these in a dictionary for 

89 further processing (and rendering in Turtle). 

90 

91 :param chunk: The sentence/chunk to be processed 

92 :param chunk_dict: The dictionary holding the chunk details 

93 :param nlp: A spaCy Language model 

94 :return: Returns the updated chunk_dict 

95 """ 

96 logging.info(f'Creating chunk dictionary for {chunk}') 

97 nlp_chunk = nlp(chunk) 

98 # There should only be 1 root verb, because the text was already split 

99 process_verb([chunk.root for chunk in nlp_chunk.sents][0], chunk_dict, nlp) 

100 return chunk_dict

103def process_verb(token: Token, dictionary: dict, nlp: Language): 

104 """ 

105 When processing a token (that is a verb), capture its details. Note that all processing 

106 of a narrative is based on the sentences/chunks and their ROOT verbs. 

107 

108 :param token: Verb token from spacy parse 

109 :param dictionary: Dictionary that the token details should be added to 

110 :param nlp: A spaCy Language model 

111 :return: None (Specified dictionary is updated) 

112 """ 

113 logging.info(f'Processing the verb, {token.text}') 

114 verb_dict = dict() 

115 verb_dict['verb_text'] = token.text 

116 verb_dict['verb_lemma'] = token.lemma_ 

117 tense = token.morph.get('Tense') 

118 verb_dict['tense'] = tense[0] if tense else 'Pres' 

119 # Processing is based on dependency parsing 

120 # Dependency tokens are defined at https://downloads.cs.stanford.edu/nlp/software/dependencies_manual.pdf 

121 for child in token.children: 

122 if child.dep_ == 'xcomp': # Clausal complement - e.g., 'he attempted robbing the bank' (xcomp = robbing) 

123 # For the above, xcomp > attempt, rob 

124 add_to_dictionary_values(dictionary, 'verb_processing', f'xcomp > {token.lemma_}, {child.lemma_}', str) 

125 process_verb(child, dictionary, nlp) 

126 for child_child in child.children: 

127 if child_child.dep_ == 'conj': 

128 # Conjunction of xcomps ('Sue failed to win the race or to finish.') handled as 

129 # 3 verbs with multiple xcomp processing details 

130 add_to_dictionary_values(dictionary, 'verb_processing', 

131 f'xcomp > {token.lemma_}, {child_child.text}', str) 

132 process_verb(child_child, dictionary, nlp) 

133 elif 'prt' in child.dep_: # Phrasal verb particle - e.g., 'he gave up the jewels' (up = prt) 

134 # For the above, prt > give up 

135 add_to_dictionary_values(dictionary, 'verb_processing', f'prt > {token.lemma_} {child.text}', str) 

136 elif child.dep_ in ('acomp', 'advcl', 'ccomp'): # Various complements of the verb 

137 add_token_details(child, verb_dict, f'verb_{child.dep_}') 

138 elif 'agent' == child.dep_: # Agent of a passive verb, introduced by the word, 'by' 

139 for child2 in child.children: 

140 if 'pobj' == child2.dep_: # The real subject is the object of the preposition, 'by' 

141 add_token_details(child2, dictionary, subjects_string) 

142 elif 'aux' == child.dep_: 

143 # Care about modals, tense of the primary auxiliaries (be, do, have), and semantics of mood 

144 _process_aux(child, verb_dict) 

145 elif 'neg' == child.dep_: # Verb is negated 

146 verb_dict['negation'] = True 

147 elif 'nsubj' == child.dep_: # Subject of the verb 

148 add_token_details(child, dictionary, subjects_string) 

149 elif 'nsubjpass' == child.dep_: # Subject of a passive verb -> Which means that it is an object 

150 add_token_details(child, dictionary, objects_string) 

151 # TODO: Handle subjects that are themselves clauses (csubj and csubjpass) 

152 elif 'obj' in child.dep_ or child.dep_ in ('attr', 'dative'): # Object or attribute of the verb 

153 add_token_details(child, verb_dict, objects_string) 

154 elif 'prep' in child.dep_: # Preposition associated with the verb 

155 # Only concerned with certain prepositions and their semantics 

156 if child.text.lower() not in processed_prepositions: 

157 continue 

158 prep_dict = dict() 

159 prep_dict['prep_text'] = child.text 

160 for prep_dep in child.children: 

161 if 'prep' in prep_dep.dep_: 

162 # Preposition followed by another preposition! 

163 prep_dict['prep_text'] += f' {prep_dep.text}' 

164 for prep2_dep in prep_dep.children: # Only going 2 levels deep on prepositions 

165 add_token_details(prep2_dep, prep_dict, 'prep_details') 

166 else: 

167 add_token_details(prep_dep, prep_dict, 'prep_details') 

168 add_to_dictionary_values(verb_dict, 'preps', prep_dict, dict) 

169 # Need to adjust the processing if there are prt and xcomp children 

170 if 'verb_processing' in dictionary and not dictionary['verb_processing']: 

171 # dictionary['verb_processing'] = _adjust_xcomp_prt(dictionary['verb_processing']) 

172 del dictionary['verb_processing'] 

173 add_to_dictionary_values(dictionary, 'verbs', verb_dict, dict)
