13def _check_agent_relevance(agent_text: str, sent_dict: dict) -> bool: 

14 """ 

15 Determine if the text identified by spaCy NER is actually an Agent. Adjectives may be 

16 identified - for example, 'Jewish' in a sentence with the words 'Jewish settlement'. 

17 These are not actually agents. 

18 

19 :param agent_text: Text identifying the agent 

20 :param sent_dict: The sentence dictionary 

21 :return: A boolean indicating that the text is an Agent (if true) or not (if false) 

22 """ 

23 chunks = sent_dict['chunks'] 

24 for chunk in chunks: 

25 chunk_str = str(chunk) 

26 indexes = [] 

27 if chunk_str.find("'subjects'") > 0: 

28 indexes.append(chunk_str.index("'subjects'")) 

29 if chunk_str.find("'objects'") > 0: 

30 indexes.append(chunk_str.index("'objects'")) 

31 indexes.append(chunk_str.index("'verb")) # Always have a verbs in a chunk, but may have 'verb_processing' 

32 details_after_text = min(indexes) # Get the details directly following the 'chunk_text' entry 

33 chunk_str = chunk_str[details_after_text:] 

34 agent_words = agent_text.split() 

35 for agent_word in agent_words: 

36 if agent_word in chunk_str: 

37 return True 

38 return False

41def _check_if_agent_is_known(agent_text: str, agent_type: str, alet_dict: dict) -> (str, str): 

42 """ 

43 Determines if the agent is already processed and identified in the alet_dict. 

44 

45 The acronym, alet, stands for agent-location-event-time. 

46 

47 :param agent_text: Input string identifying the agent 

48 :param agent_type: String identifying NER type for the text, as defined by spaCy 

49 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

50 the full narrative - For co-reference resolution (it may be updated by this function); 

51 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'agents') = array of arrays 

52 with index 0 holding an array of labels associated with the agent (variations on their 

53 name), index 1 storing the agent's entity type and index 2 storing the agent's IRI 

54 :return: A tuple holding the entity type and IRI of the agent, or two empty strings 

55 """ 

56 if 'agents' not in alet_dict: 

57 return agent_type, empty_string 

58 known_agents = alet_dict['agents'] 

59 for known_agent in known_agents: 

60 if agent_text in known_agent[0]: # Strings match 

61 return known_agent[1], known_agent[2] # NER maps to a known/processed agent, with type and IRI 

62 return agent_type, empty_string

65def _get_agent_iri_and_ttl(agent_text: str, agent_type: str, alet_dict: dict, 

66 use_sources: bool) -> (str, str, str, list): 

67 """ 

68 Handle agent names identified by spaCy's NER, stored in the sentence dictionary with the 

69 key, 'AGENTS'. 

70 

71 :param agent_text: Text identifying the agent 

72 :param agent_type: String holding the NER type identified by spaCy 

73 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

74 the full narrative - For co-reference resolution (it may be updated by this function); 

75 Keys = 'agents', 'locs', 'events', 'times', Values (for 'agents') = array of arrays 

76 with index 0 holding an array of labels associated with the agent (variations on their 

77 name), index 1 storing the agent's entity type and index 2 storing the agent's IRI 

78 :param use_sources: Boolean indicating whether additional information on an agent should 

79 be retrieved from Wikidata (recommended) 

80 :return: A tuple holding 1-3 strings with the agent's entity type, DNA class mapping and IRI, 

81 and 4) a list of Turtle statements defining the agent (also, the alet_dict is updated) 

82 """ 

83 agent_iri = re.sub(r'[^:a-zA-Z0-9_]', underscore, f':{agent_text}').replace('__', '_') 

84 labels = [] 

85 family_names = [] 

86 wiki_details = empty_string 

87 if use_sources: 

88 wiki_details = get_wikipedia_description(agent_text.replace(' ', underscore)) 

89 if wiki_details and 'See the web site' not in wiki_details: 

90 labels = get_wikidata_labels(wiki_details.split('wikibase_item: ')[1].split(')')[0]) 

91 # Update alet_dict 

92 known_agents = alet_dict['agents'] if 'agents' in alet_dict else [] 

93 if 'PERSON' in agent_type: 

94 family_names = _update_agent_names(agent_text, labels) 

95 agent_type = check_name_gender(agent_text) 

96 known_agents.append([labels, agent_type, agent_iri]) 

97 for fam_name in family_names: 

98 known_agents.append([[fam_name], 'PLURALPERSON', f':{fam_name}']) 

99 alet_dict['agents'] = known_agents 

100 agent_class = get_agent_or_loc_class(agent_type) 

101 if agent_type.endswith('NORP'): 

102 agent_iri = f'{agent_iri}_{str(uuid.uuid4())[:13]}' 

103 return agent_type, agent_class, agent_iri, \ 

104 create_norp_ttl(agent_text, agent_type, agent_iri, wiki_details, labels) 

105 return agent_type, agent_class, agent_iri, \ 

106 create_agent_ttl(agent_iri, labels, agent_type, agent_class, wiki_details, family_names)

109def _update_agent_names(agent_text: str, alt_names: list) -> list: 

110 """ 

111 Create the possible permutations of an agent's name. 

112 

113 :param agent_text: Text specifying the agent's name 

114 :param alt_names: An array of possible alternative names for the agent 

115 :return: An array of possible family names (if the agent_name includes a space) and 

116 the alt_names array may be updated 

117 """ 

118 no_paren = agent_text.replace('(', empty_string).replace(')', empty_string) 

119 split_names = no_paren.split() 

120 for name in split_names: 

121 if name not in alt_names: 

122 alt_names.append(name) 

123 perm_list = [] 

124 family_names = [] 

125 if len(split_names) > 1: 

126 for i in range(1, len(split_names)): 

127 family_names.append(f'{split_names[i]}s') 

128 perm_list = get_name_permutations(no_paren) 

129 for name in perm_list: 

130 if name not in alt_names: 

131 alt_names.append(name) 

132 if agent_text not in alt_names: 

133 alt_names.append(agent_text) 

134 if no_paren not in alt_names: 

135 alt_names.append(no_paren) 

136 return family_names

139def get_name_permutations(name: str) -> list: 

140 """ 

141 Get the combinations of first and maiden/last names. 

142 

143 :param name: A string holding a Person's full name 

144 :return: A list of strings combining the first and second, first and third, ... names 

145 """ 

146 poss_names = [] 

147 names = name.split() 

148 for i in range(1, len(names)): 

149 poss_names.append(f'{names[0]} {names[i]}') 

150 return poss_names

153def get_sentence_agents(sent_dict: dict, alet_dict: dict, last_nouns: list, use_sources: bool) -> (list, list): 

154 """ 

155 Handle agent names identified by spaCy's NER, stored in the sentence dictionary with the 

156 key, 'AGENTS'. 

157 

158 :param sent_dict: The sentence dictionary 

159 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

160 the full narrative - For co-reference resolution (it may be updated by this function); 

161 Keys = 'agents', 'locs', 'events', 'times', Values (for 'agents') = array of arrays 

162 with index 0 holding an array of labels associated with the agent (variations on their 

163 name), index 1 storing the agent's entity type and index 2 storing the agent's IRI 

164 :param last_nouns: An array of tuples = noun texts, type, class mapping and IRI 

165 from the current paragraph 

166 :param use_sources: Boolean indicating whether additional information on an agent should 

167 be retrieved from Wikidata (recommended) 

168 :return: A tuple consisting of two arrays: (1) the IRIs of the agents identified in the sentence, 

169 and 2) Turtle statements defining them (also the alet_dict may be updated) 

170 """ 

171 agents_turtle = [] 

172 new_agents = sent_dict['AGENTS'] 

173 new_agents_dict = dict() 

174 agent_iris = [] 

175 for new_agent in new_agents: 

176 agent_split = new_agent.split('+') 

177 # Ignore NORP in sentence AGENT analysis (will be addressed at the level of the verb, if needed) 

178 if agent_split[1] != 'NORP': 

179 new_agents_dict[agent_split[0]] = _check_if_agent_is_known(agent_split[0], agent_split[1], alet_dict) 

180 for new_agent, agent_details in new_agents_dict.items(): 

181 agent_type, agent_iri = agent_details 

182 if not agent_iri: 

183 # Need to define the Turtle for a new agent 

184 agent_type, agent_class, agent_iri, agent_ttl = \ 

185 _get_agent_iri_and_ttl(new_agent, agent_type, alet_dict, use_sources) 

186 agents_turtle.extend(agent_ttl) 

187 else: 

188 agent_class = get_agent_or_loc_class(agent_type) 

189 last_nouns.append((new_agent, agent_type, [agent_class], agent_iri)) 

190 agent_iris.append(agent_iri) 

191 return agent_iris, agents_turtle
