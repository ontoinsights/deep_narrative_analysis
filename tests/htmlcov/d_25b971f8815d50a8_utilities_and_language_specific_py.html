162def add_to_dictionary_values(dictionary: dict, key: str, value, value_type): 

163 """ 

164 Add the value (cast using the value_type parameter) for the specified key in the dictionary, 

165 if that key exists, or create a new dictionary entry for the key (where the value is a list). 

166 

167 :param dictionary: The dictionary to be updated 

168 :param key: String holding the dictionary key 

169 :param value: The value to be added to the current list of values of the dictionary entry, or 

170 a new list is created with the value 

171 :param value_type: The type of the value 

172 :return: None (Dictionary is updated) 

173 """ 

174 values = [] 

175 if key in dictionary: 

176 values = dictionary[key] 

177 values.append(value_type(value)) 

178 dictionary[key] = values

194def check_name_gender(name_str: str) -> str: 

195 """ 

196 Determines if a reference is to male/female name. Returns a type of either <FEMALE/MALE>SINGPERSON 

197 (if the gender can be determined) or 'SINGPERSON'. 

198 

199 :param name_str: The string representing the noun or proper name 

200 :return: A tuple of 2 strings representing the entity's text and type (with gender if known) 

201 """ 

202 gender = empty_string 

203 if space in name_str: 

204 names = name_str.split() 

205 else: 

206 names = [name_str] 

207 for name in names: 

208 gender = 'FEMALE' if name in female_names else ('MALE' if name in male_names else empty_string) 

209 if gender: 

210 break 

211 return f'{gender}SINGPERSON' if gender else 'SINGPERSON'
