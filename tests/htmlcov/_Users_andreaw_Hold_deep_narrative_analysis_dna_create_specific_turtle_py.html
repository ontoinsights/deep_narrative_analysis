68def create_environment_ttl(text: str, specific_class: str, general_type: str, subjects: list) -> list: 

69 """ 

70 Return the Turtle for a Person identified as a member of an organization (NORP), as having 

71 an emotion or as involved in a line of business (LoB). 

72 

73 :param text: Specific text from the sentence that was used for the identification 

74 :param specific_class: The most specific class in the DNA ontology to which the text was mapped 

75 :param general_type: String = either 'EmotionalResponse'|'Ethnicity'|'ReligiousBelief'| 

76 'LineOfBusiness'|'PoliticalIdeology' 

77 :param subjects: Array of tuples that are the subjects' texts, mappings, types and IRIs 

78 :return: Appropriate Turtle statement given the input parameters 

79 """ 

80 event_iri = f':{general_type}_{str(uuid.uuid4())[:13]}' 

81 if general_type == 'EmotionalResponse': 

82 new_ttl = [f'{event_iri} a {specific_class} ; rdfs:label "{text}" .'] 

83 else: 

84 new_ttl = [f'{event_iri} a :EnvironmentAndCondition ; :has_topic :{general_type} ; ' 

85 f'rdfs:label "The {general_type} of the holder - {text}." .'] 

86 for subj_text, subj_type, subj_mappings, subj_iri in subjects: 

87 new_ttl.append(f'{event_iri} :has_holder {subj_iri} .') 

88 if general_type == 'LineOfBusiness': 

89 new_ttl.append(f'{subj_iri} :has_line_of_business {specific_class} ; :line_of_business "{text}" .') 

90 if general_type == 'PoliticalIdeology': 

91 new_ttl.append(f'{subj_iri} :has_political_ideology {specific_class} ; :political_ideology "{text}" .') 

92 if general_type == 'ReligiousBelief': 

93 new_ttl.append(f'{subj_iri} :has_agent_aspect {specific_class} ; :religion "{text}" .') 

94 if general_type == 'Ethnicity': 

95 new_ttl.append(f'{subj_iri} :has_agent_aspect {specific_class} ; :ethnicity "{text}" .') 

96 return new_ttl
