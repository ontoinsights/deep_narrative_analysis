34def _clean_environment_condition(turtle: str, object_tuples: list, iri: str) -> str: 

35 """ 

36 Updates the Turtle for a subject/EnvironmentAndCondition statement. 

37 

38 :param turtle: The original Turtle from idiom processing 

39 :param object_tuples: An array of tuples consisting of the verb's objects' text, type and IRI 

40 :param iri: The IRI identifying the subject/holder of the environment/condition 

41 :returns: The updated Turtle statement 

42 """ 

43 topic = 'owl:Thing' 

44 word = 'unknown' 

45 if 'has_topic ' in turtle: 

46 if ' dobj' not in turtle: 

47 topic = turtle.split('has_topic ')[1].split(space)[0] 

48 elif ' dobj' in turtle and not object_tuples: # Possible for a verb to have no objects, just adverbs 

49 turtle = turtle.replace(':has_topic dobj ; ', empty_string) # So, remove the triple 

50 if 'word_detail' in turtle: 

51 word = turtle.split('word_detail ')[1].split(space)[0] 

52 if 'LineOfBusiness' in turtle: 

53 turtle = f'{turtle.replace("LineOfBusiness", empty_string).replace(":word_detail", "rdfs:label")} . ' \ 

54 f'subj :has_line_of_business {topic} ; :line_of_business {word} .' 

55 elif 'PoliticalIdeology' in turtle: 

56 turtle = f'{turtle.replace("PoliticalIdeology", empty_string).replace(":word_detail", "rdfs:label")} . ' \ 

57 f'subj :has_agent_aspect {topic} .' 

58 elif 'Ethnicity' in turtle: 

59 turtle = f'{turtle.replace("Ethnicity", empty_string).replace(":word_detail", "rdfs:label")} . ' \ 

60 f'subj :has_agent_aspect {topic} .' 

61 return turtle.replace('subj', iri).replace("'", '"')

64def _verify_iri_requirements(event_state_turtle: str, noun_type: str) -> bool: 

65 """ 

66 Verifies if a noun type meets the requirements of a dobj or pobj reference in a parsed idiom rule. 

67 

68 :param event_state_turtle: The Turtle statement resulting from processing an idiom rule 

69 :param noun_type: The type of the object/noun as defined by the spaCy parse 

70 :returns: A boolean indicating that 

71 """ 

72 if ('(Agent)' in event_state_turtle and 

73 (noun_type.endswith('PERSON') or noun_type.endswith('ORG') or noun_type.endswith('NORP') or 

74 noun_type.endswith('GPE'))) or \ 

75 ('(Location)' in event_state_turtle and 

76 (noun_type.endswith('GPE') or noun_type.endswith('LOC') or noun_type.endswith('FAC'))): 

77 return True 

78 # TODO: May have a NOUN but its definition could indicate an agent or location / Does using WordNet address this? 

79 return False

113def get_noun_idiom(noun: str, noun_phrase: str, noun_type: str, sentence: str, noun_iri: str) -> list: 

114 """ 

115 Specific semantics have been defined for several nouns - as specified by fixes to the knowledge 

116 graph. This method looks up the input noun in the nouns dictionary and retrieves the 

117 semantics. 

118 

119 :param noun: String holding the noun text 

120 :param noun_phrase: String specifying the complete noun phrase from the original narrative sentence 

121 :param noun_type: String holding the type of the noun (e.g., 'FEMALESINGPERSON' or 'PLURALNOUN') 

122 :param sentence: The full text of the sentence (needed for checking for idioms) 

123 :param noun_iri: String identifying the IRI/URL/individual associated with the noun_text 

124 :returns: Returns an array of strings defining the Turtle for the noun/noun_phrase 

125 """ 

126 process_ttl = [] 

127 if noun in noun_idiom_dict.keys(): 

128 processing = [noun_idiom_dict[noun]] 

129 if processing: 

130 noun_dict = {'text': noun_phrase, 'type': noun_type} 

131 idiom_ttl = process_idiom_detail(processing, sentence, noun_dict, []) 

132 for line in idiom_ttl: 

133 process_ttl.append(f'{noun_iri} a {line} .') 

134 process_ttl.append(f'{noun_iri} rdfs:label "{noun_phrase}" .') 

135 if noun_type.startswith('NEG'): 

136 process_ttl.append(f'{noun_iri} :negation true .') 

137 else: 

138 if space in noun: 

139 head_noun = get_head_word(noun) 

140 process_ttl = get_noun_idiom(head_noun, noun, noun_type, sentence, noun_iri) 

141 return process_ttl

144def get_verb_processing(verb_dict: dict) -> list: 

145 """ 

146 Specific semantics have been defined for idiomatic text, where the text has meaning different 

147 from the individual words (such as 'fish around' meaning 'search'). This method looks up verb + 

148 acomp or xcomp combinations in the verb-idioms dictionary and retrieves the 

149 semantics. 

150 

151 :param verb_dict: Dictionary holding the verb details 

152 :returns: If special processing is found, return the details (there may be more than 1 match, 

153 so an array is returned) 

154 """ 

155 lemma = verb_dict['verb_lemma'] 

156 if lemma in ('be', 'being', 'become', 'becoming'): 

157 return determine_processing_be(verb_dict) 

158 processing = [] 

159 if preps_string in verb_dict.keys(): 

160 for prep in verb_dict[preps_string]: 

161 preposition = prep['prep_text'] 

162 looking_for = f'{lemma} {preposition}' 

163 if looking_for in verb_prep_idiom_dict.keys(): 

164 processing.append(verb_prep_idiom_dict[looking_for]) 

165 if lemma in verb_idiom_dict.keys(): 

166 processing.append(verb_idiom_dict[lemma]) 

167 return processing

170def handle_event_state_idiosyncrasies(event_state_turtle: str, sentence: str, verb_dict: dict, subj_tuples: list, 

171 obj_tuples: list, prep_tuples: list) -> str: 

172 """ 

173 Handles subj (identified by the text, ':word_detail'), dobj/pobj and xcomp references in a 

174 parsed idiom rule. These could be resolved in the idiom processing but are not since 

175 1) there is no difference (or efficiencies to be had) in the code and 2) the processing involves 

176 calling the get_event_state_class function, which would result in a circular reference from the 

177 idiom_processing module. 

178 

179 At this time, subj and xcomp processing does not involve any object references. 

180 

181 :param event_state_turtle: The Turtle statement resulting from processing an idiom rule 

182 :param sentence: The sentence text 

183 :param verb_dict: The verb details from the sentence dictionary 

184 :param subj_tuples: An array of tuples consisting of the verb's subjects' text, type and IRI 

185 :param obj_tuples: An array of tuples consisting of the verb's objects' text, type and IRI 

186 :param prep_tuples: An array of tuples consisting of the verb's prepositions' text, and its objects' 

187 text, type and IRI 

188 :returns: The updated Turtle statement if 'word_detail', 'dobj', 'pobj' or 'xcomp' text was present in the 

189 input event_state_turtle 

190 """ 

191 if 'EnvironmentAndCondition' in event_state_turtle: # Indicates that a subj_rule was parsed 

192 subj_iris = [] 

193 for subj_text, subj_type, subj_iri in subj_tuples: 

194 subj_iris.append(subj_iri) 

195 iri_str = ', '.join(subj_iris) 

196 event_state_turtle = _clean_environment_condition(event_state_turtle, obj_tuples, iri_str) 

197 if 'xcomp' in event_state_turtle: # Appears as ':Event_xyz a xcomp(root_verb_text, xcomp_verb_text)' 

198 # Should always have 2 verbs, the root and the xcomp 

199 verb1_text = event_state_turtle.split('(')[1].split(', ')[0] 

200 verb2_text = event_state_turtle.split(', ')[1].split(')')[0] 

201 # Determine if the first verb has an idiom defined 

202 if verb1_text in verb_xcomp_dict.keys(): 

203 verb1_class = verb_xcomp_dict[verb1_text] 

204 else: 

205 verb1_class = get_event_state_class(verb1_text) # Get the Event/State class for the text 

206 # Check for second verb having an idiom (already handled the idiom for the first verb) 

207 xcomp_verb_dict = dict() 

208 xcomp_verb_dict['verb_lemma'] = verb2_text 

209 if preps_string in verb_dict.keys(): 

210 xcomp_verb_dict[preps_string] = verb_dict[preps_string] 

211 verb2_processing = get_verb_processing(xcomp_verb_dict) 

212 if verb2_processing: 

213 # TODO: Is it sufficient to take the first result? 

214 verb2_class = process_idiom_detail(verb2_processing, sentence, xcomp_verb_dict, prep_tuples)[0] 

215 verb2_class = handle_event_state_idiosyncrasies(verb2_class, sentence, verb_dict, subj_tuples, 

216 obj_tuples, prep_tuples) 

217 else: 

218 verb2_class = get_event_state_class(verb2_text) # Get the Event/State class for the text 

219 return f'{event_state_turtle.split("(")[0]}({verb1_class}, {verb2_class}) .' 

220 if 'dobj' in event_state_turtle: 

221 # Appears as dobj(Agent), dobj(Location), dobj(verb) or dobj (mutually exclusive) 

222 obj_iris = [] 

223 for obj_text, obj_type, obj_iri in obj_tuples: 

224 if dobj_verb in event_state_turtle: 

225 # Get the class name for the dobj 

226 if space in obj_text: 

227 obj_text = get_head_word(obj_text)[0] 

228 obj_iris.append(get_event_state_class(obj_text)) 

229 elif 'dobj(Agent)' not in event_state_turtle and 'dobj(Location)' not in event_state_turtle and \ 

230 dobj_verb not in event_state_turtle: 

231 obj_iris.append(obj_iri) 

232 elif _verify_iri_requirements(event_state_turtle, obj_type): 

233 obj_iris.append(obj_iri) 

234 if obj_iris: 

235 iri_str = ', '.join(obj_iris) 

236 event_state_turtle = event_state_turtle.replace(dobj_verb, iri_str).replace('dobj(Agent)', iri_str). \ 

237 replace('dobj(Location)', iri_str).replace('dobj', iri_str) 

238 else: 

239 logging.warning(f'Dobj in processed idiom but not found in sentence, {sentence}') 

240 if 'pobj' in event_state_turtle: 

241 # Appears as pobj(Agent), pobj(Location), pobj(prep_xxx), pobj(prep_xxx)(Agent | Location) 

242 # or pobj (mutually exclusive) 

243 obj_iris = [] 

244 if '(prep_' in event_state_turtle: 

245 prep_of_interest = event_state_turtle.split('(prep_')[1].split(')')[0] 

246 # Processing specific to a single preposition 

247 for preposition, obj_text, obj_type, obj_iri in prep_tuples: 

248 if preposition == prep_of_interest and \ 

249 (f'pobj(prep_{prep_of_interest})(Agent)' not in event_state_turtle and 

250 f'pobj(prep_{prep_of_interest})(Location)' not in event_state_turtle) or \ 

251 _verify_iri_requirements(event_state_turtle, obj_type): 

252 obj_iris.append(obj_iri) 

253 else: 

254 prep_of_interest = '' 

255 for preposition, obj_text, obj_type, obj_iri in prep_tuples: 

256 if ('(Agent)' not in event_state_turtle and '(Location)' not in event_state_turtle) or \ 

257 _verify_iri_requirements(event_state_turtle, obj_type): 

258 obj_iris.append(obj_iri)

261 event_state_turtle = event_state_turtle.replace(f'pobj(prep_{prep_of_interest})(Agent)', iri_str). \ 

262 replace(f'pobj(prep_{prep_of_interest})(Location)', iri_str). \ 

263 replace(f'pobj(prep_{prep_of_interest})', iri_str). \ 

264 replace('pobj(Agent)', iri_str).replace('pobj(Location)', iri_str).replace('pobj', iri_str) 

265 else: 

266 logging.warning(f'Pobj in processed idiom but not found in sentence, {sentence}') 

267 if event_state_turtle[-1] != '.': 

268 event_state_turtle += ' .' 

269 return event_state_turtle

272def parse_idiom(tree: Tree, sentence: str, term_dict: dict, preps: list) -> str: 

273 """ 

274 Interprets the DSL of the idiom rules to determine if the idiom applies to the sentence, and 

275 if so, to define the resulting Turtle. 

276 

277 :param tree: A Lark parse tree of a verb/noun idiom rule 

278 :param sentence: String holding the complete sentence text 

279 :param term_dict: Either a noun or a verb dictionary 

280 :param preps: For a verb idiom, an array holding tuples consisting of the preposition text, and its 

281 object text and type; For a noun idiom, the array is empty 

282 :returns: Either a string holding the Turtle for a processed idiom or an empty string (the string 

283 starts with the class name and may include Event/State properties) 

284 """ 

285 result = empty_string 

286 if tree.data == 'class_names': # Example: ":Possession" 

287 result = get_class_names(tree) 

288 if 'type' in term_dict.keys() and 'PLURAL' in term_dict['type'] and ':Collection' not in result: 

289 # Processing is a reference to 1+ noun or verb classes 

290 result += ', :Collection' 

291 elif tree.data == 'complex_rule': 

292 for paren_rule in tree.children: # There can be 1 or more parenthesized rules 

293 # Each paren_rule has 1 child (children[0]) = 'rule' and then that has 1 child with the specific rule type 

294 result = parse_idiom(paren_rule.children[0].children[0], sentence, term_dict, preps) 

295 if result: # First non-empty result should be chosen 

296 break 

297 elif tree.data == 'noun_rule': 

298 # Example: "secretary": ":Person . affiliation Secretary Organization" 

299 result = owl_thing2 # Default 

300 increment = 0 

301 needed_text = empty_string 

302 affiliation_details = [] 

303 lob_detail = empty_string 

304 if tree.children[0].data == 'text': # May have text to check for 

305 increment = 1 

306 needed_text = get_needed_text(tree.children[0]) 

307 if not needed_text or needed_text in sentence: # For the rule to be valid, needed_text must be in the sentence 

308 noun_classes = get_class_names(tree.children[increment]) 

309 if (needed_text and len(tree.children) == 3) or (not needed_text and len(tree.children) == 2): 

310 if tree.children[increment + 1].data == "affiliation_statement": 

311 affiliation_details.append(tree.children[increment + 1].children[0].children[0].value) 

312 affiliation_details.append(tree.children[increment + 1].children[1].children[0].value) 

313 else: # lob_statement 

314 lob_detail = tree.children[increment + 1].children[0].children[0].value 

315 result = noun_classes 

316 if result: 

317 if affiliation_details: 

318 result += f' . noun_iri{affiliation_details[0]}Affiliation a :Affiliation ; ' \ 

319 f':affiliated_agent noun_iri ; ' \ 

320 f':affiliated_with noun_iri{affiliation_details[1]}' 

321 if lob_detail: 

322 result += f'; :has_line_of_business :{lob_detail}' 

323 elif tree.data == 'verb_rule': # Example: "Agent dobj > :CaringForDependents ; :has_affected_agent dobj" 

324 result = event_and_state_class 

325 objects = [] 

326 verbs = empty_string 

327 other_verb = empty_string 

328 property_details = [] 

329 property_refs = [] 

330 for j in range(0, len(tree.children)): 

331 child = tree.children[j] 

332 if child.data == 'text': # May have text to check for 

333 need_text = get_needed_text(child) 

334 if need_text not in sentence: # For the rule to be valid, need_text must be in the sentence 

335 break 

336 elif child.data == 'obj_detail': 

337 obj_text = '' 

338 for k in range(0, len(child.children)): 

339 if child.children[k].data == 'entity_type': 

340 obj_text = child.children[k].children[0].data + ' ' 

341 else: 

342 obj_text += child.children[k].children[0].data 

343 objects.append(obj_text) 

344 elif child.data == 'class_names': 

345 verbs = get_class_names(child) 

346 elif child.data == 'property_detail': 

347 property_details.append( 

348 f'; :{child.children[0].children[0].children[0].value} {child.children[1].children[0].data}') 

349 elif child.data == 'property_ref': 

350 property_refs.append( 

351 f'; :{child.children[0].children[0].children[0].value} ' 

352 f':{child.children[1].children[0].children[0].value}') 

353 elif child.data == 'other_verb': 

354 other_verb = f'dobj {child.children[0].value}' 

355 if verbs: # Might have broken out of the above loop, since the need_text was not in the sentence 

356 # Need to at least have verbs to continue 

357 # Check if the rule references dobj/pobj, then there are 'objects'/'preps' lists in the term dictionary 

358 ok_to_continue = True 

359 for obj in objects: 

360 if not (('dobj' in obj and objects_string in term_dict) or 

361 ('pobj' in obj and preps_string in term_dict)): 

362 ok_to_continue = False 

363 if ok_to_continue: # Requirements are met 

364 if other_verb: # Another verb; Example: "dobj > :Attempt, dobj verb" 

365 if ':Ignore' in verbs: 

366 result = 'dobj(verb)' 

367 else: 

368 result = f'{verbs}, dobj(verb)' 

369 else: 

370 result = verbs 

371 obj_dict = dict() 

372 for obj in objects: 

373 if obj == 'dobj': 

374 obj_dict['dobj'] = obj 

375 elif ' dobj' in obj: 

376 obj_dict['dobj'] = f'dobj({obj.split(space)[0].title()})' 

377 if obj == 'pobj': 

378 obj_dict['pobj'] = obj 

379 elif ' pobj' in obj: 

380 obj_dict['pobj'] = f'pobj({obj.split(space)[0].title()})' 

381 for prop_detail in property_details: # Example from above: ":has_affected_agent dobj" 

382 if 'dobj' in prop_detail: 

383 result += prop_detail.replace('dobj', obj_dict['dobj']) 

384 if 'pobj' in prop_detail: 

385 result += prop_detail.replace('pobj', obj_dict['pobj']) 

386 for prop_detail in property_details: # Example from above: ":has_affected_agent dobj" 

387 if 'dobj' in prop_detail: 

388 result += prop_detail.replace('dobj', obj_dict['dobj']) 

389 for prop_ref in property_refs: # Example ":has_topic :PoliticalEnvironment" 

390 result += prop_ref 

391 elif tree.data == 'verb_subject_rule': 

392 # Example: "subj > :EnvironmentAndConditionLineOfBusiness ; :has_topic :Judiciary ; 

393 # :has_holder subj ; :word_detail 'attorney'" 

394 result = get_class_names(tree.children[0]) 

395 for i in range(1, len(tree.children)): 

396 if tree.children[i].data == 'topic': 

397 if tree.children[i].children[0].data == 'dobj': 

398 result += '; :has_topic dobj ' 

399 else: 

400 result += f'; :has_topic :{tree.children[i].children[0].children[0].children[0].value} ' 

401 elif tree.children[i].data == 'word_detail': 

402 result += f"; :word_detail '{tree.children[i].children[0].value}' " 

403 elif tree.children[i].data == 'holder': 

404 result += f'; :has_holder {tree.children[i].children[0].value} ' 

405 result = result.strip() 

406 elif tree.data == 'verb_preposition_rule': 

407 # Example: "prep_for (Agent pobj > :CaringForDependent ; :has_affected_agent pobj)" 

408 prep_tree, complex_rule = tree.children 

409 preposition = prep_tree.value 

410 prep_found = False 

411 if preps_string in term_dict: 

412 for prep in term_dict[preps_string]: 

413 if prep['prep_text'].lower() == preposition: 

414 prep_found = True 

415 break 

416 if prep_found: # Preposition must be found in the sentence to continue 

417 result = parse_idiom(complex_rule, sentence, term_dict, preps) 

418 result = result.replace('pobj', f'pobj(prep_{preposition})') 

419 else: 

420 result = event_and_state_class 

421 elif tree.data == 'verb_xcomp_rule': # Example: "xcomp > attempt, attack" 

422 if len(tree.children) == 2: 

423 verb1 = tree.children[0].value 

424 verb2 = tree.children[1].value 

425 else: 

426 verb1 = f'{tree.children[0].value} {tree.children[1].value}' 

427 verb2 = tree.children[2].value 

428 result = f'xcomp({verb1}, {verb2})' 

429 return result

432def process_idiom_detail(processing: list, sentence_text: str, term_dict: dict, preps: list) -> list: 

433 """ 

434 Iterate through the processing strings attempting to match to the conditions. If a match is found, 

435 translate the processing string into an ontology class and additional Turtle statements, which 

436 may require the term_dictionary to get info such as the sentence object, preposition object, ... 

437 

438 An example of a processing string is 'prep_to (Location pobj > :MovementTravelAndTransportation ; 

439 :has_destination obj_iri) | (pobj > :ReturnRecoveryAndRelease ; :has_topic obj_iri)' which states 

440 that if a preposition of 'to' is associated with the verb, and the object of the preposition is 

441 a type of Location, then the event/state is :MovementTravelAndTransportation with the Location 

442 being the destination. Or, if the object of the preposition is NOT a Location, then the event/state 

443 is :ReturnRecoveryAndRelease and the object is the entity returned (identified as the topic). 

444 

445 :param processing: An array of strings holding the possible processing 'rules'/idioms 

446 :param sentence_text: String holding the complete sentence text 

447 :param term_dict: Either a noun or a verb dictionary 

448 :param preps: For a verb idiom, an array holding tuples consisting of the preposition text, and its 

449 object text, type and IRI; For a noun idiom, the array is empty 

450 :returns: An array of strings holding the individual Turtle statements given the matched idiom 

451 (note that only the subj_rule will return multiple strings) 

452 """ 

453 logging.info(f'Processing the idiom string, {processing}, for the sentence, {sentence_text}') 

454 # Process all the individual strings, looking for a match 

455 # Once a match is found, we are finished since the processing rules should be ordered from specific to default 

456 result = [] 

457 for process_str in processing: 

458 if 'EnvironmentAndCondition' not in process_str and result: 

459 # There may be multiple EnvironmentAndCondition clauses (due to multiple objects) 

460 # Each needs to be processed 

461 # But for the other idioms, the first match is assumed to be the best and final 

462 break 

463 try: 

464 process_parse = idiom_dsl.parse(process_str) 

465 except lark_exceptions.UnexpectedCharacters: 

466 logging.error(f'Unexpected characters in idiom string') 

467 continue
