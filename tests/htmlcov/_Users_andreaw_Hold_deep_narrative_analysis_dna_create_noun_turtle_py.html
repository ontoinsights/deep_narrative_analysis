284def create_time_ttl(time_text: str, time_iri: str) -> list: 

285 """ 

286 Return the Turtle related to a point in time. 

287 

288 :param time_text: The text defining the time 

289 :param time_iri: The IRI created for the time 

290 :return: An array of Turtle statements 

291 """ 

292 time_ttl = [f'{time_iri} a :PointInTime ; rdfs:label "{time_text}" .'] 

293 if '_Yr' in time_iri: 

294 time_ttl.append(f'{time_iri} :year {time_iri.split("_Yr")[1].split("_")[0]} .') 

295 if '_Mo' in time_iri: 

296 time_ttl.append(f'{time_iri} :month_of_year ' 

297 f'{str(months.index(time_iri.split("_Mo")[1].split("_")[0]) + 1)} .') 

298 if '_Day' in time_iri: 

299 day = time_iri.split("_Day")[1] 

300 if day in days: 

301 time_ttl.append(f'{time_iri} :day_of_week {days.index(day) + 1} .') 

302 else: 

303 time_ttl.append(f'{time_iri} :day_of_month {day} .') 

304 return time_ttl
