25def _adjust_xcomp_prt(processing: list) -> list: 

26 """ 

27 Clean up the verb_processing text if there are both prt and xcomp details. If there is only a 

28 single 'xcomp' or 'prt', then just remove the final '$'. If both, correct either the first or 

29 second verb of the 'xcomp' details for one of the verbs (the one matching the prt verb) to h 

30 ave it reference the full 'prt' text (verb + prt). 

31 

32 :param processing: Array holding the xcomp and prt details 

33 :return: List holding the final xcomp/prt details 

34 """ 

35 xcomp_tuples = [] 

36 prts = [] 

37 for detail in processing: 

38 if 'xcomp' in detail: 

39 xcomp_tuples.append((detail.split('> ')[1].split(', ')[0], detail.split(', ')[1])) 

40 elif 'prt' in detail: 

41 prts.append(detail.split('> ')[1]) 

42 if xcomp_tuples and prts: 

43 new_xcomps = [] 

44 for verb1, verb2 in xcomp_tuples: 

45 for prt in prts: 

46 if prt.startswith(verb1): 

47 verb1 = prt 

48 break 

49 elif prt.startswith(verb2): 

50 verb2 = prt 

51 break 

52 new_xcomps.append(f'xcomp > {verb1}, {verb2}') 

53 return new_xcomps 

54 elif xcomp_tuples: 

55 return [f'xcomp > {verb1}, {verb2}' for verb1, verb2 in xcomp_tuples] 

56 elif prts: 

57 return [f'prt > {prt}' for prt in prts]

85def extract_chunk_details(chunk: str, chunk_dict: dict, nlp: Language) -> dict: 

86 """ 

87 Extract the subject, verb, object details from a sentence and store these in a dictionary for 

88 further processing (and rendering in Turtle). 

89 

90 :param chunk: The sentence/chunk to be processed 

91 :param chunk_dict: The dictionary holding the chunk details 

92 :param nlp: A spaCy Language model 

93 :return: Returns the updated chunk_dict 

94 """ 

95 logging.info(f'Creating chunk dictionary for {chunk}') 

96 nlp_chunk = nlp(chunk) 

97 # There should only be 1 root verb, because the text was already split 

98 process_verb([chunk.root for chunk in nlp_chunk.sents][0], chunk_dict, nlp) 

99 return chunk_dict

102def process_verb(token: Token, dictionary: dict, nlp: Language): 

103 """ 

104 When processing a token (that is a verb), capture its details. Note that all processing 

105 of a narrative is based on the sentences/chunks and their ROOT verbs. 

106 

107 :param token: Verb token from spacy parse 

108 :param dictionary: Dictionary that the token details should be added to 

109 :param nlp: A spaCy Language model 

110 :return: None (Specified dictionary is updated) 

111 """ 

112 logging.info(f'Processing the verb, {token.text}') 

113 verb_dict = dict() 

114 verb_dict['verb_text'] = token.text 

115 verb_dict['verb_lemma'] = token.lemma_ 

116 tense = token.morph.get('Tense') 

117 verb_dict['tense'] = tense[0] if tense else 'Pres' 

118 # Processing is based on dependency parsing 

119 # Dependency tokens are defined at https://downloads.cs.stanford.edu/nlp/software/dependencies_manual.pdf 

120 for child in token.children: 

121 if child.dep_ == 'xcomp': # Clausal complement - e.g., 'he attempted robbing the bank' (xcomp = robbing) 

122 # For the above, xcomp > attempt, rob 

123 add_to_dictionary_values(dictionary, 'verb_processing', f'xcomp > {token.lemma_}, {child.lemma_}', str) 

124 process_verb(child, dictionary, nlp) 

125 for child_child in child.children: 

126 if child_child.dep_ == 'conj': 

127 # Conjunction of xcomps ('Sue failed to win the race or to finish.') handled as 

128 # 3 verbs with multiple xcomp processing details 

129 add_to_dictionary_values(dictionary, 'verb_processing', 

130 f'xcomp > {token.lemma_}, {child_child.text}', str) 

131 process_verb(child_child, dictionary, nlp) 

132 elif 'prt' in child.dep_: # Phrasal verb particle - e.g., 'he gave up the jewels' (up = prt) 

133 # For the above, prt > give up 

134 add_to_dictionary_values(dictionary, 'verb_processing', f'prt > {token.lemma_} {child.text}', str) 

135 elif child.dep_ in ('acomp', 'advcl', 'ccomp'): # Various complements of the verb 

136 add_token_details(child, verb_dict, f'verb_{child.dep_}') 

137 elif 'agent' == child.dep_: # Agent of a passive verb, introduced by the word, 'by' 

138 for child2 in child.children: 

139 if 'pobj' == child2.dep_: # The real subject is the object of the preposition, 'by' 139 ↛ 138line 139 didn't jump to line 138, because the condition on line 139 was never false

140 add_token_details(child2, dictionary, subjects_string) 

141 elif 'aux' == child.dep_: 

142 # Care about modals, tense of the primary auxiliaries (be, do, have), and semantics of mood 

143 _process_aux(child, verb_dict) 

144 elif 'neg' == child.dep_: # Verb is negated 

145 verb_dict['negation'] = True 

146 elif 'nsubj' == child.dep_: # Subject of the verb 

147 add_token_details(child, dictionary, subjects_string) 

148 elif 'nsubjpass' == child.dep_: # Subject of a passive verb -> Which means that it is an object 

149 add_token_details(child, dictionary, objects_string) 

150 # TODO: Handle subjects that are themselves clauses (csubj and csubjpass) 

151 elif 'obj' in child.dep_ or child.dep_ in ('attr', 'dative'): # Object or attribute of the verb 

152 add_token_details(child, verb_dict, objects_string) 

153 elif 'prep' in child.dep_: # Preposition associated with the verb 

154 # Only concerned with certain prepositions and their semantics 

155 if child.text.lower() not in processed_prepositions: 

156 continue 

157 prep_dict = dict() 

158 prep_dict['prep_text'] = child.text 

159 for prep_dep in child.children: 

160 add_token_details(prep_dep, prep_dict, 'prep_details')
