215def get_y_x_values(number_narrators: int, variable: str, query: str, store_name: str) -> (tuple, tuple): 

216 """ 

217 Get the count of narrators by gender, birth country and birth year. Also provide counts where 

218 this info is not known. 

219 

220 :param number_narrators: Total number of narrators 

221 :param variable: String indicating the name of the returned SPARQL query variable - either 

222 'gender', 'year' or 'country' 

223 :param query: String holding the query to run to get the count 

224 :param store_name: String holding the database/data store name with the narratives and narrator details 

225 :return: Two tuples of integers - for the y and x axes of a horizontal bar chart 

226 The y axis is the list of genders, birth countries and birth years, 

227 and the x axis is the number of narrators 

228 """ 

229 dictionary = dict() 

230 # Get count 

231 success, count_results = query_database('select', query, store_name) 

232 if success: 

233 if 'results' in count_results.keys() and 'bindings' in count_results['results'].keys(): 

234 for binding in count_results['results']['bindings']: 

235 # Manipulate the key value in case it is an IRI 

236 update_dictionary_count(dictionary, str(binding[variable]['value']).split(':')[-1]) 

237 # Make sure that all narrators are addressed 

238 total_count = 0 

239 y_list = [] 

240 for key in dictionary.keys(): 

241 total_count += dictionary[key] 

242 y_list.append(key) 

243 if total_count != number_narrators: 

244 dictionary['Unknown'] = number_narrators - total_count 

245 y_list.append('Unknown') 

246 # Sort the results and create the returned tuples 

247 y_values = sorted(y_list) 

248 x_values = [] 

249 for y_value in y_values: 

250 x_values.append(dictionary[y_value]) 

251 return tuple(y_values), tuple(x_values)

255def _display_horiz_histogram(y_values: tuple, x_values: tuple, x_label: str, title: str): 

256 """ 

257 Display a horizontal bar chart/histogram using matplotlib. 

258 

259 :param y_values: The y-axis (horizontal) values 

260 :param x_values: The x-axis (vertical) values 

261 :param x_label: The labels on the x-axis (MUST correspond to the order of the values) 

262 :param title: The title of the histogram chart 

263 :return: None (Histogram is displayed) 

264 """ 

265 logging.info('Displaying histogram') 

266 # Setup the histogram 

267 plt.rcdefaults() 

268 fig, ax = plt.subplots() 

269 y_pos = np.arange(len(y_values)) 

270 ax.barh(y_pos, x_values, align='edge') 

271 ax.set_yticks(y_pos) 

272 ax.set_yticklabels(y_values) 

273 ax.invert_yaxis() # labels read top-to-bottom 

274 ax.xaxis.set_major_locator(MaxNLocator(integer=True)) 

275 ax.set_xlabel(x_label) 

276 plt.tight_layout(pad=2.0) 

277 # Display in a window 

278 layout = [[sg.Canvas(key="-CANVAS-")]] 

279 window_histogram = sg.Window(title, layout, icon=encoded_logo, element_justification='center', 

280 finalize=True) 

281 draw_figure(window_histogram["-CANVAS-"].TKCanvas, fig) 

282 # Non-blocking window 

283 window_histogram.read(timeout=0)

402def _output_words_in_csv(narratives: str, nouns_in_csv: int, verbs_in_csv: int, directory_name: str): 

403 """ 

404 Output the top xx 'unknown to the ontology' nouns and verbs in the narratives' texts 

405 to the files, <directory_name>/Nouns.csv and <directory_name>/Verbs.csv. 

406 The nouns, verbs and a count of their occurrences is returned. 

407 

408 :param narratives: String consisting of all the narratives' texts 

409 :param nouns_in_csv: Integer value indicating the number of nouns to be output 

410 :param verbs_in_csv: Integer value indicating the number of verbs to be output 

411 :param directory_name: String indicating the base file name to which the nouns, verbs and 

412 their frequencies are output 

413 :return: None (.csv files are generated) 

414 """ 

415 sorted_nouns, sorted_verbs = get_nouns_verbs(narratives) 

416 # TODO: Make words lower case 

417 with open(f'{directory_name}/Nouns.csv', 'w', newline=NEW_LINE) as noun_file: 

418 noun_writer = csv.writer(noun_file, delimiter=',') 

419 noun_writer.writerow(['Noun', 'Count']) 

420 count = 0 

421 for key, value in sorted_nouns.items(): 

422 if count < nouns_in_csv: 

423 count += 1 

424 noun_writer.writerow([key, value]) 

425 else: 

426 break 

427 with open(f'{directory_name}/Verbs.csv', 'w') as verb_file: 

428 verb_writer = csv.writer(verb_file, delimiter=',') 

429 verb_writer.writerow(['Verb', 'Count']) 

430 count = 0 

431 for key, value in sorted_verbs.items(): 

432 if count < verbs_in_csv: 

433 count += 1 

434 verb_writer.writerow([key, value]) 

435 else: 

436 break 

437 return
