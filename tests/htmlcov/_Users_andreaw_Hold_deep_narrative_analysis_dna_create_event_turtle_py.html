408 ttl_list.append(f'{event_uri} :has_location {loc_uri} .')

411 # Except for xcomp details which may get subj/obj defined when processing the ROOT verb

416 ttl_list = [f'{event_uri} :text "{sentence_text}" ; :sentence_offset {sentence_offset} .']

419 ttl_list.append(f'{event_uri} :sentiment {get_sentence_sentiment(sentence_text)} .')
