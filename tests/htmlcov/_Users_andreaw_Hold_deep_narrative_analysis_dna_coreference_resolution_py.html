17def _check_criteria(text: str, last_nouns: list, looking_for_singular: Union[bool, None], 

18 looking_for_female: Union[bool, None], looking_for_person: bool) -> list: 

19 """ 

20 Checks the values of the nouns in the last_nouns array for matches of the specified 

21 gender/number criteria. 

22 

23 :param text: A string with the noun text 

24 :param last_nouns: A list of noun texts, types, class mappings and IRIs, from the current paragraph 

25 :param looking_for_singular: Boolean indicating that a singular noun is needed 

26 :param looking_for_female: Boolean indicating that a female gender noun is needed 

27 :param looking_for_person: Boolean indicating that a 'matched' noun should be a person 

28 :return: Array of tuples of texts, types, class_mappings and IRIs of already processed nouns that 

29 match the criteria; Note that an array is returned to support matching the pronouns 'they'/'them' 

30 """ 

31 poss_nouns = [] 

32 alt_nouns = [] 

33 for noun_tuple in last_nouns: 

34 noun_text, noun_type, noun_mapping, noun_iri = noun_tuple 

35 if text not in pronouns and (text not in noun_text or noun_text not in text): 

36 continue # First match the text if not a pronoun; If no match, skip the rest of the criteria 

37 if (looking_for_person and 'PERSON' not in noun_type) or \ 

38 (not looking_for_person and 'PERSON' in noun_type): 

39 continue 

40 # Check number 

41 found_number = False 

42 if looking_for_singular is None or (looking_for_singular and 'SING' in noun_type) or \ 

43 (not looking_for_singular and 'PLURAL' in noun_type): 

44 found_number = True 

45 found_gender = False 

46 if looking_for_female is None or (looking_for_female and 'FEMALE' in noun_type) or \ 

47 (not looking_for_female and 'FEMALE' not in noun_type and 'MALE' in noun_type): 

48 found_gender = True 

49 # Check criteria 

50 if found_gender and found_number: 

51 poss_nouns.append(noun_tuple) 

52 elif found_gender or found_number: 

53 alt_nouns.append(noun_tuple) 

54 match_nouns = poss_nouns if poss_nouns else (alt_nouns if alt_nouns else []) 

55 if looking_for_singular and len(match_nouns) > 1: 

56 return [match_nouns[-1]] 

57 return match_nouns

60def _check_last_nouns(text: str, text_type: str, last_nouns: list) -> list: 

61 """ 

62 Get the most likely co-reference for the noun text using the last_nouns details. 

63 Subject/object information (the noun, and its type, mapping and IRI) is returned. 

64 

65 :param text: String holding the noun text 

66 :param text_type: String holding the noun type (such as 'FEMALESINGPERSON') 

67 :param last_nouns: An array of tuples of noun texts, types, mappings and IRI, from the current paragraph 

68 :return: A tuple that is the 'matched' noun mapping and IRI, or two empty strings (if 

69 no match is found) 

70 """ 

71 looking_for_female = True if 'FEMALE' in text_type else (False if 'MALE' in text_type else None) 

72 looking_for_singular = False if 'PLURAL' in text_type else True # Default is singular 

73 looking_for_person = True if 'PERSON' in text_type else False 

74 match_nouns = _check_criteria(text, last_nouns, looking_for_singular, looking_for_female, looking_for_person) 

75 final_nouns = [] 

76 for match_noun in match_nouns: 

77 # Don't need the noun texts or entity types (those are used for pronouns) 

78 final_nouns.append((match_noun[2], match_noun[3])) 

79 return final_nouns

82def _check_plet_dict(text: str, text_type: str, plet_dict: dict, last_nouns: list) -> (list, str): 

83 """ 

84 Get the most likely co-reference for the text using the plet_dict details to 

85 resolve co-references/anaphora. Subject/object information (the noun, and its types 

86 and IRI) is returned. 

87 

88 :param text: String holding the noun text 

89 :param text_type: String holding the noun type (such as 'FEMALESINGPERSON') 

90 :param plet_dict: A dictionary holding the persons, locations, events & times encountered in 

91 the full narrative - For co-reference resolution; Keys = 'persons', 'locs', 'events', 

92 'times' and Values vary by the key 

93 :param last_nouns: An array of tuples of noun texts, types, mappings and IRI, from the current paragraph 

94 :return: A tuple that consists of the matched noun's class mappings and IRI, or an empty array and string 

95 """ 

96 person_match = [] # Match of text and type 

97 person_text_match = [] # Match of text only, not type 

98 loc_text_match = [] 

99 event_text_match = [] 

100 if not text_type or 'PERSON' in text_type: # Type is sometimes missing in the spaCy nlp doc details 

101 person_arrays = plet_dict['persons'] if 'persons' in plet_dict else [] 

102 for person_array in person_arrays: 

103 alt_names = person_array[0] 

104 person_type = person_array[1] 

105 if text not in pronouns and text in alt_names: 

106 if text_type and (text_type in person_type or person_type in text_type): 

107 person_match.append((person_type, person_array[2])) # index 2 holds the IRI 

108 else: 

109 person_text_match.append((person_type, person_array[2])) 

110 if not text_type or ('LOC' in text_type or 'GPE' in text_type or 'FAC' in text_type): 

111 loc_arrays = plet_dict['locs'] if 'locs' in plet_dict else [] 

112 for loc_array in loc_arrays: 

113 alt_names = loc_array[0] 

114 loc_map = loc_array[1] 

115 if text in alt_names: 

116 loc_text_match.append((loc_map, loc_array[2])) # index 2 holds the IRI 

117 if not text_type or 'EVENT' in text_type: 

118 event_arrays = plet_dict['events'] if 'events' in plet_dict else [] 

119 for event_array in event_arrays: 

120 alt_names = event_array[0] 

121 if text in alt_names: 

122 # event_array[1] holds the class mappings and [2] holds the IRI 

123 event_text_match.append((event_array[1], event_array[2])) 

124 return (_update_last_nouns(text, person_match[-1][0], person_match[-1][1], [':Person'], last_nouns) if person_match 

125 else (_update_last_nouns(text, person_text_match[-1][0], person_text_match[-1][1], [':Person'], 

126 last_nouns) if person_text_match 

127 else (_update_last_nouns(text, text_type, loc_text_match[-1][1], loc_text_match[-1][0], last_nouns) 

128 if loc_text_match 

129 else (_update_last_nouns(text, text_type, event_text_match[-1][1], event_text_match[-1][0], 

130 last_nouns) if event_text_match else [], empty_string))))

133def _check_pronouns(pronoun: str, last_nouns: list) -> list: 

134 """ 

135 Get the most likely co-reference(s) for the pronoun using the last_nouns details. 

136 Subject/object information (the noun, and its types, mappings and IRI) is returned. 

137 

138 :param pronoun: String holding the pronoun text 

139 :param last_nouns: An array of tuples of noun texts, types, mappings and IRI, from the current paragraph 

140 :return: Array of tuples of class_mappings and IRIs of already processed nouns that match the criteria; 

141 Note that an array is returned to support matching the pronouns 'they'/'them' 

142 """ 

143 pronoun_details = [] 

144 pronoun_lower = pronoun.lower() 

145 if pronoun == 'I' or pronoun_lower in ('me', 'myself'): 

146 pronoun_details.append(('Narrator', 'SINGPERSON', ':Person', ':Narrator')) 

147 elif pronoun_lower in ('we', 'us', 'ourselves'): 

148 # Find singular or plural person nouns (any gender) 

149 pronoun_details.extend(_check_criteria(pronoun_lower, last_nouns, None, None, True)) 

150 pronoun_details.append(('Narrator', 'SINGPERSON', ':Person', ':Narrator')) 

151 elif pronoun_lower in ('they', 'them', 'themselves'): 

152 # Give preference to persons (any gender or number) 

153 noun_list = _check_criteria(pronoun_lower, last_nouns, None, None, True) 

154 if noun_list: 

155 pronoun_details.extend(noun_list) 

156 else: 

157 # Check for non-persons 

158 pronoun_details.extend(_check_criteria(pronoun_lower, last_nouns, None, None, False)) 

159 elif pronoun_lower in ('she', 'herself'): 

160 # Find singular, feminine, person nouns 

161 pronoun_details.extend(_check_criteria(pronoun_lower, last_nouns, True, True, True)) 

162 elif pronoun_lower in ('he', 'himself'): 

163 # Find singular, masculine, person nouns 

164 pronoun_details.extend(_check_criteria(pronoun_lower, last_nouns, True, False, True)) 

165 elif pronoun_lower in ('it', 'itself'): 

166 # Find singular, non-person nouns (no gender) 

167 pronoun_details.extend(_check_criteria(pronoun_lower, last_nouns, True, None, False)) 

168 return pronoun_details

171def _update_last_nouns(text: str, text_type: str, text_iri: str, class_maps: list, last_nouns: list) -> (list, str): 

172 """ 

173 Update the last_nouns array and return the class mappings and IRI. 

174 

175 :param text: String holding the noun text 

176 :param text_type: String holding the noun type (such as 'FEMALESINGPERSON') 

177 :param text_iri: String holding the noun IRI 

178 :param class_maps: An array of strings holding the mapping(s) to the DNA ontology for the text 

179 :param last_nouns: An array of tuples of noun texts, types, mappings and IRI, from the current paragraph 

180 :return: A tuple that consists of the matched noun's class mappings and IRI, or an empty array and string 

181 """ 

182 last_nouns.append((text, text_type, class_maps, text_iri)) 

183 return class_maps, text_iri

186def check_event(text: str, last_events: list) -> (list, str): 

187 """ 

188 Get a possible verb/event mapping for the noun and check it against any events 

189 (from the current paragraph) that have a type = mapping. 

190 

191 :param text: The text which is possibly mapped to an event 

192 :param last_events: The list/array of tuples defining event types and IRIs from the 

193 current paragraph 

194 :return: A tuple specifying the event class mappings and IRI if there is a type match, 

195 or an empty list and string otherwise 

196 """ 

197 ontol_classes, noun_ttl = get_noun_mapping(text, empty_string) # Get the event class to which the noun is mapped 

198 if not ontol_classes: 

199 return [], empty_string 

200 poss_events = [] 

201 for event_type, event_iri in last_events: 

202 if event_type in ontol_classes: 

203 poss_events.append(event_iri) 

204 if poss_events: 

205 return ontol_classes, poss_events[-1] 

206 return [], empty_string

209def check_nouns(elem_dictionary: dict, key: str, plet_dict: dict, last_nouns: list, last_events: list, 

210 turtle: list, ext_sources: bool) -> list: 

211 """ 

212 Get the subject or object nouns (as indicated by the key input parameter) in the dictionary, 

213 using last_nouns, plet_dict and last_events details to attempt to resolve co-references/anaphora. 

214 Subject/object information (the nouns and their types and IRIs) is returned. 

215 

216 The order of checking for a match is last_nouns, plet_dict and then last_events. If there are no 

217 matches, a new noun is created and added to either last_nouns or last_events. 

218 

219 For example, consider the sentence/chunk "She was sickly." following "Mary was born on June 12, 

220 1972, in Znojmo, Czechia." If the function parameters are (chunk_dictionary, 'subjects',  

221 plet_dict, last_events), then the tuple, 'Mary', 'FEMALESINGPERSON' and ':Mary' will be returned 

222 since 'she' should be resolved to Mary. 

223 

224 :param elem_dictionary: The dictionary (holding the details for the noun text and type from the spaCy parse) 

225 :param key: Either 'subjects' or 'objects' 

226 :param plet_dict: A dictionary holding the persons, locations, events & times encountered in 

227 the full narrative - For co-reference resolution; Keys = 'persons', 'locs', 'events',  

228 'times' and Values vary by the key 

229 :param last_nouns: An array of tuples of noun texts, types, class mappings and IRIs, 

230 found in the current paragraph 

231 :param last_events: An array of verb texts, mappings and IRIs from the current paragraph 

232 :param turtle: A list of Turtle statements which will be updated in this function if a new 

233 noun is found 

234 :param ext_sources: A boolean indicating that data from GeoNames, Wikidata, etc. should be 

235 added to the parse results if available 

236 :return: A tuple holding an array of tuples of the noun's texts, types, mappings and IRIs (also, the last_nouns and 

237 last_events arrays may be updated) 

238 """ 

239 nouns = [] 

240 for elem in elem_dictionary[key]: # The subject or object nouns 

241 elem_key = key[0:-1] # Create dictionary key = 'subject' or 'object' 

242 elem_text = elem[f'{elem_key}_text'] 

243 elem_type = elem[f'{elem_key}_type'] 

244 if elem_type == 'CARDINAL': # For example, 'one of the band' 

245 if 'preps' in elem: 

246 # Assemble the full text to aid in the ontology mapping 

247 for prep in elem['preps']: 

248 # TODO: Assess if more than the first detail_text is needed, since it is an array 

249 elem_text += f' {prep["prep_text"]} {prep["prep_details"][0]["detail_text"]}' 

250 iri = f":{elem_text.replace(' ', '_')}_{str(uuid.uuid4())[:13]}" 

251 nouns.append((elem_text, 'CARDINAL', [owl_thing2], iri)) 

252 turtle.extend([f'{iri} a owl:Thing .', 

253 f'{iri} rdfs:label "{elem_text}" .']) 

254 continue 

255 if elem_text.lower() in pronouns: 

256 # Array of tuples of matched text, type, mappings and IRIs 

257 nouns.extend(_check_pronouns(elem_text, last_nouns)) 

258 continue 

259 # Not a pronoun; Check for a match in instances of the ontology 

260 if ('PERSON' in elem_type or elem_type.endswith('GPE') or 

261 elem_type.endswith('ORG') or elem_type.endswith('NORP')): 

262 match_iri, match_type = check_specific_match(elem_text, elem_type) 

263 if match_iri: 

264 # Tuple = noun text, type and IRI 

265 nouns.append((elem_text, elem_type, match_type, match_iri)) 

266 continue 

267 # No match - Try to match text and type in last_nouns and then the plet_dict 

268 match_noun_tuples = _check_last_nouns(elem_text, elem_type, last_nouns) 

269 if match_noun_tuples: 

270 nouns.append((elem_text, elem_type, match_noun_tuples[0][0], match_noun_tuples[0][1])) 

271 continue 

272 match_maps, match_iri = _check_plet_dict(elem_text, elem_type, plet_dict, last_nouns) # May update last_nouns 

273 if match_iri: 

274 nouns.append((elem_text, elem_type, match_maps, match_iri)) 

275 continue 

276 # No match - Check if the noun is aligned with an event that has already been described 

277 event_classes, event_iri = check_event(elem_text, last_events) 

278 if event_iri: 

279 # Tuple = string = 'event' and the noun text, event class mapping and IRI 

280 nouns.append((elem_text, elem_type, event_classes, event_iri)) 

281 continue 

282 # No match - Create new entity 

283 # TODO: possessives - 'her father', 'her arm' or "Sue's arm" (versus 'her book') 

284 iri = f":{elem_text.replace(' ', '_')}_{str(uuid.uuid4())[:13]}" 

285 # Tuple = noun text, type and IRI 

286 noun_mappings, noun_turtle = create_noun_ttl(iri, elem_text, elem_type, 

287 True if key == 'subjects' else False, ext_sources) 

288 nouns.append((elem_text, elem_type, noun_mappings, iri)) 

289 turtle.extend(noun_turtle) 

290 last_nouns.append((elem_text, elem_type, noun_mappings, iri)) 

291 return nouns

294def check_specific_match(noun: str, noun_type: str) -> (str, str): 

295 """ 

296 Checks if the concept/Person/Location/... is already defined in the DNA ontologies. 

297 

298 :param noun: String holding the text to be matched 

299 :param noun_type: String holding the noun type (PERSON/GPE/LOC/...) from spacy's NER 

300 :return: A tuple consisting of the matched IRI and its class mapping (if a match is found), 

301 or two empty strings 

302 """ 

303 if noun_type.endswith('GPE') and noun in names_to_geo_dict: 

304 return f'geo:{names_to_geo_dict[noun]}', ':Country' 

305 class_type = 'Person' if noun_type.endswith('PERSON') else \ 

306 ('OrganizationalEntity' if noun_type.endswith('ORG') else 

307 ('GeopoliticalEntity' if noun_type.endswith('GPE') else 

308 ('Location' if noun_type.endswith('LOC') else 

309 ('EventAndState' if noun_type.endswith('EVENT') else empty_string)))) 

310 match_details = query_database( 

311 'select', query_specific_noun.replace('keyword', noun).replace('class_type', class_type), ontologies_database) 

312 if match_details: 

313 return match_details[0]['iri']['value'].replace(dna_prefix, ':'), match_details[0]['type']['value'] 

314 return empty_string, empty_string
