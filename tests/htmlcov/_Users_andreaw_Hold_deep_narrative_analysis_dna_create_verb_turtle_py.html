145def handle_xcomp_processing(verb1_tuple: tuple, chunk_dict: dict, plet_dict: dict, last_nouns: list, 

146 last_events: list, turtle: list, event_iri: str, ext_sources: bool) -> list: 

147 """ 

148 Handles xcomp processing for a sentence, which requires adjusting the subject/object, 

149 and more. 

150 

151 :param verb1_tuple: The text of the root verb and mapping to the DNA ontology 

152 :param chunk_dict: A dictionary holding details about the chunk 

153 :param plet_dict: A dictionary holding the persons, locations, events & times encountered in 

154 the narrative to-date - For co-reference resolution (it may be updated by this function); 

155 Keys = 'persons', 'locs', 'events', 'times' and Values that vary for the different keys 

156 :param last_nouns: An array of noun texts, type, class mapping and IRI from the current paragraph 

157 :param last_events: A list/array of tuples defining an event DNA class mapping and IRI, found in the 

158 current paragraph 

159 :param turtle: The current Turtle definition for the sentence (which may be updated) 

160 :param event_iri: The xcomp event's IRI 

161 :param ext_sources: A boolean indicating that data from GeoNames, Wikidata, etc. should be 

162 added to the parse results if available 

163 :return: An array of Turtle statements intended to replace the existing verb Turtle 

164 """ 

165 verb1_text, verb1_mapping = verb1_tuple # Mappings are arrays due to multiple inheritance 

166 # If root verb = Attempt or Continuation, End, ..., then can do simple multiple inheritance 

167 if verb1_mapping[0] in (':Attempt', ':Continuation', ':StartAndBeginning', ':End', ':Success', ':Failure'): 

168 turtle.append(f'{event_iri} a {verb1_mapping[0]} .') 

169 return turtle 

170 # Otherwise, create new event IRI for the scoping event/root verb (e.g., "permitted to leave") 

171 new_iri = f':Event_{str(uuid.uuid4())[:13]}' 

172 # And create the Turtle 

173 new_ttl = create_type_turtle(verb1_mapping, new_iri, True, empty_string) 

174 new_ttl.append(f'{new_iri} :has_topic {event_iri} .') 

175 root_objects = [] # Account for the objects of the root verb 

176 for verb in chunk_dict['verbs']: 

177 if verb['verb_lemma'] == verb1_text and objects_string in verb: 

178 add_unique_to_array(check_nouns(verb, objects_string, plet_dict, last_nouns, last_events, 

179 new_ttl, ext_sources), root_objects) 

180 # TODO: Process prepositions for the root verb 

181 add_subj_obj_to_ttl(new_iri, [], root_objects, new_ttl) 

182 # Use the existing the Turtle to add the details for the new scoping event and to correct the xcomp event 

183 turtle_text = str(turtle) 

184 has_affected = True if ':has_affected_agent' in turtle_text else False 

185 for turtle_stmt in turtle: 

186 # Deal with subjects, objects: 

187 # - If 'affected_agent', the root verb is passive and its subj becomes the 'active_agent' of the xcomp verb 

188 # - If the root verb has its own objects, these become the 'active_agent's of the xcomp verb 

189 # - If there is no 'affected_agent' and no root objects, the subject of the xcomp and root verbs are the same 

190 if ':has_affected_agent' in turtle_stmt: 

191 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split("has_")[1]}') 

192 new_ttl.append(turtle_stmt.replace('affected_agent', 'active_agent')) 

193 elif ':has_active_agent' in turtle_stmt: 

194 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

195 if root_objects: 

196 # Add root objects as subjects for the xcomp verb/event 

197 for root_text, root_type, root_mappings, root_iri in root_objects: 

198 new_ttl.append(f'{event_iri} :has_active_agent {root_iri} .') 

199 else: 

200 new_ttl.append(turtle_stmt) 

201 elif ':has_agent' in turtle_stmt: 

202 new_ttl.append(turtle_stmt) 

203 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

204 # Deal with time - the root and xcomp verbs may have same time or the root verb precedes the xcomp 

205 # For ex, "she loved to play" (same relative time) vs. "she was permitted to leave" 

206 # (permitted before leaving, leaving may not happen) 

207 elif ':has_latest_end' in turtle_stmt: 

208 time_iri = turtle_stmt.split(':has_latest_end ')[1].split(' .')[0] 

209 new_ttl.append(f'{new_iri} :has_latest_end {time_iri} .') 

210 new_ttl.append(turtle_stmt if not has_affected else new_ttl.append(f'{event_iri} :has_time {time_iri} .')) 

211 elif ':has_earliest_beginning' in turtle_stmt: 

212 new_ttl.append(turtle_stmt) 

213 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

214 elif ':has_time ' in turtle_stmt: 

215 new_ttl.append(turtle_stmt) 

216 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

217 # Deal with location 

218 elif has_location in turtle_stmt: 

219 new_ttl.append(turtle_stmt) 

220 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

221 elif ':has_origin ' in turtle_stmt: 

222 new_ttl.append(turtle_stmt) 

223 new_ttl.append(f'{new_iri} {has_location} {turtle_stmt.split(":has_origin")[1]}') 

224 # Deal with label and a few more properties for the scoping event 

225 elif ':text ' in turtle_stmt: 

226 new_ttl.append(turtle_stmt) 

227 new_ttl.append(f'{new_iri} :text {turtle_stmt.split(":text ")[1]}') 

228 elif 'rdfs:label ' in turtle_stmt: 

229 new_ttl.append(turtle_stmt) 

230 new_ttl.append(f'{new_iri} rdfs:label {turtle_stmt.split("rdfs:label ")[1]}') 

231 elif ':sentiment ' in turtle_stmt: 

232 new_ttl.append(f'{new_iri} :sentiment {turtle_stmt.split(":sentiment ")[1]}') 

233 else: 

234 new_ttl.append(turtle_stmt) # For everything else, there is no change to the Turtle 

235 # The xcomp event may not occur (e.g., 'she was asked to leave' - but she might not leave) 

236 new_ttl.append(f'{event_iri} a :OpportunityAndPossibility .') 

237 return new_ttl
