15def add_subj_obj_to_ttl(event_iri: str, subjs: list, objs: list, ttl_list: list): 

16 """ 

17 Given the details related to the subjects/objects of a sentence, capture the 

18 corresponding Turtle that relates the concepts to a chunk's verb (identified by 

19 the event_iri). 

20 

21 :param event_iri: An IRI identifying the verb event 

22 :param subjs: Array of tuples that are the subject nouns' text, type, mappings and IRI 

23 :param objs: Array of tuples that are the object nouns' text, type, mappings and IRI 

24 :param ttl_list: An array of Turtle statements capturing the semantics of the chunk 

25 :return: None (ttl_list is updated) 

26 """ 

27 ttl_str = str(ttl_list) 

28 for subj_text, subj_type, subj_mappings, subj_iri in subjs: 

29 if ':Affiliation' in ttl_str: 

30 ttl_list.append(f'{event_iri} :affiliated_agent {subj_iri} .') 

31 else: 

32 ttl_list.append(f'{event_iri} :has_active_agent {subj_iri} .') 

33 for obj_text, obj_type, obj_mappings, obj_iri in objs: 

34 is_agent = False 

35 if 'PERSON' in obj_type or obj_type.endswith('GPE') or obj_type.endswith('ORG') \ 

36 or obj_type.endswith('NORP'): 

37 is_agent = True 

38 if not is_agent: # May not have a type, so have to check the class mappings 

39 for obj_map in obj_mappings: 

40 if ',' not in obj_map and ('Agent' in obj_map or 

41 check_subclass(f'{dna_prefix}{obj_map[1:]}', 'Agent')): 

42 is_agent = True 

43 break 

44 # Deal with a comma separated list of classes (multiple inheritance) 

45 indiv_classes = obj_map.split(', ') 

46 for indiv_class in indiv_classes: 

47 if check_subclass(f'{dna_prefix}{indiv_class[1:]}', 'Agent'): 

48 is_agent = True 

49 break 

50 if is_agent: 

51 break 

52 if is_agent: 

53 if ':Affiliation' in ttl_str: 

54 ttl_list.append(f'{event_iri} :affiliated_with {obj_iri} .') 

55 else: 

56 ttl_list.append(f'{event_iri} :has_affected_agent {obj_iri} .') 

57 else: 

58 ttl_list.append(f'{event_iri} :has_topic {obj_iri} .') 

59 return

62def create_ttl_for_prep_detail(prep_detail: tuple, prepositions: list, event_iri: str, alet_dict: dict, 

63 last_nouns: list, last_events: list, objects: list, ext_sources: bool) -> (list, str): 

64 """ 

65 Parse the details for a verb's prepositions and create the corresponding Turtle. Note that 

66 dates/times are not handled in this code, but for the sentence overall. 

67 

68 :param prep_detail: The text of the object of a preposition 

69 :param prepositions: An array of tuples holding the preposition text, object text, object type, 

70 object class mappings, and object IRI 

71 :param event_iri: The IRI for the verb/event 

72 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

73 the narrative to-date - For co-reference resolution (it may be updated by this function); 

74 Keys = 'agents', 'locs', 'events', 'times' and Values that vary for the different keys 

75 :param last_nouns: An array of noun texts, type, class mapping and IRI from the current paragraph 

76 :param last_events: A list of all event types and their IRIs from the current paragraph 

77 :param objects: An array of the direct objects of the chunk, to which the prepositional objects 

78 are added; The array is a list of tuples holding an array of tuples of the 

79 noun's texts, types, mappings and IRIs 

80 :param ext_sources: Boolean indicating whether additional information on a noun should 

81 be retrieved from Wikidata (recommended) 

82 :return: An array holding the Turtle statements describing the preposition object (and the 

83 prepositions array is updated); Also if the prepositional object is location-related, 

84 and the preposition indicates that a new location is discussed, that location's IRI is 

85 returned (also, the objects array is updated) 

86 """ 

87 # TODO: Get full prepositional text ('with aid of the local police chief') 

88 prep_turtle = [] 

89 prep_text, prep_obj_text, prep_obj_type = prep_detail 

90 prep_mapping = check_nouns( 

91 {'objects': [{'object_text': prep_obj_text, 'object_type': prep_obj_type}]}, 'objects', 

92 alet_dict, last_nouns, last_events, prep_turtle, ext_sources) 

93 # Should only be 1 tuple in prep_details array where 2nd value is the updated obj type, 3rd value is 

94 # an array of class mappings and the 4th value is the IRI 

95 prep_obj_type = prep_mapping[0][1] 

96 prep_obj_classes = prep_mapping[0][2] 

97 prep_obj_iri = prep_mapping[0][3] 

98 objects.append((prep_obj_text, prep_obj_type, prep_obj_classes, prep_obj_iri)) 

99 prepositions.append((prep_text, prep_obj_text, prep_obj_type, prep_obj_classes, prep_obj_iri)) 

100 # TODO: Relationships for an Agent are different from associations for other "things"; Is more needed? 

101 new_loc_iri = empty_string 

102 if 'PERSON' in prep_obj_type or 'Agent' in prep_obj_classes or 'ORG' in prep_obj_type: 

103 if prep_text == 'from': 

104 prep_turtle.append(f'{event_iri} :has_provider {prep_obj_iri} .') 

105 elif prep_text == 'to': 

106 prep_turtle.append(f'{event_iri} :has_recipient {prep_obj_iri} .') 

107 elif prep_text in ('for', 'about'): 

108 prep_turtle.append(f'{event_iri} :has_affected_agent {prep_obj_iri} .') 

109 elif prep_text == 'with': 

110 prep_turtle.append(f'{event_iri} :has_active_agent {prep_obj_iri} .') 

111 else: 

112 if prep_text == 'with': 

113 prep_turtle.append(f'{event_iri} :has_instrument {prep_obj_iri} .') 

114 elif prep_text in prep_to_predicate_for_locs and \ 

115 (prep_obj_type.endswith('LOC') or prep_obj_type.endswith('GPE') or prep_obj_type.endswith('FAC')): 

116 if prep_text in ('at', 'in', 'near', 'outside', 'to'): 

117 new_loc_iri = prep_obj_iri 

118 prep_turtle.append(f'{event_iri} {prep_to_predicate_for_locs[prep_text]} {prep_obj_iri} .') 

119 elif prep_text in ('about', 'from', 'in', 'of', 'to'): 

120 prep_turtle.append(f'{event_iri} :has_topic {prep_obj_iri} .') 

121 return prep_turtle, new_loc_iri

124def handle_locations(turtle: list, loc_iri: str, loc_time_iris: list, event_iri: str, is_bio: bool): 

125 """ 

126 Add an origin location for a Movement event (if possible and one is not already defined) and 

127 determine if an Event/State should have a location. 

128 

129 :param turtle: The current Turtle definition for the sentence 

130 :param loc_iri: A string holding a new location for the current event 

131 :param loc_time_iris: An array holding the last processed location (index 0) and time (index 1) 

132 :param event_iri: A string holding the IRI identifier for the sentence's verb/event 

133 :param is_bio: A boolean indicating that the text is biographical/autobiographical, and so 

134 may consistently define/use location references 

135 :return: N/A (the turtle and loc_time_iris arrays may be updated) 

136 """ 

137 ttl_str = str(turtle) 

138 # Get the origin for a MovementTravelAndTransportation event 

139 # Origin is the location defined by the preposition 'from' OR the last location from processed_locs 

140 if ':Movement' in ttl_str and 'has_origin' not in ttl_str: 

141 if loc_time_iris[0]: 

142 turtle.append(f'{event_iri} :has_origin {loc_time_iris[0]} .') 

143 loc_time_iris[0] = loc_iri if loc_iri else loc_time_iris[0] # Reset with the 'new' last known loc if available 

144 if is_bio and 'has_location' not in ttl_str and 'has_destination' not in ttl_str: 

145 # No location - so add one 

146 if loc_iri: 

147 turtle.append(f'{event_iri} :has_location {loc_iri} .') 

148 else: 

149 turtle.append(f'{event_iri} :has_location {loc_time_iris[0]} .') 

150 return

153def handle_xcomp_processing(root_text: str, root_map: list, chunk_dict: dict, alet_dict: dict, 

154 last_nouns: list, last_events: list, turtle: list, event_iri: str, 

155 ext_sources: bool) -> list: 

156 """ 

157 Handles xcomp processing for a sentence, which requires adjusting the subject/object, 

158 and more. 

159 

160 :param root_text: The text of the root verb 

161 :param root_map: An array holding the mapping of the root verb to the DNA ontology 

162 :param chunk_dict: A dictionary holding details about the chunk 

163 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

164 the narrative to-date - For co-reference resolution (it may be updated by this function); 

165 Keys = 'agents', 'locs', 'events', 'times' and Values that vary for the different keys 

166 :param last_nouns: An array of noun texts, type, class mapping and IRI from the current paragraph 

167 :param last_events: A list/array of tuples defining an event DNA class mapping and IRI, found in the 

168 current paragraph 

169 :param turtle: The current Turtle definition for the sentence (which may be updated) 

170 :param event_iri: The xcomp event's IRI 

171 :param ext_sources: A boolean indicating that data from GeoNames, Wikidata, etc. should be 

172 added to the parse results if available 

173 :return: An array of Turtle statements intended to replace the existing verb Turtle 

174 """ 

175 # If root verb has a DNA mapping of Attempt or Continuation, End, ..., then can do simple multiple inheritance 

176 if root_map[0] in (':Attempt', ':Continuation', ':StartAndBeginning', ':End', ':Success', ':Failure'): 

177 turtle.append(f'{event_iri} a {root_map[0]} .') 

178 return turtle 

179 # Otherwise, create new event IRI for the scoping event/root verb (e.g., "permitted to leave") 

180 new_iri = f':Event_{str(uuid.uuid4())[:13]}' 

181 # And create the Turtle 

182 new_ttl = create_type_turtle(root_map, new_iri, True, root_text) 

183 new_ttl.append(f'{new_iri} :has_topic {event_iri} .') 

184 root_objects = [] # Account for the objects of the root verb 

185 for verb in chunk_dict['verbs']: 

186 if verb['verb_lemma'] == root_text and objects_string in verb: 

187 add_unique_to_array(check_nouns(verb, objects_string, alet_dict, last_nouns, 

188 last_events, new_ttl, ext_sources), root_objects) 

189 # TODO: Also process prepositions for the root verb (this is only processing the xcomp verb) 

190 add_subj_obj_to_ttl(new_iri, [], root_objects, new_ttl) 

191 # Use the existing the Turtle to add the details for the new scoping event and to correct the xcomp event 

192 turtle_text = str(turtle) 

193 has_affected = True if ':has_affected_agent' in turtle_text else False 

194 for turtle_stmt in turtle: 

195 # Deal with subjects, objects: 

196 # - If 'affected_agent', the root verb is passive and its subj becomes the 'active_agent' of the xcomp verb 

197 # - If the root verb has its own objects, these become the 'active_agent's of the xcomp verb 

198 # - If there is no 'affected_agent' and no root objects, the subject of the xcomp and root verbs are the same 

199 if ':describes' in turtle_stmt: # Chunk also describes the root verb/event 

200 new_ttl.append(turtle_stmt) 

201 new_ttl.append(f'{turtle_stmt.split(" :describes")[0]} :describes {new_iri} .') 

202 elif ':has_affected_agent' in turtle_stmt: 

203 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split("has_")[1]}') 

204 new_ttl.append(turtle_stmt.replace('affected_agent', 'active_agent')) 

205 elif ':has_active_agent' in turtle_stmt: 

206 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

207 if root_objects: 

208 # Add root objects as subjects for the xcomp verb/event 

209 for root_text, root_type, root_mappings, root_iri in root_objects: 

210 new_ttl.append(f'{event_iri} :has_active_agent {root_iri} .') 

211 else: 

212 new_ttl.append(turtle_stmt) 

213 elif ':has_agent' in turtle_stmt: 

214 new_ttl.append(turtle_stmt) 

215 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

216 # Deal with time - the root and xcomp verbs may have same time or the root verb precedes the xcomp 

217 # For ex, "she loved to play" (same relative time) vs. "she was permitted to leave" 

218 # (permitted before leaving, leaving may not happen) 

219 elif ':has_latest_end' in turtle_stmt: 

220 time_iri = turtle_stmt.split(':has_latest_end ')[1].split(' .')[0] 

221 new_ttl.append(f'{new_iri} :has_latest_end {time_iri} .') 

222 new_ttl.append(turtle_stmt if not has_affected else new_ttl.append(f'{event_iri} :has_time {time_iri} .')) 

223 elif ':has_earliest_beginning' in turtle_stmt: 

224 new_ttl.append(turtle_stmt) 

225 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

226 elif ':has_time ' in turtle_stmt: 

227 new_ttl.append(turtle_stmt) 

228 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

229 # Deal with location 

230 elif has_location in turtle_stmt: 

231 new_ttl.append(turtle_stmt) 

232 new_ttl.append(f'{new_iri} :has_{turtle_stmt.split(":has_")[1]}') 

233 elif ':has_origin ' in turtle_stmt: 

234 new_ttl.append(turtle_stmt) 

235 new_ttl.append(f'{new_iri} {has_location} {turtle_stmt.split(":has_origin")[1]}') 

236 # Deal with label and a few more properties for the scoping event 

237 elif ':text ' in turtle_stmt: 

238 new_ttl.append(turtle_stmt) 

239 new_ttl.append(f'{new_iri} :text {turtle_stmt.split(":text ")[1]}') 

240 elif 'rdfs:label ' in turtle_stmt and turtle_stmt.startswith(':Event'): 

241 new_ttl.append(turtle_stmt) 

242 new_ttl.append(f'{new_iri} rdfs:label {turtle_stmt.split("rdfs:label ")[1]}') 

243 elif ':sentiment ' in turtle_stmt: 

244 new_ttl.append(f'{new_iri} :sentiment {turtle_stmt.split(":sentiment ")[1]}') 

245 else: 

246 new_ttl.append(turtle_stmt) # For everything else, there is no change to the Turtle 

247 # The xcomp event may not occur (e.g., 'she was asked to leave' - but she might not leave) 

248 new_ttl.append(f'{event_iri} a :OpportunityAndPossibility .') 

249 return new_ttl
