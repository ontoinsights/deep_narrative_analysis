32def display_narratives(store_name): 

33 """ 

34 Display a list of all narratives in the specified store and allow selection of one. 

35 

36 :param store_name The database/data store name 

37 :return: None (Narrative timeline is displayed) 

38 """ 

39 logging.info('Narrative selection') 

40 # Create the GUI Window 

41 narrative_dict = dict() 

42 try: 

43 success, narrative_names = query_database('select', query_narratives, store_name) 

44 if success and 'results' in narrative_names.keys() and \ 

45 'bindings' in narrative_names['results'].keys(): 

46 for binding in narrative_names['results']['bindings']: 

47 narrative_dict[binding['name']['value']] = binding['narrator']['value'].split(':')[-1] 

48 else: 

49 sg.popup_error(f'No narratives are defined in {store_name}. ' 

50 f'Narrative timelines cannot be displayed.', 

51 font=('Arial', 14), button_color='dark blue', icon=encoded_logo) 

52 except Exception as e: 

53 capture_error(f'Exception getting narrative names from {store_name}: {str(e)}', True) 

54 return 

55 if not len(narrative_dict): 

56 sg.popup_ok('No narratives were found in the store. ' 

57 'Please ingest one or more using the "Load Narratives" button.', font=('Arial', 14), 

58 button_color='dark blue', icon=encoded_logo) 

59 return 

60 else: 

61 narrative_list = list(narrative_dict.keys()) 

62 

63 # Setup the PySimpleGUI window 

64 sg.theme('Material2') 

65 layout = [[sg.Text("Select a narrative and then press 'OK'.", font=('Arial', 16))], 

66 [sg.Text("To exit without making a selection, press 'End' or close the window.", 

67 font=('Arial', 16))], 

68 [sg.Listbox(narrative_list, size=(30, 10), key='narrative_list', font=('Arial', 14), 

69 background_color='#fafafa', highlight_background_color='light grey', 

70 highlight_text_color='black', text_color='black')], 

71 [sg.Text()], 

72 [sg.Button('OK', button_color='dark blue', font=('Arial', 14), size=(5, 1)), 

73 sg.Button('End', button_color='dark blue', font=('Arial', 14), size=(5, 1))]] 

74 

75 # Create the GUI Window 

76 window_narrative_list = sg.Window('Select Narrative', layout, icon=encoded_logo).Finalize() 

77 

78 # Event Loop to process window "events" 

79 while True: 

80 event_narrative_list, values = window_narrative_list.read() 

81 if event_narrative_list in (sg.WIN_CLOSED, 'End'): 

82 # If user closes window or clicks 'End' 

83 break 

84 if event_narrative_list == 'OK': 

85 if len(values['narrative_list']) != 1: 

86 sg.popup_error('Either no narrative was selected, or more than one was selected.', 

87 font=('Arial', 14), button_color='dark blue', icon=encoded_logo) 

88 else: 

89 narrative_name = values['narrative_list'][0] 

90 narrative_text = '' 

91 try: 

92 success1, narrative_text_results = query_database( 

93 'select', query_narrative_text.replace('narrative_name', narrative_name), 

94 store_name) 

95 if success1 and 'results' in narrative_text_results.keys() and \ 

96 'bindings' in narrative_text_results['results'].keys(): 

97 narrative_text = narrative_text_results['results']['bindings'][0]['text']['value'] 

98 else: 

99 sg.popup_error(f'Error retrieving the text for the narrative, {narrative_name}, ' 

100 f'from {store_name}. The narrative details cannot be displayed.', 

101 font=('Arial', 14), button_color='dark blue', icon=encoded_logo) 

102 except Exception as e: 

103 capture_error(f'Exception getting narrative text for {narrative_name} ' 

104 f'from {store_name}: {str(e)}', True) 

105 return 

106 if narrative_text: 

107 _display_metadata(narrative_name, narrative_dict[narrative_name], 

108 narrative_text, store_name) 

109 _display_timeline(narrative_name, narrative_text) 

110 

111 # Done 

112 window_narrative_list.close() 

113 return

305def _display_timeline(narrative_name: str, narrative_text: str): 

306 """ 

307 Displays a timeline of the narrative events. 

308 

309 :param narrative_name The name/label of the narrative 

310 :param narrative_text The narrative text (which will not be needed when the event graph is created 

311 in load.py) 

312 :return None (Timeline is displayed) 

313 """ 

314 logging.info(f'Displaying narrative timeline for {narrative_name}') 

315 # TODO: Move following processing to load.py and change to retrieving from DB 

316 sentences = parse_narrative(narrative_text) 

317 key_date = '' 

318 current_loc = '' 

319 orig_dates = set() 

320 date_text_dict = dict() 

321 for sent_dict in sentences: 

322 key_date, current_loc = \ 

323 get_timeline_data(sent_dict, key_date, orig_dates, current_loc, date_text_dict) 

324 # TODO: What if there are no dates? 

325 # Sort the dates and simplify the text 

326 orig_dates = sorted(orig_dates) 

327 names = [] 

328 for orig_date in orig_dates: 

329 date_text = '' 

330 loc_text = date_text_dict[orig_date][0].split('Loc: ')[1].split('\n')[0] 

331 for text in date_text_dict[orig_date]: 

332 text = text.replace(f'Loc: {loc_text}\n', '') 

333 date_text += text 

334 names.append(f'Loc: {loc_text}\n{date_text}') 

335 # For matplotlib timeline, need datetime formatting 

336 dates = [datetime.strptime(d, "%Y") for d in orig_dates] 

337 # Create a stem plot with some variation in levels as to distinguish close-by events. 

338 # Add markers on the baseline for visual emphasis 

339 # For each event, add a text label via annotate, which is offset in units of points from the tip of the event line 

340 levels = np.tile([-5, 5, -3, 3, -1, 1], 

341 int(np.ceil(len(dates) / 6)))[:len(dates)] 

342 # Create figure and plot a stem plot with the date 

343 fig, ax = plt.subplots(figsize=(10, 7)) 

344 ax.set(title="Narrative Timeline") 

345 ax.vlines(dates, 0, levels, color="tab:red") # The vertical stems 

346 ax.plot(dates, np.zeros_like(dates), "-o", 

347 color="k", markerfacecolor="w") # Baseline and markers on it 

348 # Annotate lines 

349 for d, l, r in zip(dates, levels, names): 

350 ax.annotate(r, xy=(d, l), 

351 xytext=(-2, np.sign(l) * 3), textcoords="offset points", 

352 horizontalalignment="right", 

353 verticalalignment="bottom" if l > 0 else "top", fontsize='x-small') 

354 # Format x-axis with yearly intervals 

355 ax.xaxis.set_major_locator(mdates.YearLocator()) 

356 ax.xaxis.set_major_formatter(mdates.DateFormatter("%Y")) 

357 plt.setp(ax.get_xticklabels(), rotation=30, ha="right", fontsize='small') 

358 # Remove y axis and spines 

359 ax.yaxis.set_visible(False) 

360 ax.spines[["left", "top", "right"]].set_visible(False) 

361 ax.margins(y=0.1) 

362 plt.tight_layout() 

363 # Display in a window 

364 layout = [[sg.Canvas(key="-CANVAS-")]] 

365 window_timeline = sg.Window("Narrative Timeline", layout, 

366 icon=encoded_logo, element_justification='center').Finalize() 

367 draw_figure(window_timeline["-CANVAS-"].TKCanvas, fig) 

368 window_timeline.Maximize() 

369 # Non-blocking window 

370 window_timeline.read(timeout=0) 

371 return
