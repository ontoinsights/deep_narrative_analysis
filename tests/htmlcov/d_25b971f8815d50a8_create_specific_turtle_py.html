12def _process_class_mapping(class_map: str, indiv_iri: str, process_verb: bool, is_alt_coll: bool) -> list: 

13 """ 

14 Create the Turtle assigning an individual a type (rdf:type) based on the class mapping string. The 

15 individual is identified by the indiv_iri. The details in the class mapping may require special 

16 handling - such as when a noun map indicates both a Person and an Event (which indicates that 

17 the Person should be associated with an event of that type). 

18 

19 :param class_map: A possible type definition for an individual 

20 :param indiv_iri: A string/IRI identifying the individual 

21 :param process_verb: Boolean indicating that this processing is for a verb mapping 

22 :param is_alt_coll: Boolean indicating that this processing defines a collection of alternative 

23 mappings (an AlternativeCollection) 

24 :return: A list of strings holding the Turtle type declaration 

25 """ 

26 type_turtle = [] 

27 if not process_verb and '+' in class_map and \ 

28 ('dna:Person+' in class_map or 'dna:OrganizationalEntity' in class_map): 

29 for detail in class_map.split('+'): # Class names that define the type, '+' indicates multiple inheritance 

30 if query_class(detail, query_subclass.replace('check', 'LineOfBusiness')): 

31 class_map = class_map.replace(f'+{detail}', empty_string) 

32 class_map = class_map.replace(detail, empty_string) 

33 type_turtle.append( 

34 f'{indiv_iri} :has_line_of_business {detail.replace(dna_prefix, ":")} .') 

35 elif query_class(detail, query_subclass.replace('check', 'EventAndState')): 

36 class_map = class_map.replace(f'+{detail}', empty_string) 

37 class_map = class_map.replace(detail, empty_string) 

38 match_iri = f':PersonEvent_{str(uuid.uuid4())[:13]}' 

39 type_turtle.append( 

40 f'{match_iri} a {detail.replace(dna_prefix, ":")} ; ' 

41 f':has_actor {indiv_iri} .') 

42 if is_alt_coll: 

43 type_turtle.append(f'{indiv_iri} :has_member {class_map.replace("+", ", ")} .') 

44 else: 

45 type_turtle.append(f'{indiv_iri} a {class_map.replace("+", ", ")} .') 

46 return type_turtle

49def cleanup_turtle(turtle: list) -> list: 

50 """ 

51 Use rdflib to examine the Turtle statements, to determine if IRIs (related to sentence objects) 

52 are included, but are never referenced as objects. They are removed since they might be confusing 

53 to a user. 

54 

55 :param turtle: An array of the Turtle statements for a sentence/chunk 

56 :return: The Turtle with any triples with a subject IRI that is not referenced as an object removed 

57 """ 

58 complete_turtle = ttl_prefixes[:] 

59 complete_turtle.extend(turtle) 

60 rdf_graph = RDFGraph() 

61 graph = rdf_graph.parse(data=' '.join(complete_turtle), format='text/turtle') 

62 unused_iris = [] 

63 subjects = [subj for subj in graph.subjects() if subj.startswith('urn:')] # TODO: unique positional arg 

64 objects = [obj for obj in graph.objects() if obj.startswith('urn:')] # TODO: unique positional arg 

65 for subj in subjects: 

66 if subj not in objects: 

67 unused_iris.append(subj) 

68 if unused_iris: 

69 if len(unused_iris) == 1 and ':Sentence' in str(unused_iris[0]): # The Sentence reference is always 'unused' 

70 return turtle 

71 new_turtle = [] 

72 for ttl in turtle: 

73 found_unused = False 

74 for unused in unused_iris: 

75 if ':Sentence_' in unused: 

76 continue 

77 if ttl.startswith(f':{str(unused.split(":")[-1])} '): 

78 found_unused = True 

79 continue 

80 if not found_unused: 

81 new_turtle.append(ttl) 

82 return new_turtle 

83 else: 

84 return turtle

87def create_environment_ttl(text: str, specific_class: str, subjects: list, event_iri: str) -> list: 

88 """ 

89 Return the Turtle for a Person identified as a member of an organization (NORP), as having 

90 an emotion or as involved in a line of business (LoB). 

91 

92 :param text: Specific text from the sentence that was used for the identification 

93 :param specific_class: The most specific class in the DNA ontology to which the text was mapped 

94 :param subjects: Array of tuples that are the subjects' texts, mappings, types and IRIs 

95 :param event_iri: String holding the IRI of the Event being described/processed 

96 :return: Appropriate Turtle statement given the input parameters 

97 """ 

98 new_ttl = [f'{event_iri} a {specific_class} ; rdfs:label "{text}" .'] 

99 for subj_text, subj_type, subj_mappings, subj_iri in subjects: 

100 new_ttl.append(f'{event_iri} :has_holder {subj_iri} .') 

101 return new_ttl

104def create_metadata_ttl(graph_id: str, narr_text: str, created_at: str, 

105 number_triples: int, narr_details: list) -> list: 

106 """ 

107 Return the metadata Turtle for an ingested or updated narrative. 

108 

109 :param graph_id: The ID of the graph where the narrative is stored 

110 :param narr_text: The complete narrative text 

111 :param created_at: The time that the graph was created or updated 

112 :param number_triples: Number of triples in the graph 

113 :param narr_details: The metadata for the narrative - an array of strings indicating the 

114 author, publication date, publisher, source and title of the narrative 

115 :return: An array of Turtle statements 

116 """ 

117 narr_text = narr_text.replace('"', "'") 

118 turtle = [f'@prefix : <urn:ontoinsights:dna:> .', '@prefix dc: <http://purl.org/dc/terms/> .', 

119 f':{graph_id} a :KnowledgeGraph ; dc:created "{created_at}"^^xsd:dateTime ; ' 

120 f':number_triples {number_triples} ; :encodes :Narrative_{graph_id} .', 

121 f':Narrative_{graph_id} a :Narrative ; dc:creator "{narr_details[0]}" ; ' 

122 f'dc:publisher "{narr_details[2]}" ; dc:source "{narr_details[3]}" ; ' 

123 f'dc:title "{narr_details[4]}" .', 

124 f':Narrative_{graph_id} :text "{narr_text}" .'] 

125 if narr_details[1] != 'not defined': 

126 turtle.append(f':Narrative_{graph_id} dc:created "{narr_details[1]}"^^xsd:dateTime .') 

127 return turtle

130def create_quotations_ttl(graph_id: str, quotations: list, quot_dict: dict, 

131 quotations_ttl: list, for_default_graph: bool) -> list: 

132 """ 

133 Return the Turtle related to the quotations in a text - for the data to be added to 

134 the meta_dna database (if for_meta_dna is True) or adding the detailed 'Quotation#' 

135 text for the knowledge graph (if for_meta_dna is False). 

136 

137 :param graph_id: The ID of the graph where the narrative is stored 

138 :param quotations: An array of quotations extracted from the original text 

139 :param quot_dict: An array of quotation dictionaries (key = 'Quotation#' which is referenced 

140 in a chunk, and value = full quotation text) extracted from the original text 

141 :param quotations_ttl: The current list of Turtle statements related to the quotations 

142 :param for_default_graph: A boolean indicating whether this processing is for the default/ 

143 meta-data graph of the repository 

144 :return: An array of Turtle statements 

145 """ 

146 narr_iri = f':Narrative_{graph_id}' 

147 if for_default_graph: 

148 for quotation in quotations: 

149 quotations_ttl.append(f'{narr_iri} :text_quote "{quotation}" .') 

150 else: 

151 for quote_numb, text in quot_dict.items(): 

152 quotations_ttl.append(f':{quote_numb} :text "{text}" .') 

153 return quotations_ttl

156def create_type_turtle(class_mappings: list, subj_iri: str, is_verb: bool, noun_text: str) -> list: 

157 """ 

158 Create the Turtle assigning an individual a type (rdf:type). The individual is identified by the 

159 subj_iri and the type(s) are identified by the class_mappings. Due to a word having multiple 

160 possible meanings (which may be disambiguated in future code drops), a list of possible 

161 class types are provided as input. 

162 

163 :param class_mappings: A list of possible types for an individual 

164 :param subj_iri: A string/IRI identifying the individual 

165 :param is_verb: Boolean indicating that this processing is for a verb mapping 

166 :param noun_text: A string holding the original text for a noun 

167 :return: A list of strings holding the Turtle representation 

168 """ 

169 if len(class_mappings) > 1: 

170 ttl_list = [f'{subj_iri} a :AlternativeCollection ; :text "{noun_text}" .'] 

171 for class_map in class_mappings: 

172 ttl_list.extend(_process_class_mapping(class_map, subj_iri, is_verb, True)) 

173 return ttl_list 

174 else: 

175 return _process_class_mapping(class_mappings[0], subj_iri, is_verb, False)
