98def get_y_x_values(number_narrators: int, variable: str, query: str, store_name: str) -> (tuple, tuple): 

99 """ 

100 Get the count of narrators by gender, birth country and birth year. Also provide counts where 

101 this info is not known. 

102 

103 :param number_narrators: Total number of narrators 

104 :param variable: String indicating the name of the returned SPARQL query variable - either 

105 'gender', 'year' or 'country' 

106 :param query: String holding the query to run to get the count 

107 :param store_name: String holding the database/data store name with the narratives and narrator details 

108 :return: Two tuples of integers - for the y and x axes of a horizontal bar chart ... 

109 The y axis is the list of genders, birth countries and birth years, 

110 and the x axis is the number of narrators 

111 """ 

112 dictionary = dict() 

113 # Get count 

114 count_results = query_database('select', query, store_name) 

115 for binding in count_results: 

116 # Manipulate the key value in case it is an IRI 

117 update_dictionary_count(dictionary, str(binding[variable]['value']).split(':')[-1]) 

118 # Make sure that all narrators are addressed 

119 total_count = 0 

120 y_list = [] 

121 for key in dictionary.keys(): 

122 total_count += dictionary[key] 

123 y_list.append(key) 

124 if total_count != number_narrators: 

125 dictionary['Unknown'] = number_narrators - total_count 

126 y_list.append('Unknown') 

127 # Sort the results and create the returned tuples 

128 y_values = sorted(y_list) 

129 x_values = [] 

130 for y_value in y_values: 

131 x_values.append(dictionary[y_value]) 

132 return tuple(y_values), tuple(x_values)

135def output_words_in_csv(narratives: str, nouns_in_csv: int, verbs_in_csv: int, directory_name: str): 

136 """ 

137 Output the top xx 'unknown to the ontology' nouns and verbs in the narratives' texts 

138 to the files, <directory_name>/Nouns.csv and <directory_name>/Verbs.csv. 

139 The nouns, verbs and a count of their occurrences is returned. 

140 

141 :param narratives: String consisting of all the narratives' texts 

142 :param nouns_in_csv: Integer value indicating the number of nouns to be output 

143 :param verbs_in_csv: Integer value indicating the number of verbs to be output 

144 :param directory_name: String indicating the base file name to which the nouns, verbs and 

145 their frequencies are output 

146 :return: None (.csv files are generated) 

147 """ 

148 sorted_nouns, sorted_verbs = get_nouns_verbs(narratives) 

149 # TODO: Only report words with 'unknown' semantics 

150 with open(f'{directory_name}/Nouns.csv', 'w', newline=new_line) as noun_file: 

151 noun_writer = csv.writer(noun_file, delimiter=',') 

152 noun_writer.writerow(['Noun', 'Count']) 

153 count = 0 

154 for key, value in sorted_nouns.items(): 

155 if count < nouns_in_csv: 

156 count += 1 

157 noun_writer.writerow([key, value]) 

158 else: 

159 break 

160 with open(f'{directory_name}/Verbs.csv', 'w') as verb_file: 

161 verb_writer = csv.writer(verb_file, delimiter=',') 

162 verb_writer.writerow(['Verb', 'Count']) 

163 count = 0 

164 for key, value in sorted_verbs.items(): 

165 if count < verbs_in_csv: 

166 count += 1 

167 verb_writer.writerow([key, value]) 

168 else: 

169 break 

170 return
