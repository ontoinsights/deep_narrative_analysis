11def _check_if_loc_is_known(loc_text: str, loc_type: str, alet_dict: dict) -> (list, str): 

12 """ 

13 Determines if the location is already known/defined in either the geo-names country list or if 

14 it has been already processed (and is therefore in the alet dictionary). 

15 

16 :param loc_text: Input location string 

17 :param loc_type: String identifying NER type for the text, as defined by spaCy 

18 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

19 the full narrative - For co-reference resolution (it may be updated by this function); 

20 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'locs') = array of arrays 

21 with index 0 holding an array of strings/alt names associated with the location, index 1 

22 storing the class mappings, and index 2 defining the location's IRI 

23 :return: A tuple holding an array of class mappings and a string specifying the location IRI, 

24 or a tuple of the location class mappings and an empty string 

25 """ 

26 if loc_text in names_to_geo_dict: # Location is a country name 

27 return [':Country'], f'geo:{names_to_geo_dict[loc_text]}' 

28 class_map = get_agent_or_loc_class(loc_type) 

29 if 'locs' not in alet_dict: 

30 return [class_map], empty_string 

31 known_locs = alet_dict['locs'] 

32 for known_loc in known_locs: 

33 if loc_text in known_loc[0]: # Strings match 

34 return known_loc[1], known_loc[2] # NER maps to a known/processed location, with an IRI 

35 return [class_map], empty_string

38def _get_location_iri_and_ttl(loc_text: str, loc_maps: list, alet_dict: dict, 

39 ext_sources: bool) -> (list, list, str, list): 

40 """ 

41 Get a location IRI based on the input string and if appropriate, add the Turtle explaining/ 

42 defining that location. 

43 

44 :param loc_text: Input location string 

45 :param loc_maps: Array holding the ontology mapping for the NER type from spaCy 

46 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

47 the full narrative - For co-reference resolution (it may be updated by this function); 

48 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'locs') = array of arrays 

49 with index 0 holding an array of strings/alt names associated with the location, index 1 

50 storing the class mappings, and index 2 defining the location's IRI 

51 :param ext_sources: Boolean indicating whether additional information on the location should 

52 be retrieved from GeoNames (recommended) 

53 :return: A tuple holding 1) an array of alternate names for the location, 2) an array with the 

54 location's class mappings, 3) a string with the loc's IRI and 4) a list of Turtle 

55 statements defining the location (if the location is not already 'known'/processed, 

56 the alet_dict is updated) 

57 """ 

58 loc_ttl = [] 

59 loc_iri = re.sub(r'[^:a-zA-Z0-9_]', underscore, f':{loc_text}').replace('__', '_') 

60 # Know that this is identified as a LOC, GPE, ...; See if there is more info 

61 if ext_sources: 

62 geo_ttl, geonames_map, alt_names = create_geonames_ttl(loc_iri, loc_text) 

63 # Class map from geonames is a string 

64 loc_maps = [geonames_map] 

65 loc_ttl.extend(geo_ttl) 

66 else: 

67 alt_names = [loc_text.strip()] 

68 loc_ttl.append(f'{loc_iri} a {loc_maps[0]} ; rdfs:label "{loc_text}" .') # Only have 1 mapping from NER 

69 # Record location text in alet_dict so that the details are not added again 

70 known_locs = alet_dict['locs'] if 'locs' in alet_dict else [] 

71 known_locs.append([alt_names, loc_maps, loc_iri]) 

72 alet_dict['locs'] = known_locs 

73 return alt_names, loc_maps, loc_iri, loc_ttl

76def get_sentence_locations(sentence_dictionary: dict, alet_dict: dict, last_nouns: list, 

77 use_sources: bool) -> (list, list): 

78 """ 

79 Handle locations identified by spaCy's NER, stored in the sentence dictionary with the 

80 key, 'LOCS'. If there is only 1 location specified, it is returned as "the" location 

81 for the events of this sentence. 

82 

83 :param sentence_dictionary: The sentence dictionary 

84 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

85 the full narrative - For co-reference resolution (it may be updated by this function); 

86 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'locs') = array of arrays 

87 with index 0 holding an array of strings/alt names associated with the location, index 1 

88 storing the class mappings, and index 2 defining the location's IRI 

89 :param last_nouns: An array of tuples = noun texts, type, class mapping and IRI 

90 from the current paragraph 

91 :param use_sources: Boolean indicating whether additional information on the location should 

92 be retrieved from GeoNames (recommended) 

93 :return: A tuple consisting of two arrays: (1) the IRIs of the locations identified in the sentence, 

94 and 2) Turtle statements defining them (also the alet_dict may be updated) 

95 """ 

96 locs_turtle = [] 

97 orig_locs = sentence_dictionary['LOCS'] 

98 # Handle location words occurring together - for ex, 'Paris, Texas' - which are returned as 2 separate entities 

99 new_locs = [] 

100 loc_iris = [] 

101 if len(orig_locs) > 1: 

102 sent_str = sentence_dictionary['text'] 

103 offset = 0 

104 while offset < (len(orig_locs) - 1): 

105 if f'{orig_locs[offset]}, {orig_locs[offset + 1]}' in sent_str: 105 ↛ 106line 105 didn't jump to line 106, because the condition on line 105 was never true

106 new_locs.append(f'{orig_locs[offset].split("+")[0]}, {orig_locs[offset + 1].split("+")[0]}' 

107 f'+{orig_locs[offset].split("+")[1]}') 

108 offset += 2 

109 else: 

110 new_locs.append(orig_locs[offset]) 

111 offset += 1 

112 new_locs.append(orig_locs[offset]) 

113 else: 

114 new_locs.append(orig_locs[0]) 

115 new_locs_dict = dict() 

116 for new_loc in new_locs: 

117 loc_split = new_loc.split('+') 

118 new_locs_dict[loc_split[0]] = _check_if_loc_is_known(loc_split[0], loc_split[1], alet_dict) 

119 for new_loc, loc_details in new_locs_dict.items(): 

120 loc_maps, loc_iri = loc_details 

121 if not loc_iri: 

122 # Need to define the Turtle for a new location - Have only the NER mapping for the loc_type from above 

123 alt_names, loc_maps, loc_iri, loc_turtle = \ 

124 _get_location_iri_and_ttl(new_loc, loc_maps, alet_dict, use_sources) 

125 locs_turtle.extend(loc_turtle) 

126 last_nouns.append((new_loc, 'LOC', loc_maps, loc_iri)) 

127 loc_iris.append(loc_iri) 

128 return loc_iris, locs_turtle
