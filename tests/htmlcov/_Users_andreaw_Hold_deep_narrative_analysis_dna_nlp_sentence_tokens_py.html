19def _handle_noun(noun_token: Token) -> (str, str): 

20 """ 

21 When processing a token (in add_token_details) that is a noun, first handle the special 

22 cases of a proper noun, pronoun, family member role, etc. Lastly, check a general noun. 

23 

24 :param noun_token: Token of the noun 

25 :return: Two strings, the noun text and type 

26 """ 

27 noun_text = noun_token.text 

28 nouns = [] 

29 full_noun_text = space.join(t.text for t in noun_token.subtree if 

30 (t.pos_ not in unwanted_tokens and t.dep_ not in unwanted_dependencies)) 

31 # Handle proper nouns and pronouns 

32 if 'PROPN' in noun_token.pos_: # Proper noun 

33 return _process_proper_noun(noun_token) 

34 elif 'Prs' in noun_token.morph.get('PronType'): # Personal pronoun 

35 return _process_personal_pronoun(noun_token) 

36 # Handle references to family members ('brother', 'sister', ...) or 'family' 

37 elif noun_text.lower() in family_roles: 

38 gender = family_members[noun_text.lower()] 

39 return full_noun_text, f'{gender}SINGPERSON' if gender else f'SINGPERSON' 

40 elif noun_text.lower()[0:-1] in family_roles: # Check if plural family members were referenced 

41 gender = family_members[noun_text.lower()[0:-1]] # Check if plural family members were referenced 

42 return full_noun_text, f'{gender}PLURALPERSON' if gender else f'PLURALPERSON' 

43 elif noun_text.lower() in family_text: 

44 return full_noun_text, 'PLURALPERSON' 

45 else: 

46 return _process_noun(noun_token)

109def _process_personal_pronoun(token: Token) -> (str, str): 

110 """ 

111 When processing a token that is a personal pronoun, return the text that identifies 

112 the person and their 'type' (gender + single/plural + PERSON). 

113 

114 :param token: Token of the personal pronoun 

115 :return: A tuple holding the text that identifies the person and their 'type' 

116 (gender + single/plural + PERSON). 

117 """ 

118 entity = token.text 

119 entity_type = token.ent_type_ 

120 if '1' in token.morph.get('Person'): 

121 entity_type = 'PERSON' 

122 if '2' in token.morph.get('Person'): 

123 entity_type = 'AUDIENCE' 

124 elif '3' in token.morph.get('Person'): 

125 if token.text.lower() == 'it': 

126 entity_type = 'NOUN' 

127 else: 

128 entity_type = 'PERSON' 

129 # Include gender and plurality in ent_type 

130 if 'Plur' in token.morph.get('Number'): 

131 entity_type = f'PLURAL{entity_type}' 

132 elif 'Sing' in token.morph.get('Number'): 

133 entity_type = f'SING{entity_type}' 

134 if 'Fem' in token.morph.get('Gender'): 

135 entity_type = f'FEMALE{entity_type}' 

136 elif 'Masc' in token.morph.get('Gender'): 

137 entity_type = f'MALE{entity_type}' 

138 return entity, entity_type

162 """ 

163 Set up a dictionary to hold the entity's/token's details.
