11def _check_if_loc_is_known(loc_text: str, loc_type: str, alet_dict: dict) -> (list, str): 

12 """ 

13 Determines if the location is already known/defined in either the geo-names country list or if 

14 it has been already processed (and is therefore in the alet dictionary). 

15 

16 The acronym, alet, stands for agent-location-event-time. 

17 

18 :param loc_text: Input location string 

19 :param loc_type: String identifying NER type for the text, as defined by spaCy 

20 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

21 the full narrative - For co-reference resolution (it may be updated by this function); 

22 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'locs') = array of arrays 

23 with index 0 holding an array of strings/alt names associated with the location, index 1 

24 storing the class mappings, and index 2 defining the location's IRI 

25 :return: A tuple holding an array of class mappings and a string specifying the location IRI, 

26 or a tuple of the location class mappings and an empty string 

27 """ 

28 if loc_text in names_to_geo_dict: # Location is a country name 

29 return [':Country'], f'geo:{names_to_geo_dict[loc_text]}' 

30 class_map = get_agent_or_loc_class(loc_type) 

31 if 'locs' not in alet_dict: 

32 return [class_map], empty_string 

33 known_locs = alet_dict['locs'] 

34 for known_loc in known_locs: 

35 if loc_text in known_loc[0]: # Strings match 

36 return known_loc[1], known_loc[2] # NER maps to a known/processed location, with an IRI 

37 return [class_map], empty_string

40def _get_location_iri_and_ttl(loc_text: str, loc_maps: list, alet_dict: dict, 

41 ext_sources: bool) -> (list, list, str, list): 

42 """ 

43 Get a location IRI based on the input string and if appropriate, add the Turtle explaining/ 

44 defining that location. 

45 

46 :param loc_text: Input location string 

47 :param loc_maps: Array holding the ontology mapping for the NER type from spaCy 

48 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

49 the full narrative - For co-reference resolution (it may be updated by this function); 

50 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'locs') = array of arrays 

51 with index 0 holding an array of strings/alt names associated with the location, index 1 

52 storing the class mappings, and index 2 defining the location's IRI 

53 :param ext_sources: Boolean indicating whether additional information on the location should 

54 be retrieved from GeoNames (recommended) 

55 :return: A tuple holding 1) an array of alternate names for the location, 2) an array with the 

56 location's class mappings, 3) a string with the loc's IRI and 4) a list of Turtle 

57 statements defining the location (if the location is not already 'known'/processed, 

58 the alet_dict is updated) 

59 """ 

60 loc_ttl = [] 

61 loc_iri = re.sub(r'[^:a-zA-Z0-9_]', underscore, f':{loc_text}').replace('__', '_') 

62 # Know that this is identified as a LOC, GPE, ...; See if there is more info 

63 if ext_sources: 

64 geo_ttl, geonames_map, alt_names = create_geonames_ttl(loc_iri, loc_text) 

65 # Class map from geonames is a string 

66 loc_maps = [geonames_map] 

67 loc_ttl.extend(geo_ttl) 

68 else: 

69 alt_names = [loc_text.strip()] 

70 loc_ttl.append(f'{loc_iri} a {loc_maps[0]} ; rdfs:label "{loc_text}" .') # Only have 1 mapping from NER 

71 # Record location text in alet_dict so that the details are not added again 

72 known_locs = alet_dict['locs'] if 'locs' in alet_dict else [] 

73 known_locs.append([alt_names, loc_maps, loc_iri]) 

74 alet_dict['locs'] = known_locs 

75 return alt_names, loc_maps, loc_iri, loc_ttl

78def get_sentence_locations(sentence_dictionary: dict, alet_dict: dict, last_nouns: list, 

79 use_sources: bool) -> (list, list): 

80 """ 

81 Handle locations identified by spaCy's NER, stored in the sentence dictionary with the 

82 key, 'LOCS'. If there is only 1 location specified, it is returned as "the" location 

83 for the events of this sentence. 

84 

85 :param sentence_dictionary: The sentence dictionary 

86 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

87 the full narrative - For co-reference resolution (it may be updated by this function); 

88 Keys = 'agents', 'locs', 'events', 'times' and Values (for 'locs') = array of arrays 

89 with index 0 holding an array of strings/alt names associated with the location, index 1 

90 storing the class mappings, and index 2 defining the location's IRI 

91 :param last_nouns: An array of tuples = noun texts, type, class mapping and IRI 

92 from the current paragraph 

93 :param use_sources: Boolean indicating whether additional information on the location should 

94 be retrieved from GeoNames (recommended) 

95 :return: A tuple consisting of two arrays: (1) the IRIs of the locations identified in the sentence, 

96 and 2) Turtle statements defining them (also the alet_dict may be updated) 

97 """ 

98 locs_turtle = [] 

99 orig_locs = sentence_dictionary['LOCS'] 

100 # Handle location words occurring together - for ex, 'Paris, Texas' - which are returned as 2 separate entities 

101 new_locs = [] 

102 loc_iris = [] 

103 if len(orig_locs) > 1: 

104 sent_str = sentence_dictionary['text'] 

105 offset = 0 

106 while offset < (len(orig_locs) - 1): 

107 if f'{orig_locs[offset]}, {orig_locs[offset + 1]}' in sent_str: 107 ↛ 108line 107 didn't jump to line 108, because the condition on line 107 was never true

108 new_locs.append(f'{orig_locs[offset].split("+")[0]}, {orig_locs[offset + 1].split("+")[0]}' 

109 f'+{orig_locs[offset].split("+")[1]}') 

110 offset += 2 

111 else: 

112 new_locs.append(orig_locs[offset]) 

113 offset += 1 

114 new_locs.append(orig_locs[offset]) 

115 else: 

116 new_locs.append(orig_locs[0]) 

117 new_locs_dict = dict() 

118 for new_loc in new_locs: 

119 loc_split = new_loc.split('+') 

120 new_locs_dict[loc_split[0]] = _check_if_loc_is_known(loc_split[0], loc_split[1], alet_dict) 

121 for new_loc, loc_details in new_locs_dict.items(): 

122 loc_maps, loc_iri = loc_details 

123 if not loc_iri: 

124 # Need to define the Turtle for a new location - Have only the NER mapping for the loc_type from above 

125 alt_names, loc_maps, loc_iri, loc_turtle = \ 

126 _get_location_iri_and_ttl(new_loc, loc_maps, alet_dict, use_sources) 

127 locs_turtle.extend(loc_turtle) 

128 last_nouns.append((new_loc, 'LOC', loc_maps, loc_iri)) 

129 loc_iris.append(loc_iri) 

130 return loc_iris, locs_turtle
