1# Processes one or more narratives as described in a CSV file 

2# Displays one of two windows to: a) load new narratives from a CSV or b) load narratives already ingested 

3# If new narratives are ingested (a): 

4# 1) The text is loaded and simplified/cleaned 

5# 2) Narrative text and metadata are saved to the specified database/store name 

6# 3) nlp's parse_narrative is called to do the spaCy parsing and create a dictionary of the sentence details 

7# 4) create_event_turtle is called to turn the dictionary into Turtle 

8# 5) The Turtle from (4) is loaded to a named graph identified by the narrative metadata's title 

9# 6) A check for multiple narrator names (due to aliases/maiden and married names/etc.) is performed and 

10# if found, the metadata for these multiple narrators is combined

165def process_csv(csv_file: str, store_name: str, store_list: list) -> int: # pragma: no cover 

166 """ 

167 Input the specified CSV file and process the narratives defined in it. 

168 The format of the CSV MUST be: 

169 Source,Title,Person,Type,Given,Given2,Surname,Maiden,Maiden2,Gender,Start,End,Remove,Header,Footer 

170 

171 @param csv_file: CSV file name 

172 @param store_name: Database/data store name 

173 @param store_list: List of the existing dbs - 

174 Need to determine if a db name is new or existing 

175 @return: Count of the number of narratives ingested 

176 """ 

177 logging.info(f'Processing the CSV, {csv_file}') 

178 count = 0 

179 db_exception = empty_string 

180 if store_name not in store_list: 

181 db_exception = create_delete_database('create', store_name) 

182 if db_exception: 

183 capture_error(f'Error creating or deleting {store_name}: {db_exception}', True) 

184 return 0 

185 try: 

186 with open(csv_file, newline=empty_string) as meta_file: 

187 narr_dict = csv.DictReader(meta_file) 

188 # Process each narrative based on the metadata: 

189 # Source,Title,Person,Type,Given,Given2,Surname,Maiden,Maiden2,Gender,Start,End,Header,Footer 

190 for narr_meta in narr_dict: 

191 if 'Title' not in narr_meta.keys() or 'Given' not in narr_meta.keys(): 

192 capture_error('Expected columns not found in the CSV file. Processing stopped.', False) 

193 title = narr_meta['Title'] 

194 if narr_meta['Type'] != 'T': # TODO: Only process life timelines for now 

195 continue 

196 # TODO: Remove 

197 print(f'Title: {title}') 

198 logging.info(f'Ingesting the document, {title}') 

199 source = narr_meta['Source'] 

200 # Must have at least the Source, Title, Person and Gender values defined 

201 if not source or not title or not narr_meta['Person'] \ 

202 or not narr_meta['Gender']: 

203 sg.popup_error(f'For any source, the Source, Title, Person and Gender details MUST be ' 

204 f'provided. This is not true for the CSV record with source file, ' 

205 f'{source}, and narrative title, {title}. That record is skipped.', 

206 font=('Arial', 14), button_color='dark blue', icon=encoded_logo) 

207 continue 

208 if source.endswith('.pdf'): 

209 # Capture each narrative text from the metadata details in the CSV 

210 if not narr_meta['Start'] and not narr_meta['End']: 

211 sg.popup_error(f'For PDF source files, the Start and End page details MUST be ' 

212 f'provided. This is not true for the CSV record with source file, ' 

213 f'{source}, and narrative title, {title}. That record is skipped.', 

214 font=('Arial', 14), button_color='dark blue', icon=encoded_logo) 

215 continue 

216 # TODO: If the text is multi-page, the first line of all pages but the first is lost 

217 in_file = f'{resources_root}{title}.txt' 

218 subprocess.run(['../tools/pdftotext', '-f', narr_meta['Start'], '-l', narr_meta['End'], 

219 '-simple', f'{resources_root}{source}', in_file]) 

220 else: 

221 in_file = f'{resources_root}{source}' 

222 with open(in_file, 'r', encoding='utf8', errors='ignore') as narr_in: 

223 if source.endswith('.pdf'): 

224 text = clean_text(narr_in.read(), narr_meta) 

225 else: 

226 text = narr_in.read() 

227 narrative = simplify_text(text, narr_meta) 

228 # Get the narrator's gender (if possible), a list of family members mentioned 

229 # (and their proper names, if possible) and the narrative + metadata Turtle 

230 gender, family_dict, turtle_list = \ 

231 create_metadata_turtle(narrative.replace('"', "'"), narr_meta) 

232 # Add the triples to the data store, default graph 

233 try: 

234 add_remove_data('add', ' '.join(turtle_list), store_name) 

235 except Exception as e: 

236 capture_error( 

237 f'Exception adding ({narr_meta["Title"]}) metadata to store: {str(e)}', True) 

238 sentence_dicts = parse_narrative(narrative, gender, family_dict) 

239 print(sentence_dicts) 

240 event_turtle_list = create_event_turtle(gender, sentence_dicts) 

241 print(event_turtle_list) 

242 # Add the triples to the data store, to a named graph with name = metadata title 

243 try: 

244 add_remove_data('add', ' '.join(event_turtle_list), store_name, 

245 f'urn:{title}') 

246 

247 except Exception as e: 

248 capture_error( 

249 f'Exception adding ({narr_meta["Title"]}) events to store: {str(e)}', True) 

250 if source.endswith('.pdf'): 

251 # Cleanup - Delete the text file created by pdftotext 

252 os.remove(in_file) 

253 count += 1 

254 # Determine if any narrators/subjects (different names) are really the same 

255 logging.info('Checking for unification of narrators') 

256 unified_triples = unify_narrators(store_name) 

257 if unified_triples: 

258 # Add the triples to the data store 

259 add_remove_data('add', ' '.join(unified_triples), store_name) 

260 except Exception as e: 

261 traceback.print_exc(file=sys.stdout) 

262 capture_error(f'Exception ingesting narratives: {str(e)}', True) 

263 return count

266def unify_narrators(store_name: str) -> list: 

267 """ 

268 Determine if any of the narrators should be 'unified' (are the same person but have 

269 different names). 

270 

271 @param store_name: The database/data store name 

272 @return: List holding strings of the new triples to add 

273 """ 

274 new_triples = [] 

275 results = query_database('select', query_for_unification, store_name) 

276 for result in results: 

277 narr1 = result['narr1']['value'] 

278 narr2 = result['narr2']['value'] 

279 unify1 = result['unify1']['value'] if 'unify1' in result.keys() else empty_string 

280 unify2 = result['unify2']['value'] if 'unify2' in result.keys() else empty_string 

281 if not unify1 and not unify2: 

282 # Create new UnifyingCollection 

283 iri_collection = f'{narr1}{narr2.split("urn:ontoinsights:dna:")[-1]}' 

284 new_triples.append(f'@prefix : <urn:ontoinsights:dna:> . ' 

285 f'<{iri_collection}> a :UnifyingCollection ; :has_member <{narr1}>, <{narr2}> .') 

286 elif unify1 and unify2 and unify1 != unify2: 

287 # Move unify2 members to unify1 via SPARQL UPDATE and delete unify2 

288 query_database('update', merge_insert.replace('unify1', unify1).replace('unify2', unify2), store_name) 

289 query_database('update', merge_delete.replace('unify2', unify2), store_name) 

290 elif unify1 and not unify2: 

291 # Add narr2 to unify1 

292 new_triples.append(f'@prefix : <urn:ontoinsights:dna:> . <{unify1}> :has_member <{narr2}> .') 

293 elif unify2 and not unify1: 

294 # Add narr1` to unify2 

295 new_triples.append(f'@prefix : <urn:ontoinsights:dna:> . <{unify2}> :has_member <{narr1}> .') 

296 return new_triples
