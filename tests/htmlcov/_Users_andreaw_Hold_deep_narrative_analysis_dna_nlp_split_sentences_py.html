14def get_chunks(verb: Token, connector: list, chunk_sentence: Span, is_conj: bool) -> list: 

15 """ 

16 Given a conjunctive verb, a connector token relating that verb to the root verb 

17 (such as 'and', 'but', ...) and the sentence that contains these tokens, determine if 

18 the sentence can be separated into clauses. Also separate sentences that are two clauses 

19 split by a semi-colon. 

20 

21 @param verb: The token of the conjunctive verb 

22 @param connector: Tokens of the connectors between the root and conjunctive verbs 

23 (if is_conj is true) or adverb modifiers of the conjunctive verb 

24 (if is_conj is false) 

25 @param chunk_sentence: A Span representing the parsed sentence 

26 @param is_conj: A boolean indicating whether the connector is related to terms such as 

27 'and', 'but', 'or' or is related to an adverbial clause 

28 @return: A list of the text of the clauses of the sentence (split if indicated by the 

29 logic below) or just the original sentence returned 

30 """ 

31 chunks = [] 

32 # Is there a subject of the other clause's verb? That is required to split the sentence 

33 # An 'expl' is the word, 'there' 

34 subj2 = [subj for subj in verb.children if ('subj' in subj.dep_ or subj.dep_ == 'expl')] 

35 if len(subj2): 

36 # Yes ... Separate clauses 

37 # Get the tokens in sentence related to the 'other' verb's subtree 

38 seen = [ww for ww in verb.subtree if ww.pos_ != 'PUNCT'] 

39 seen_chunk = ' '.join([ww.text for ww in seen]).strip() + '.' 

40 unseen = [ww for ww in chunk_sentence if ww not in seen and ww.pos_ != 'PUNCT'] 

41 unseen_chunk = ' '.join([ww.text for ww in unseen]).strip() + '.' 

42 for conn in connector: 

43 if is_conj: 

44 # Remove the connector words (to prevent cycles and sentences such as 'Mary biked to the market and.') 

45 unseen_chunk = unseen_chunk.replace(f' {conn.text} ', ' ').replace(f' {conn.text}.', '.') 

46 else: 

47 # Remove the connector words (to prevent cycles) but NOT if there is a single auxiliary verb 

48 # If so, then the advmod is needed 

49 # For example, 'my daughter is home' (is = auxiliary verb, home = adverbial modifier) 

50 if not any([ww for ww in seen if ww.pos_ == 'AUX']) or any([ww for ww in seen if ww.pos_ == 'VERB']): 

51 seen_chunk = _remove_startswith(seen_chunk, conn.text) 

52 seen_chunk = seen_chunk.replace(f' {conn.text} ', ' ').replace(f' {conn.text}.', '.') 

53 if not any([ww for ww in unseen if ww.pos_ == 'AUX']) or any( 

54 [ww for ww in unseen if ww.pos_ == 'VERB']): 

55 unseen_chunk = _remove_startswith(unseen_chunk, conn.text) 

56 # Store the seen and unseen clauses as separate sentences 

57 chunks.append(unseen_chunk) 

58 chunks.append(seen_chunk) 

59 if len(chunks) > 0: 

60 return chunks 

61 else: 

62 return [chunk_sentence.text]

65def split_clauses(sent_text: str, nlp: Language) -> list: 

66 """ 

67 Perform splitting of sentences first by splitting by coordinating conjunctions (calling the function, 

68 split_by_conjunctions) and second by adverbial clauses and clausal complements. 

69 

70 @param sent_text: The sentence to be split (as a string) 

71 @param nlp: A spaCy Language model 

72 @return: A list of new sentences/Spans created from the split 

73 """ 

74 orig_sents = [sent_text] 

75 # Iterate until the sentences cannot be further decomposed/split 

76 while True: 

77 new_sents = [] 

78 # Go through each sentence in the array 

79 for orig_sent in orig_sents: 

80 # Semi-colons automatically split sentences 

81 if '; ' in orig_sent: 

82 semicolon_index = orig_sent.index('; ') 

83 new_sents = [f'{orig_sent[:semicolon_index]}.', f'{orig_sent[semicolon_index + 2:]}.'] 

84 break 

85 # First split by words such as 'and', 'but', 'or', ... related to the 'root' verb 

86 for chunk_sent in nlp(orig_sent).sents: 

87 # nlp(orig_sent) creates a new Document to allow .sents processing of ROOT verbs 

88 # TODO: Capture 'or', 'nor' as the splitting word (e.g., as an alternative) 

89 intermed_sents, splitting_word = _split_by_conjunctions(chunk_sent) 

90 # Split by advcl and xcomp IF these have their own subject/verb 

91 # Example: 'When I went to the store, I met George.' ('when ...' is an adverbial clause) 

92 for intermed_sent in intermed_sents: 

93 for chunk_sent2 in nlp(intermed_sent).sents: 

94 # nlp(intermed_sent) 're-tokenizes' in order to create a new Document, as above 

95 clausal_verbs, connectors = _check_subject_in_clause(chunk_sent2) 

96 # Need to have both a clause and a connector/modifier for this logic 

97 if len(clausal_verbs) > 0: 

98 new_sents.extend(get_chunks(clausal_verbs[0], connectors, chunk_sent2, False)) 

99 else: 

100 new_sents.append(intermed_sent) 

101 # Check if the processing has resulted in new sentence clauses 

102 if len(orig_sents) == len(new_sents): 

103 # If not, break out of the while loop 

104 break 

105 else: 

106 # New clauses, so keep processing 

107 orig_sents = copy.deepcopy(new_sents) 

108 return new_sents

112def _check_subject_in_clause(cls_sentence: Span) -> (list, list): 

113 """ 

114 Return a list of the verb and connector tokens in a sentence/Span IF the sentence 

115 has dependent clauses with their own subject and verb (and a few other conditions are 

116 met as documented below). 

117 

118 @param cls_sentence: The sentence/Span to be analyzed 

119 @return: Two lists - the first with the independent verbs and the second with any connectors 

120 """ 

121 # Returns a list of the verb and connector tokens 

122 # Check for adverbial clauses, which will have the advmod in the clause 

123 clausal_verbs = [child for child in cls_sentence.root.children if child.dep_ == 'advcl'] 

124 connectors = [] 

125 keep_separate_clauses = False 

126 if len(clausal_verbs) > 0: 

127 connectors = [conn for conn in clausal_verbs[0].children if conn.dep_ in ('advmod', 'mark')] 

128 # Only keep the verbs and connectors under certain circumstances 

129 for conn in connectors: 

130 if conn.text.lower() in cause_connectors: 

131 keep_separate_clauses = True 

132 if not keep_separate_clauses: 

133 connectors = [] 

134 # Also check for clausal complement, where we want the advmod (or mark) in the original/root verb clause 

135 new_verbs = [child for child in cls_sentence.root.children if child.dep_ == 'ccomp'] 

136 if new_verbs: 

137 new_connectors = [conn for conn in cls_sentence.root.children if conn.dep_ in ('advmod', 'mark')] 

138 if new_connectors: 

139 keep_separate_clauses = False 

140 for new_conn in new_connectors: 

141 if new_conn.text.lower() in effect_connectors: 

142 keep_separate_clauses = True 

143 if keep_separate_clauses: 

144 clausal_verbs.extend(new_verbs) 

145 connectors.extend(new_connectors) 

146 return clausal_verbs, connectors

149def _remove_startswith(chunk: str, connector: str) -> str: 

150 """ 

151 Returns first string with the second string removed from its start. 

152 

153 @param chunk: String to be updated 

154 @param connector: String to be removed from the beginning of the chunk 

155 @return: Updated 'chunk' string 

156 """ 

157 if chunk.startswith(f'{connector} '): 

158 chunk = chunk[len(connector) + 1:] 

159 elif chunk.startswith(f' {connector} '): 

160 chunk = chunk[len(connector) + 2:] 

161 return chunk

164def _split_by_conjunctions(conj_sentence: Span) -> (list, str): 

165 """ 

166 Split the clauses of a sentence into separate sentences, if they are connected by a coordinating 

167 conjunction (for, and, but, or, nor, yet, so). For example,"Mary and John walked to the store." would 

168 NOT be split, but "Mary biked to the market and John walked to the store." should be split. 

169 

170 @param conj_sentence The sentence which is analyzed 

171 @return: An array of simpler sentence(s) (note that only the text of the sentences is 

172 returned, not the spacy tokens) and the coordinating conjunction. The latter is 

173 needed to process or/nor alternatives. 

174 """ 

175 conj_sents = [] 

176 conj_verb = [child for child in conj_sentence.root.children if child.dep_ == 'conj'] 

177 connectors = [conn for conn in conj_sentence.root.children if conn.dep_ == 'cc'] 

178 if len(conj_verb) > 0: 

179 for chunk in get_chunks(conj_verb[0], connectors, conj_sentence, True): 

180 conj_sents.append(str(chunk)) 

181 else: 

182 conj_sents = [conj_sentence.text] 

183 return conj_sents, ('' if not connectors else connectors[0].text)
