113def create_ttl_for_prep_detail(prep_detail: tuple, prepositions: list, event_iri: str, alet_dict: dict,

115 """

127 :param last_events: A list of all event types and their IRIs from the current paragraph

191def handle_xcomp_processing(xcomp_iri: str, root_iri: str, xcomp_dict: dict, xcomp_incorporates_root: bool, 

192 turtle: list) -> list: 

193 """ 

194 Cleans up the Turtle for a chunk with a root-xcomp pair. 

195 

196 :param xcomp_iri: String holding the IRI of the xcomp verb 

197 :param root_iri: String holding the IRI of the root verb 

198 :param xcomp_dict: A dictionary with keys = root verb and the xcomp verb and values = 'root' for 

199 the root verb or the verb_processing string for the xcomp 

200 :param xcomp_incorporates_root: A boolean indicating that the semantics of the root verb were collapsed 

201 into the xcomp verb (via multiple inheritance) 

202 :param turtle: An array of the current Turtle statements for a chunk 

203 :return: An array of updated Turtle statements 

204 """ 

205 new_ttl = [] 

206 if xcomp_incorporates_root: 

207 # Only need to fix the label to incorporate the root verb 

208 for turtle_stmt in turtle: 

209 if 'rdfs:label' in turtle_stmt: 

210 updated = False 

211 label = turtle_stmt.split('rdfs:label "')[1].split('" .')[0] 

212 for verb, processing in xcomp_dict.items(): 

213 if 'xcomp >' in processing and verb in label: 

214 root_text = processing.split('/')[1].split(',')[0] 

215 if f' {verb} ' not in label: 

216 label = label.replace(verb, f'{root_text} to {verb} / {verb}') # TODO: Non-English, 'to' 

217 else: 

218 label = label.replace(verb, f'{root_text} to {verb}') 

219 new_ttl.append(f'{xcomp_iri} rdfs:label "{label}" .') 

220 updated = True 

221 break 

222 if not updated: 

223 new_ttl.append(turtle_stmt) 

224 else: 

225 new_ttl.append(turtle_stmt) 

226 return new_ttl 

227 new_ttl = [f'{root_iri} :has_topic {xcomp_iri} .'] 

228 # Correct the time and location in the existing Turtle - Details apply to the root verb and not the xcomp 

229 for turtle_stmt in turtle: 

230 if ':has_latest_end' in turtle_stmt or ':has_earliest_beginning' in turtle_stmt or ':has_time' in turtle_stmt: 

231 new_ttl.append(f'{root_iri} :has_{turtle_stmt.split(":has_")[1]}') 

232 elif has_location in turtle_stmt: 

233 new_ttl.append(f'{root_iri} :has_{turtle_stmt.split(":has_")[1]}') 

234 else: 

235 new_ttl.append(turtle_stmt) 

236 # Also, the xcomp event may not occur (e.g., 'she was asked to leave' - but she might not leave) 

237 new_ttl.append(f'{xcomp_iri} a :OpportunityAndPossibility .') 

238 return new_ttl
