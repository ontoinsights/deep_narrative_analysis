87def construct_database(construct: str, database: str) -> (bool, list): 

88 """ 

89 Process a CONSTRUCT query 

90 

91 :param construct: The text of the CONSTRUCT query 

92 :param database: The database (name) to be queried 

93 :return: A tuple holding a boolean indicating success (if true) or failure and an array 

94 with the Turtle results of the CONSTRUCT query or an error message 

95 """ 

96 try: 

97 conn = stardog.Connection(database, **sd_conn_details) 

98 construct_results = conn.graph(construct, content_type='text/turtle') 

99 turtle_details = rdf_graph.parse(format='text/turtle', data=construct_results) 

100 final_turtle = [] 

101 for stmt in turtle_details: 

102 subj, pred, obj = stmt 

103 final_turtle.append(f'{subj.n3()} {pred.n3()} {obj.n3()} .\n') 

104 return True, final_turtle 

105 except Exception as e: 

106 error = f'Database ({database}) construct exception: {str(e)}' 

107 logging.error(error) 

108 return False, [str(e)]
