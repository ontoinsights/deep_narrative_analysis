47def _process_noun(token: Token) -> (str, str): 

48 """ 

49 When processing a token (in add_token_details) that is a noun, capture its number, gender, etc. 

50 

51 :param token: Token of the noun 

52 :return: Two strings, the entity text and type 

53 """ 

54 # Retrieve the full text with adjectives, etc. 

55 ent = space.join(t.text for t in token.children if (t.dep_ in ('amod', 'compound', 'det', 'nummod', 'nmod') 

56 and t.pos_ not in unwanted_tokens)) 

57 symbols = [t.text for t in token.children if t.pos_ == 'SYM'] 

58 if ent: 

59 ent = f'{ent}{space}{token.text}' 

60 else: 

61 ent = token.text 

62 ent_type = token.ent_type_ if token.ent_type_ else 'NOUN' 

63 # Include plurality in ent_type 

64 if 'Plur' in token.morph.get('Number'): 

65 ent_type = f'PLURAL{ent_type}' 

66 elif 'Sing' in token.morph.get('Number'): 

67 ent_type = f'SING{ent_type}' 

68 gender = empty_string 

69 if 'Fem' in token.morph.get('Gender'): 69 ↛ 70line 69 didn't jump to line 70, because the condition on line 69 was never true

70 gender = 'FEMALE' 

71 elif 'Masc' in token.morph.get('Gender'): 71 ↛ 72line 71 didn't jump to line 72, because the condition on line 71 was never true

72 gender = 'MALE' 

73 ent_type = f'{gender}{ent_type}' if gender else ent_type 

74 # Account for a determiner of 'no' - i.e., nothing (for ex, 'no information was found')

76 ent_type = f'NEG{ent_type}'

111def _process_personal_pronoun(token: Token) -> (str, str): 

112 """ 

113 When processing a token that is a personal pronoun, return the text that identifies 

114 the person and their 'type' (gender + single/plural + PERSON). 

115 

116 :param token: Token of the personal pronoun 

117 :return: A tuple holding the text that identifies the person and their 'type'

119 """

129 else: 

130 entity_type = 'PERSON' 

131 # Include gender and plurality in ent_type

133 entity_type = f'PLURAL{entity_type}'

135 entity_type = f'SING{entity_type}' 

136 if 'Fem' in token.morph.get('Gender'): 

137 entity_type = f'FEMALE{entity_type}' 

138 elif 'Masc' in token.morph.get('Gender'): 

139 entity_type = f'MALE{entity_type}'
