1# Processing to create the Turtle rendering of the sentences in a narrative 

2# The sentences are defined by their sentence dictionaries, which have the form: 

3# 'text': 'narrative_text', 'sentence_offset': #, 

4# 'AGENTS': ['agent1', 'agent2', ...], 'LOCS': ['location1', ...], 

5# 'TIMES': ['dates_or_times1', ...], 'EVENTS': ['event1', ...], 

6# 'chunks': [chunk1, chunk2, ...] 

7# Note that 'punct': '?' may also be included above 

8# Where each chunk is a dictionary with the form: 

9# 'chunk_text': 'chunk_text', 'verb_processing': 'xcomp_or_prt_details', 

10# 'verbs': [{'verb_text': 'verb_text', 'verb_lemma': 'verb_lemma', 'tense': 'tense_such_as_Past', 

11# 'subject_text': 'subject_text', 'subject_type': 'type_such_as_SINGNOUN', 

12# 'objects': [{'object_text': 'verb_object_text', 'object_type': 'type_eg_NOUN'}] 

13# 'preps': [{'prep_text': 'preposition_text', 

14# 'prep_details': [{'detail_text': 'preposition_object', 'detail_type': 'type_eg_SINGGPE'}]}] 

15# There may be more than 1 verb when there is a root verb + an xcomp

73def _process_chunk(chunk_dict: dict, chunk_iri: str, offset: int, alet_dict: dict, loc_time_iris: list, 

74 last_nouns: list, last_events: list, ext_sources: bool, is_bio: bool) -> list: 

75 """ 

76 Generate the Turtle for each chunk in a sentence, from the chunk's dictionary definition. 

77 

78 :param chunk_dict: The dictionary for the chunk 

79 :param chunk_iri: The IRI identifier for the chunk 

80 :param offset: Integer indicating the ordering of the chunk within the sentence 

81 :param alet_dict: A dictionary holding the agents, locations, events & times encountered in 

82 the full narrative - For co-reference resolution (it may be updated by this function); 

83 Keys = 'agents', 'locs', 'events', 'times' and Values vary by the key 

84 :param loc_time_iris: An array holding the last processed location (index 0) and time (index 1) 

85 :param last_nouns: An array of noun texts, type, class mapping and IRI from the current paragraph 

86 :param last_events: An array of verb texts, class mapping and IRI from the current paragraph 

87 :param ext_sources: A boolean indicating that data from GeoNames, Wikidata, etc. should be 

88 added to the parse results if available 

89 :param is_bio: A boolean indicating that the text is biographical/autobiographical 

90 :return: An array of Turtle statements defining the chunk's knowledge graph 

91 """ 

92 # TODO: Deal with '$&', special mark, in text 

93 chunk_text = chunk_dict['chunk_text'] 

94 chunk_ttl_list = [f'{chunk_iri} a :Chunk ; :offset {offset} .', 

95 f'{chunk_iri} :text "{chunk_text}" .', 

96 f'{chunk_iri} :sentiment {_get_text_sentiment(chunk_text, empty_string, empty_string)} .'] 

97 if chunk_text.startswith('Quotation'): 

98 return chunk_ttl_list 

99 new_ttl_list = process_verb(chunk_iri, chunk_dict, alet_dict, loc_time_iris, last_nouns, 

100 last_events, ext_sources, is_bio) 

101 if new_ttl_list: 101 ↛ 104line 101 didn't jump to line 104, because the condition on line 101 was never false

102 chunk_ttl_list.extend(new_ttl_list)

104 return []

107def _process_family(family_dict: dict) -> list: 

108 """ 

109 Process the entries in family_dict and create entries to add to the alet_dictionary as 

110 new 'agents'. 

111 

112 :return: An array of arrays with index 0 holding an array of labels associated with the agent (variations 

113 on their name), index 1 storing the agent's entity type and index 2 storing the agent's IRI 

114 """ 

115 unique_names = [] 

116 for name1 in family_dict.keys(): 

117 found = False 

118 for name2 in family_dict.keys(): 

119 if name1 != name2 and name1 in name2 and name2.startswith(name1): # Part of name overlaps 

120 unique_names.append(name2) # Save the longer name 

121 found = True 

122 break 

123 if found: 

124 continue 

125 unique_names.append(name1) # Longer name or no overlap - so save name1 

126 agents_list = [] 

127 for unique_name in unique_names: 

128 family_role = family_dict[unique_name] 

129 role_gender = family_members[family_role] 

130 alt_names = get_name_permutations(unique_name) 

131 alt_names.append(family_role) 

132 agents_list.append((alt_names, f'{role_gender}SINGPERSON' if role_gender else 'SINGPERSON', 

133 re.sub(r'[^:a-zA-Z0-9_]', underscore, f':{unique_name}_{family_role}').replace('__', '_'))) 

134 return agents_list
